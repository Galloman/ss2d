 13 /**
 14  * A Bitmap font component for Glyph Designer XML .fnt output files.
 15  * @see http://www.71squared.com/en/glyphdesigner
 16  * @constructor
 17  * @param {string} bitmapFilename The resource name
 18  * @param {function} callbackFunction Called when the BitmapFont is loaded
 19  */
 20 ss2d.BitmapFont = function(bitmapFilename, callbackFunction)
 21 {
 22 	this.mName = bitmapFilename;
 23 	this.mCallbackFunction = callbackFunction;
 24 	this.mFontDescriptor = null;
 25 	this.mTexture = null; 
 26 	this.mMidWidth = 0;
 27 	this.mMidHeight = 0;
 28 	
 29 	var fontFileRequest = new XMLHttpRequest();
 30 	fontFileRequest.mBitmapFont = this;
 31 	fontFileRequest.open("GET", bitmapFilename, true);
 32 	fontFileRequest.responseType = "text";
 33 	fontFileRequest.onload = function() 
 34 	{
 35 		this.mBitmapFont.fontFileLoaded(this.response);
 36 	};
 37 	
 38 	fontFileRequest.send();
 39 };
 40 
 41 /**
 42  * @private
 43  * @param {string} fileData
 44  */
 45 ss2d.BitmapFont.prototype.fontFileLoaded = function(fileData)
 46 {
 47 	this.mFontDescriptor = (new XML.ObjTree()).parseXML(fileData);
 48 	/** @type {Array} */
 49 	var charArray = this.mFontDescriptor['font']['chars']['char'];
 50 	var totalWidths = 0;
 51 	var totalHeights = 0;
 52 	var indexedChars = {};
 53 	for(var c = 0; c < charArray.length; ++c)
 54 	{
 55 		var charObj = charArray[c];
 56 		totalWidths += parseInt(charObj['-xadvance']);
 57 		totalHeights += parseInt(charObj['-height']);
 58 		indexedChars[charArray[c]['-id']] = charObj;
 59 		
 60 		charObj['-x'] = parseInt(charObj['-x']);
 61 		charObj['-xoffset'] = parseInt(charObj['-xoffset']);
 62 		charObj['-y'] = parseInt(charObj['-y']);
 63 		charObj['-width'] = parseInt(charObj['-width']);
 64 		charObj['-height'] = parseInt(charObj['-height']);
 65 		charObj['-xadvance'] = parseInt(charObj['-xadvance']);
 66 		charObj['-yoffset'] = parseInt(charObj['-yoffset']);
 67 	}
 68 	this.mFontDescriptor.mIndexedChars = indexedChars;
 69 	this.mMidWidth = totalWidths / charArray.length;
 70 	this.mMidHeight = totalHeights / charArray.length;
 71 	
 72 	var texturePageList = this.mFontDescriptor['font']['pages']['page'];
 73 	var texturePage = null;
 74 	if(texturePage instanceof Array)
 75 	{
 76 		texturePage = texturePageList[0]['-file'];
 77 	}
 78 	else
 79 	{
 80 		texturePage = texturePageList['-file'];
 81 	}
 82 	
 83 	var pathEnd = this.mName.lastIndexOf('/') + 1;
 84 	texturePage = (pathEnd > 0)?this.mName.substring(0, pathEnd)+texturePage:texturePage;
 85 	
 86 	this.mTexture = new ss2d.Texture(texturePage, this.fontTextureLoaded, this);
 87 };

 89 /**
 90  * Called once the texture atlas is loaded
 91  */
 92 ss2d.BitmapFont.prototype.fontTextureLoaded = function(texture)
 93 {
 94 	this.mTexture = texture;
 95 	if(this.mCallbackFunction)
 96 		this.mCallbackFunction.call(this.mCallbackTarget, this);
 97 }

 99 /** 
100  * Retrieve the clip frame information for a given character
101  * @param {number|string} charCode
102  * @param {number[4]=} targetClipArray Optional array where the clipping information will be copied
103  * @return {number[4]} The clip array
104  */
105 ss2d.BitmapFont.prototype.getGlyphClip = function(charCode, targetClipArray)
106 {
107 	targetClipArray = targetClipArray||[];
108 	
109 	var charDesc = this.mFontDescriptor.mIndexedChars[charCode];
110 	
111 	if(charDesc)
112 	{
113 		targetClipArray[0] = charDesc['-x'];
114 		targetClipArray[1] = charDesc['-y'];
115 		targetClipArray[2] = charDesc['-width'];
116 		targetClipArray[3] = charDesc['-height'];
117 		targetClipArray[4] = charDesc['-xadvance'];
118 		targetClipArray[5] = charDesc['-yoffset'];
119 	}
120 	
121 	return targetClipArray;
122 };
