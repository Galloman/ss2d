  8 goog.provide('ss2d.materials.Particle');
  9 
 10 goog.require('ss2d.ShaderSource');
 11 goog.require('ss2d.ShaderProgram');
 12 
 13 ss2d.materials.Particle = function(support)
 14 {
 15 	var gl = support.mContext;
 16 	var vertexShader = new ss2d.ShaderSource().compileSource(ss2d.materials.Particle.VERTEX_SOURCE, gl.VERTEX_SHADER);
 17 	var fragmentShader = new ss2d.ShaderSource().compileSource(ss2d.materials.Particle.FRAGMENT_SOURCE, gl.FRAGMENT_SHADER);
 18 	
 19 	this.mShaderProgram = new ss2d.ShaderProgram([vertexShader, fragmentShader], 
 20 												 ['uPMatrix', 'uScaleMatrix', 'uSampler'], 
 21 												 ['aVertexPosition', 'aColor']);				 
 22 	this.mActiveTexture = 0;
 23 	this.mParticleDataBuffer = null;
 24 	this.mParticleDataArray = new ArrayBuffer();
 25 	this.mAuxF32Matrix = new Float32Array(9);
 26 };
 27 
 28 ss2d.materials.Particle.prototype.apply = function(support)
 29 {
 30 	var gl = support.mContext;
 31 	
 32 	this.mShaderProgram.use(gl);
 33 
 34 	support.activeTextureForSampler(this.mActiveTexture, this.mShaderProgram.mUniforms['uSampler'], gl.TEXTURE0);
 35 	
 36 	gl.bindBuffer(gl.ARRAY_BUFFER, this.mParticleDataBuffer);
 37 	gl.bufferSubData(gl.ARRAY_BUFFER, 0, this.mParticleDataArray);
 38 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aVertexPosition'], 2, gl.FLOAT, false, 24, 0);
 39 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aColor'], 4, gl.FLOAT, false, 24, 8);
 40 	
 41 	gl.uniformMatrix3fv(this.mShaderProgram.mUniforms['uPMatrix'], false, ss2d.CURRENT_VIEW.mProjection.getMatF32Array(this.mAuxF32Matrix));
 42 };
 43 
 44 ss2d.materials.Particle.VERTEX_SOURCE = ''+
 45 	'attribute vec2 aVertexPosition;'+
 46 	'attribute vec4 aColor;'+
 47 	'uniform mat3 uScaleMatrix;'+
 48 	'uniform mat3 uPMatrix;'+
 49 	'varying vec4 vColor;'+
 50 	'varying vec2 vTextureCoord;'+
 51 	'void main(void){'+
 52 		'vColor = aColor;'+
 53 		'vTextureCoord = aVertexPosition;'+
 54 		'gl_Position = vec4((uPMatrix * vec3(aVertexPosition, 1.0)).xy, 0.0, 1.0);'+
 55 	'}';
 56 
 57 ss2d.materials.Particle.FRAGMENT_SOURCE = ''+
 58 	'precision mediump float;'+
 59 	'uniform sampler2D uSampler;'+
 60 	'varying vec2 vTextureCoord;'+
 61 	'varying vec4 vColor;'+
 62 	'void main(void){'+
 63 		'vec4 color = texture2D(uSampler, vTextureCoord.xy);'+
 64 		'gl_FragData[0] = color * vColor;'+
 65 	'}';
