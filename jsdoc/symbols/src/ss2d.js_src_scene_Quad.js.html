 13 /**
 14  * @constructor
 15  * @extends {ss2d.DisplayObject}
 16  * @param {number} x
 17  * @param {number} y
 18  * @param {number} w The width
 19  * @param {number} h The height
 20  * @param {(number|string|Array.<number>|ss2d.Color)} color
 21  */
 22 ss2d.Quad = function(x, y, w, h, color)
 23 {
 24 	ss2d.DisplayObject.call(this, x, y, 1.0, 0.0, color, 1.0);
 25 	
 26 	this.mWidth = w || 10.0;
 27 	this.mHeight = h || 10.0;
 28 	
 29 	//if(RENDER_CONTEXT == 'webgl')
 30 	//{
 31 
 32 	//}
 33 };
 34 
 35 goog.inherits(ss2d.Quad, ss2d.DisplayObject);
 36 
 37 ss2d.Object.assignClassId(ss2d.Quad, 1003);
 38 
 39 if(COMPILING_CLIENT||COMPILING_OFFLINE)
 40 {
 41 	/** @override */
 42 	ss2d.Quad.prototype.render = function(renderSupport)
 43 	{
 44 		//if(RENDER_CONTEXT == 'webgl')
 45 		//{
 46 			
 47 		//}
 48 		//else
 49 		//{
 50 			renderSupport.pushTransform(this);
 51 			var ctx = renderSupport.mContext;
 52 			ctx.fillStyle = this.mColor.getHexString();
 53 			ctx.rect(0, 0, this.mWidth, this.mHeight);
 54 			renderSupport.popTransform(); 
 55 		//}
 56 	};
 57 }

 59 /** @override */
 60 ss2d.Quad.prototype.getWidth = function(v)

 63 };
 64 
 65 /** @override */
 66 ss2d.Quad.prototype.getHeight = function(v)

 68 	return this.mHeight; 
 69 };
 70 
 71 /** @override */
 72 ss2d.Quad.prototype.setWidth = function(w, excludeClip)
 73 { 
 74 	this.mWidth = w;
 75 };
 76 
 77 /** @override */
 78 ss2d.Quad.prototype.setHeight = function(h, excludeClip)
 79 { 
 80 	this.mHeight = h;
 81 };
 82 
 83 /** @override */
 84 ss2d.Quad.prototype.getBounds = function()
 85 {
 86 	return new ss2d.Rectangle(this.mLocation.mX, this.mLocation.mY, this.getWidth(), this.getHeight());
 87 };

 93 if(COMPILING_CLIENT)
 94 {
 95 	/** @override */
 96 	ss2d.Quad.convert = function(obj)
 97 	{
 98 		var objBackup =  ss2d.Object.backupAndDeleteObjectProperties(obj);
 99 		
100 		ss2d.Quad.call(obj);
101 		
102 		obj.__proto__ = ss2d.Quad.prototype;
103 		
104 		//now obj is a Quad
105 		obj.restoreSerializedProperties(objBackup);
106 		
107 		return obj;
108 	}
109 	
110 	/** @override */
111 	ss2d.Quad.prototype.restoreSerializedProperties = function(objBackup)
112 	{
113 		ss2d.DisplayObject.prototype.restoreSerializedProperties.call(this, objBackup);
114 		
115 		this.mWidth = objBackup['w'] || this.mWidth;
116 		this.mHeight = objBackup['h'] || this.mHeight;
117 	}
118 	
119 	/** @override */
120 	ss2d.Quad.createFromSerializedObject = function(obj)
121 	{
122 		var newObj = new ss2d.Quad(obj['x'], obj['y'], obj['w'], obj['h'], obj['c']);
123 		newObj.mObjectId = obj['doid'];
124 		return newObj;
125 	}
126 	
127 	/** @override */
128 	ss2d.Quad.prototype.interpolateState = function(prevState, nextState, part, deltaTime)
129 	{
130 		ss2d.DisplayObject.prototype.interpolateState.call(this, prevState, nextState, part, deltaTime);
131 		this.mWidth = prevState['w'] + (nextState['w'] - prevState['w']) * part;
132 		this.mHeight = prevState['h'] + (nextState['h'] - prevState['h']) * part;
133 	}
134 }
