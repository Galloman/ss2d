 12 /**
 13  * @constructor
 14  * @extends {ss2d.Sprite}
 15  * @param {ss2d.Bone} masterBone The master bone of this "slave" slot.
 16  * @param {number=} x Position in x axis 
 17  * @param {number=} y Position in y axis
 18  * @param {number} width The width
 19  * @param {number} height The height
 20  * @param {string|ss2d.Texture} texture The name or reference of a texture file or the name of a frame in a texture atlas.
 21  * @param {string|ss2d.TextureAtlas=} textureAtlas The name or reference of a texture atlas file.
 22  */
 23 ss2d.Slot = function(masterBone, name, x, y, w, h, texture, textureAtlas)
 24 {
 25 	ss2d.Sprite.call(this, x, y, w, h, texture, textureAtlas);
 26 	this.mName = name;
 27 	this.mBone = masterBone;
 28 	this.mAuxBoneMatrix = new ss2d.Matrix3();
 29 };
 30 
 31 goog.inherits(ss2d.Slot, ss2d.Sprite)
 32 
 33 /** 
 34  * Fakes the world transformation matrix to have it's Bone as parent.
 35  * @override
 36  */
 37 ss2d.Slot.prototype.getWorldTransformationMatrix = function(targetMatrix, upToParent, includeHimself)
 38 {
 39 	var trueParent = this.mParent;
 40 	this.mParent = this.mBone;
 41 	var worldMatrix = ss2d.DisplayObject.prototype.getWorldTransformationMatrix.call(this, targetMatrix, upToParent, includeHimself);
 42 	this.mParent = trueParent;
 43 	return worldMatrix;
 44 };

 46 //2d context require override, webgl no.
 47 if(RENDER_CONTEXT != 'webgl')
 48 {
 49 	/** @override */
 50 	ss2d.Slot.prototype.render = function(support)
 51 	{
 52 		this.mAuxBoneMatrix.identity();
 53 		this.mBone.getWorldTransformationMatrix(this.mAuxBoneMatrix, this.mParent.mParent, true);
 54 		support.pushTransform(this.mAuxBoneMatrix);
 55 		ss2d.Sprite.prototype.render.call(this, support);
 56 		support.popTransform();
 57 	};
 58 }

 60 ss2d.Slot.prototype.tick = function(deltaTime)
 61 {
 62 	var parent = this.mBone.mParentSkeletalSprite;
 63 	if(!parent.mCurrentAnimation)
 64 		return;
 65 		
 66 	var animation = parent.mCurrentAnimation.mAnimationData;	
 67 	if(!animation || !animation['slots'] || !animation['slots'][this.mName])
 68 		return;
 69 	
 70 	var attachments = animation['slots'][this.mName]['attachment'];
 71 	if(!attachments)
 72 		return;
 73 		
 74 	var currentTime = parent.mCurrentAnimationTime;
 75 
 76 	var nextAttachment = null;
 77 	for(var ai = 0; ai < attachments.length && nextAttachment == null; ai++)
 78 	{
 79 		if(attachments[ai]['time'] > currentTime)
 80 		{
 81 			ai = (ai > 0) ? ai - 1 : attachments.length - 1;
 82 			nextAttachment = attachments[ai]['name'];
 83 			break;
 84 		}
 85 	}
 86 	
 87 	if(!nextAttachment)
 88 		return;
 89 		
 90 	this.setTexture(nextAttachment, this.mTextureAtlas);
 91 	
 92 	//update slot info
 93 	var skin = null;
 94 	try { skin = parent.mSkeleton.mSkeletonData['skins'][parent.mCurrentSkin]; } catch(t){}
 95 	if(!skin) 
 96 		return;
 97 	
 98 	var skinInfo = skin[this.mName][nextAttachment];
 99 	this.mLocation.mX = skinInfo['x']||0;
100 	this.mLocation.mY = skinInfo['y']||0;
101 	this.mRotation = skinInfo['rotation']||0;
102 	this.mWidth = skinInfo['width']||1
103 	this.mHeight = skinInfo['height']||1
104 	this.mPivotX = this.mWidth*0.5;
105 	this.mPivotY = this.mHeight*0.5;
106 };
