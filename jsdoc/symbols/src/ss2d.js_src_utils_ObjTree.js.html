 26 XML.ObjTree.prototype.parseXML = function ( xml ) {
 27     var root;
 28     if ( window.DOMParser ) {
 29         var xmldom = new DOMParser();
 30 //      xmldom.async = false;           // DOMParser is always sync-mode
 31         var dom = xmldom.parseFromString( xml, "application/xml" );
 32         if ( ! dom ) return;
 33         root = dom.documentElement;
 34     } else if ( window.ActiveXObject ) {
 35         xmldom = new ActiveXObject('Microsoft.XMLDOM');
 36         xmldom.async = false;
 37         xmldom.loadXML( xml );
 38         root = xmldom.documentElement;
 39     }
 40     if ( ! root ) return;
 41     return this.parseDOM( root );
 42 };

 44 //  method: parseHTTP( url, options, callback )
 45 
 46 XML.ObjTree.prototype.parseHTTP = function ( url, options, callback ) {
 47     var myopt = {};
 48     for( var key in options ) {
 49         myopt[key] = options[key];                  // copy object
 50     }
 51     if ( ! myopt.method ) {
 52         if ( typeof(myopt.postBody) == "undefined" &&
 53              typeof(myopt.postbody) == "undefined" &&
 54              typeof(myopt.parameters) == "undefined" ) {
 55             myopt.method = "get";
 56         } else {
 57             myopt.method = "post";
 58         }
 59     }
 60     if ( callback ) {
 61         myopt.asynchronous = true;                  // async-mode
 62         var __this = this;
 63         var __func = callback;
 64         var __save = myopt.onComplete;
 65         myopt.onComplete = function ( trans ) {
 66             var tree;
 67             if ( trans && trans.responseXML && trans.responseXML.documentElement ) {
 68                 tree = __this.parseDOM( trans.responseXML.documentElement );
 69             }
 70             __func( tree, trans );
 71             if ( __save ) __save( trans );
 72         };
 73     } else {
 74         myopt.asynchronous = false;                 // sync-mode
 75     }
 76     var trans;
 77     if ( typeof(HTTP) != "undefined" && HTTP.Request ) {
 78         myopt.uri = url;
 79         var req = new HTTP.Request( myopt );        // JSAN
 80         if ( req ) trans = req.transport;
 81     } else if ( typeof(Ajax) != "undefined" && Ajax.Request ) {
 82         var req = new Ajax.Request( url, myopt );   // ptorotype.js
 83         if ( req ) trans = req.transport;
 84     }
 85     if ( callback ) return trans;
 86     if ( trans && trans.responseXML && trans.responseXML.documentElement ) {
 87         return this.parseDOM( trans.responseXML.documentElement );
 88     }
 89 }

 91 //  method: parseDOM( documentroot )
 92 
 93 XML.ObjTree.prototype.parseDOM = function ( root ) {
 94     if ( ! root ) return;
 95 
 96     this.__force_array = {};
 97     if ( this.force_array ) {
 98         for( var i=0; i<this.force_array.length; i++ ) {
 99             this.__force_array[this.force_array[i]] = 1;
100         }
101     }
102 
103     var json = this.parseElement( root );   // parse root node
104     if ( this.__force_array[root.nodeName] ) {
105         json = [ json ];
106     }
107     if ( root.nodeType != 11 ) {            // DOCUMENT_FRAGMENT_NODE
108         var tmp = {};
109         tmp[root.nodeName] = json;          // root nodeName
110         json = tmp;
111     }
112     return json;
113 };

115 //  method: parseElement( element )
116 
117 XML.ObjTree.prototype.parseElement = function ( elem ) {
118     //  COMMENT_NODE
119     if ( elem.nodeType == 7 ) {
120         return;
121     }
122 
123     //  TEXT_NODE CDATA_SECTION_NODE
124     if ( elem.nodeType == 3 || elem.nodeType == 4 ) {
125         var bool = elem.nodeValue.match( /[^\x00-\x20]/ );
126         if ( bool == null ) return;     // ignore white spaces
127         return elem.nodeValue;
128     }
129 
130     var retval;
131     var cnt = {};
132 
133     //  parse attributes
134     if ( elem.attributes && elem.attributes.length ) {
135         retval = {};
136         for ( var i=0; i<elem.attributes.length; i++ ) {
137             var key = elem.attributes[i].nodeName;
138             if ( typeof(key) != "string" ) continue;
139             var val = elem.attributes[i].nodeValue;
140             if ( ! val ) continue;
141             key = this.attr_prefix + key;
142             if ( typeof(cnt[key]) == "undefined" ) cnt[key] = 0;
143             cnt[key] ++;
144             this.addNode( retval, key, cnt[key], val );
145         }
146     }
147 
148     //  parse child nodes (recursive)
149     if ( elem.childNodes && elem.childNodes.length ) {
150         var textonly = true;
151         if ( retval ) textonly = false;        // some attributes exists
152         for ( var i=0; i<elem.childNodes.length && textonly; i++ ) {
153             var ntype = elem.childNodes[i].nodeType;
154             if ( ntype == 3 || ntype == 4 ) continue;
155             textonly = false;
156         }
157         if ( textonly ) {
158             if ( ! retval ) retval = "";
159             for ( var i=0; i<elem.childNodes.length; i++ ) {
160                 retval += elem.childNodes[i].nodeValue;
161             }
162         } else {
163             if ( ! retval ) retval = {};
164             for ( var i=0; i<elem.childNodes.length; i++ ) {
165                 var key = elem.childNodes[i].nodeName;
166                 if ( typeof(key) != "string" ) continue;
167                 var val = this.parseElement( elem.childNodes[i] );
168                 if ( ! val ) continue;
169                 if ( typeof(cnt[key]) == "undefined" ) cnt[key] = 0;
170                 cnt[key] ++;
171                 this.addNode( retval, key, cnt[key], val );
172             }
173         }
174     }
175     return retval;
176 };
177 
178 //  method: addNode( hash, key, count, value )
179 
180 XML.ObjTree.prototype.addNode = function ( hash, key, cnts, val ) {
181     if ( this.__force_array[key] ) {
182         if ( cnts == 1 ) hash[key] = [];
183         hash[key][hash[key].length] = val;      // push
184     } else if ( cnts == 1 ) {                   // 1st sibling
185         hash[key] = val;
186     } else if ( cnts == 2 ) {                   // 2nd sibling
187         hash[key] = [ hash[key], val ];
188     } else {                                    // 3rd sibling and more
189         hash[key][hash[key].length] = val;
190     }
191 };

195 XML.ObjTree.prototype.writeXML = function ( tree ) {
196     var xml = this.hash_to_xml( null, tree );
197     return this.xmlDecl + xml;
198 };

200 //  method: hash_to_xml( tagName, tree )
201 
202 XML.ObjTree.prototype.hash_to_xml = function ( name, tree ) {
203     var elem = [];
204     var attr = [];
205     for( var key in tree ) {
206         if ( ! tree.hasOwnProperty(key) ) continue;
207         var val = tree[key];
208         if ( key.charAt(0) != this.attr_prefix ) {
209             if ( typeof(val) == "undefined" || val == null ) {
210                 elem[elem.length] = "<"+key+" />";
211             } else if ( typeof(val) == "object" && val.constructor == Array ) {
212                 elem[elem.length] = this.array_to_xml( key, val );
213             } else if ( typeof(val) == "object" ) {
214                 elem[elem.length] = this.hash_to_xml( key, val );
215             } else {
216                 elem[elem.length] = this.scalar_to_xml( key, val );
217             }
218         } else {
219             attr[attr.length] = " "+(key.substring(1))+'="'+(this.xml_escape( val ))+'"';
220         }
221     }
222     var jattr = attr.join("");
223     var jelem = elem.join("");
224     if ( typeof(name) == "undefined" || name == null ) {
225         // no tag
226     } else if ( elem.length > 0 ) {
227         if ( jelem.match( /\n/ )) {
228             jelem = "<"+name+jattr+">\n"+jelem+"</"+name+">\n";
229         } else {
230             jelem = "<"+name+jattr+">"  +jelem+"</"+name+">\n";
231         }
232     } else {
233         jelem = "<"+name+jattr+" />\n";
234     }
235     return jelem;
236 };
237 
238 //  method: array_to_xml( tagName, array )
239 
240 XML.ObjTree.prototype.array_to_xml = function ( name, array ) {
241     var out = [];
242     for( var i=0; i<array.length; i++ ) {
243         var val = array[i];
244         if ( typeof(val) == "undefined" || val == null ) {
245             out[out.length] = "<"+name+" />";
246         } else if ( typeof(val) == "object" && val.constructor == Array ) {
247             out[out.length] = this.array_to_xml( name, val );
248         } else if ( typeof(val) == "object" ) {
249             out[out.length] = this.hash_to_xml( name, val );
250         } else {
251             out[out.length] = this.scalar_to_xml( name, val );
252         }
253     }
254     return out.join("");
255 };
256 
257 //  method: scalar_to_xml( tagName, text )
258 
259 XML.ObjTree.prototype.scalar_to_xml = function ( name, text ) {
260     if ( name == "#text" ) {
261         return this.xml_escape(text);
262     } else {
263         return "<"+name+">"+this.xml_escape(text)+"</"+name+">\n";
264     }
265 };
266 
267 //  method: xml_escape( text )
268 
269 XML.ObjTree.prototype.xml_escape = function ( text ) {
270     return (text + '').replace(/&/g,'&').replace(/</g,'<').replace(/>/g,'>').replace(/"/g,'"');
271 };

276 =head1 NAME
277 
278 XML.ObjTree -- XML source code from/to JavaScript object like E4X
279 
280 =head1 SYNOPSIS
281 
282     var xotree = new XML.ObjTree();
283     var tree1 = {
284         root: {
285             node: "Hello, World!"
286         }
287     };
288     var xml1 = xotree.writeXML( tree1 );        // object tree to XML source
289     alert( "xml1: "+xml1 );
290 
291     var xml2 = '<?xml version="1.0"?><response><error>0</error></response>';
292     var tree2 = xotree.parseXML( xml2 );        // XML source to object tree
293     alert( "error: "+tree2.response.error );
294 
295 =head1 DESCRIPTION
296 
297 XML.ObjTree class is a parser/generater between XML source code
298 and JavaScript object like E4X, ECMAScript for XML.
299 This is a JavaScript version of the XML::TreePP module for Perl.
300 This also works as a wrapper for XMLHTTPRequest and successor to JKL.ParseXML class
301 when this is used with prototype.js or JSAN's HTTP.Request class.
302 
303 =head2 JavaScript object tree format
304 
305 A sample XML source:
306 
307     <?xml version="1.0" encoding="UTF-8"?>
308     <family name="Kawasaki">
309         <father>Yasuhisa</father>
310         <mother>Chizuko</mother>
311         <children>
312             <girl>Shiori</girl>
313             <boy>Yusuke</boy>
314             <boy>Kairi</boy>
315         </children>
316     </family>
317 
318 Its JavaScript object tree like JSON/E4X:
319 
320     {
321         'family': {
322             '-name':    'Kawasaki',
323             'father':   'Yasuhisa',
324             'mother':   'Chizuko',
325             'children': {
326                 'girl': 'Shiori'
327                 'boy': [
328                     'Yusuke',
329                     'Kairi'
330                 ]
331             }
332         }
333     };
334 
335 Each elements are parsed into objects:
336 
337     tree.family.father;             # the father's given name.
338 
339 Prefix '-' is inserted before every attributes' name.
340 
341     tree.family["-name"];           # this family's family name
342 
343 A array is used because this family has two boys.
344 
345     tree.family.children.boy[0];    # first boy's name
346     tree.family.children.boy[1];    # second boy's name
347     tree.family.children.girl;      # (girl has no other sisiters)
348 
349 =head1 METHODS
350 
351 =head2 xotree = new XML.ObjTree()
352 
353 This constructor method returns a new XML.ObjTree object.
354 
355 =head2 xotree.force_array = [ "rdf:li", "item", "-xmlns" ];
356 
357 This property allows you to specify a list of element names
358 which should always be forced into an array representation.
359 The default value is null, it means that context of the elements
360 will determine to make array or to keep it scalar.
361 
362 =head2 xotree.attr_prefix = '@';
363 
364 This property allows you to specify a prefix character which is
365 inserted before each attribute names.
366 Instead of default prefix '-', E4X-style prefix '@' is also available.
367 The default character is '-'.
368 Or set '@' to access attribute values like E4X, ECMAScript for XML.
369 The length of attr_prefix must be just one character and not be empty.
370 
371 =head2 tree = xotree.parseXML( xmlsrc );
372 
373 This method loads an XML document using the supplied string
374 and returns its JavaScript object converted.
375 
376 =head2 tree = xotree.parseDOM( domnode );
377 
378 This method parses a DOM tree (ex. responseXML.documentElement)
379 and returns its JavaScript object converted.
380 
381 =head2 tree = xotree.parseHTTP( url, options );
382 
383 This method loads a XML file from remote web server
384 and returns its JavaScript object converted.
385 XMLHTTPRequest's synchronous mode is always used.
386 This mode blocks the process until the response is completed.
387 
388 First argument is a XML file's URL
389 which must exist in the same domain as parent HTML file's.
390 Cross-domain loading is not available for security reasons.
391 
392 Second argument is options' object which can contains some parameters:
393 method, postBody, parameters, onLoading, etc.
394 
395 This method requires JSAN's L<HTTP.Request> class or prototype.js's Ajax.Request class.
396 
397 =head2 xotree.parseHTTP( url, options, callback );
398 
399 If a callback function is set as third argument,
400 XMLHTTPRequest's asynchronous mode is used.
401 
402 This mode calls a callback function with XML file's JavaScript object converted
403 after the response is completed.
404 
405 =head2 xmlsrc = xotree.writeXML( tree );
406 
407 This method parses a JavaScript object tree
408 and returns its XML source generated.
409 
410 =head1 EXAMPLES
411 
412 =head2 Text node and attributes
413 
414 If a element has both of a text node and attributes
415 or both of a text node and other child nodes,
416 text node's value is moved to a special node named "#text".
417 
418     var xotree = new XML.ObjTree();
419     var xmlsrc = '<span class="author">Kawasaki Yusuke</span>';
420     var tree = xotree.parseXML( xmlsrc );
421     var class = tree.span["-class"];        # attribute
422     var name  = tree.span["#text"];         # text node
423 
424 =head2 parseHTTP() method with HTTP-GET and sync-mode
425 
426 HTTP/Request.js or prototype.js must be loaded before calling this method.
427 
428     var xotree = new XML.ObjTree();
429     var url = "http://example.com/index.html";
430     var tree = xotree.parseHTTP( url );
431     xotree.attr_prefix = '@';                   // E4X-style
432     alert( tree.html["@lang"] );
433 
434 This code shows C<lang=""> attribute from a X-HTML source code.
435 
436 =head2 parseHTTP() method with HTTP-POST and async-mode
437 
438 Third argument is a callback function which is called on onComplete.
439 
440     var xotree = new XML.ObjTree();
441     var url = "http://example.com/mt-tb.cgi";
442     var opts = {
443         postBody:   "title=...&excerpt=...&url=...&blog_name=..."
444     };
445     var func = function ( tree ) {
446         alert( tree.response.error );
447     };
448     xotree.parseHTTP( url, opts, func );
449 
450 This code send a trackback ping and shows its response code.
451 
452 =head2 Simple RSS reader
453 
454 This is a RSS reader which loads RDF file and displays all items.
455 
456     var xotree = new XML.ObjTree();
457     xotree.force_array = [ "rdf:li", "item" ];
458     var url = "http://example.com/news-rdf.xml";
459     var func = function( tree ) {
460         var elem = document.getElementById("rss_here");
461         for( var i=0; i<tree["rdf:RDF"].item.length; i++ ) {
462             var divtag = document.createElement( "div" );
463             var atag = document.createElement( "a" );
464             atag.href = tree["rdf:RDF"].item[i].link;
465             var title = tree["rdf:RDF"].item[i].title;
466             var tnode = document.createTextNode( title );
467             atag.appendChild( tnode );
468             divtag.appendChild( atag );
469             elem.appendChild( divtag );
470         }
471     };
472     xotree.parseHTTP( url, {}, func );
473 
474 =head2  XML-RPC using writeXML, prototype.js and parseDOM
475 
476 If you wish to use prototype.js's Ajax.Request class by yourself:
477 
478     var xotree = new XML.ObjTree();
479     var reqtree = {
480         methodCall: {
481             methodName: "weblogUpdates.ping",
482             params: {
483                 param: [
484                     { value: "Kawa.net xp top page" },  // 1st param
485                     { value: "http://www.kawa.net/" }   // 2nd param
486                 ]
487             }
488         }
489     };
490     var reqxml = xotree.writeXML( reqtree );       // JS-Object to XML code
491     var url = "http://example.com/xmlrpc";
492     var func = function( req ) {
493         var resdom = req.responseXML.documentElement;
494         xotree.force_array = [ "member" ];
495         var restree = xotree.parseDOM( resdom );   // XML-DOM to JS-Object
496         alert( restree.methodResponse.params.param.value.struct.member[0].value.string );
497     };
498     var opt = {
499         method:         "post",
500         postBody:       reqxml,
501         asynchronous:   true,
502         onComplete:     func
503     };
504     new Ajax.Request( url, opt );
