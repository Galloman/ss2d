  8 goog.provide('ss2d.ServerView');
  9 
 10 goog.require('ss2d.ServerCommunicationInterface');
 11 
 12 /**
 13  * @constructor
 14  * @param {ss2d.MultiplayerScene=} mainScene
 15  * @param {number=} serverPort
 16  * @param {number=} updateRate Number of world states per second that will be updated and broadcasted to all clients.
 17  */
 18 ss2d.ServerView = function(mainScene, serverPort, updateRate)
 19 {
 20 	this.mRunning = false;
 21 	this.mFrameRate = updateRate || 20.0;
 22 	this.mLastFrameTimestamp = new Date().getTime();
 23 	
 24 	ss2d.CURRENT_VIEW = this;
 25 	
 26 	this.mMainScene = mainScene || new ss2d.DisplayObjectContainer();
 27 	this.mMainScene.mParentView = this;
 28 	
 29 	this.mComunication = new ss2d.ServerCommunicationInterface(this, serverPort);
 30 	
 31 	//sounds that must be send to the client on the next command list, once sent, the sound list is empty
 32 	this.mSoundList = [];
 33 	//this.mDeletedObjectList = [];
 34 };
 35 
 36 /** @type {function} */
 37 ss2d.ServerView.NEXT_FRAME_CALLER = function(){ ss2d.CURRENT_VIEW.nextFrame() };
 38 
 39 /**
 40  * Called every frame to update the scene elements state and broadcast scene to clients.
 41  */
 42 ss2d.ServerView.prototype.nextFrame = function()
 43 {
 44 	//get current time in miliseconds
 45 	var now = new Date().getTime();
 46 	var timePassed = now - this.mLastFrameTimestamp;
 47 	
 48 	//update scene
 49 	this.mMainScene.tick(timePassed/1000.0);
 50 
 51 	//calculate the delay time for the nextFrame call based on the 
 52 	//frameRate and time spend in update and render operations.
 53 	this.mLastFrameTimestamp = now;
 54 	var wastedTime = (new Date().getTime() - now) / 1000.0;
 55 	var nextFrameTime = (1.0/this.mFrameRate) - wastedTime;
 56 	
 57 	//if game is running call the next frame with the specified time
 58 	if(this.mRunning)
 59 	{ 
 60 		setTimeout(ss2d.ServerView.NEXT_FRAME_CALLER, nextFrameTime * 1000.0);
 61 	}
 62 };
 63 
 64 /**
 65  * Start the main loop if it's ready.
 66  */
 67 ss2d.ServerView.prototype.startLoop = function()
 68 {
 69 	if(!this.mRunning && this.mMainScene != null)
 70 	{
 71 		this.mRunning = true;
 72 		this.nextFrame();
 73 	}	
 74 };
 75 
 76 /**
 77  * Pause the loop
 78  */
 79 ss2d.ServerView.prototype.stopLoop = function()
 80 {
 81 	this.mRunning = false;	
 82 };
