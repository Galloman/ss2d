 13 /** @constructor */
 14 ss2d.materials.GPUParticle = function(support)
 15 {
 16 	var gl = support.mContext;
 17 	var vertexShader = new ss2d.ShaderSource().compileSource(ss2d.materials.GPUParticle.VERTEX_SOURCE, gl.VERTEX_SHADER);
 18 	var fragmentShader = new ss2d.ShaderSource().compileSource(ss2d.materials.GPUParticle.FRAGMENT_SOURCE, gl.FRAGMENT_SHADER);
 19 	
 20 	this.mShaderProgram = new ss2d.ShaderProgram([vertexShader, fragmentShader], 
 21 												 ['uPMatrix', 'uTotalTime', 'uEmitterType', 'uGravity', 'uDeltaTime'], 
 22 												 ['aVertexPosition', 'aPos_Dir', 'aStartPos', 'aColor',
 23 												  'aDeltaColor', 'aRo_RoD_RaA_TaA', 'aRa_RaD_A_DPS', 'aPS_PSD_T_TS']);				 
 24 	this.mActiveTexture = 0;
 25 	this.mParticleDataBuffer = null;
 26 	this.mEmitterType = 0;
 27 	this.mGravity = [];
 28 	
 29 	this.mAuxF32Matrix = new Float32Array(9);
 30 	
 31 
 32 
 33 	
 34 };
 35 
 36 ss2d.materials.GPUParticle.prototype.updateParticle = function(buffer, data, offset)
 37 {
 38 	gl.bindBuffer(gl.ARRAY_BUFFER, support.mBuffers.mParticlesData);
 39 	gl.bufferSubData(gl.ARRAY_BUFFER, offset * 4 * 25, data);
 40 };
 41 
 42 ss2d.materials.GPUParticle.prototype.apply = function(support)
 43 {
 44 	var gl = support.mContext;
 45 	
 46 	this.mShaderProgram.use(gl);
 47 
 48 	support.activeTextureForSampler(this.mActiveTexture, this.mShaderProgram.mUniforms['uSampler'], gl.TEXTURE0);
 49 	
 50 	gl.bindBuffer(gl.ARRAY_BUFFER, support.mBuffers.mParticlesQVP);
 51 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aVertexPosition'], support.mBuffers.mParticlesQVP.itemSize, gl.FLOAT, false, 0, 0);
 52 	
 53 	gl.uniform1f(this.mShaderProgram.mUniforms['uEmitterType'], this.mEmitterType);
 54 	gl.uniform2fv(this.mShaderProgram.mUniforms['uGravity'], this.mGravity);
 55 	gl.uniform1f(this.mShaderProgram.mUniforms['uTotalTime'], ss2d.CURRENT_VIEW.mTotalTime);
 56 	gl.uniformMatrix3fv(this.mShaderProgram.mUniforms['uPMatrix'], false, ss2d.CURRENT_VIEW.mProjection.getMatF32Array(this.mAuxF32Matrix));
 57 
 58 	gl.bindBuffer(gl.ARRAY_BUFFER, this.mParticleDataBuffer);
 59 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aPos_Dir'], 		4, gl.FLOAT, false, 26 * 4, 0);
 60 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aStartPos'], 		2, gl.FLOAT, false, 26 * 4, 4 * 4); 
 61 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aColor'], 			4, gl.FLOAT, false, 26 * 4, 6 * 4);
 62 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aDeltaColor'], 		4, gl.FLOAT, false, 26 * 4, 10 * 4); 
 63 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aRo_RoD_RaA_TaA'], 	4, gl.FLOAT, false, 26 * 4, 14 * 4); 
 64 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aRa_RaD_A_DPS'], 	4, gl.FLOAT, false, 26 * 4, 18 * 4); 
 65 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aPS_PSD_T_TS'], 	4, gl.FLOAT, false, 26 * 4, 22 * 4);
 66 
 67 };

 69 ss2d.materials.GPUParticle.VERTEX_SOURCE = ''+
 70 	'const float EMITTER_TYPE_GRAVITY = 0.0;'+
 71 	'const float EMITTER_TYPE_RADIAL = 1.0;'+
 72 	'attribute vec2 aVertexPosition;'+
 73 	'attribute vec4 aPos_Dir;'+
 74 	'attribute vec2 aStartPos;'+
 75 	'attribute vec4 aColor;'+
 76 	'attribute vec4 aDeltaColor;'+
 77 	'attribute vec4 aRo_RoD_RaA_TaA;'+
 78 	'attribute vec4 aRa_RaD_A_DPS;'+
 79 	'attribute vec4 aPS_PSD_T_TS;'+
 80 	'uniform mat3 uPMatrix;'+
 81 	'uniform float uTotalTime;'+
 82 	'uniform float uDeltaTime;'+
 83 	'uniform float uEmitterType;'+
 84 	'uniform vec2 uGravity;'+
 85 	'varying vec4 vColor;'+
 86 	'varying vec2 vTextureCoord;'+
 87 	'void main(void){'+
 88 		
 89 		'float timeLapse = (uTotalTime - aPS_PSD_T_TS.w);'+
 90 		'float sizeToZero = 1.0; if(timeLapse > aPS_PSD_T_TS.z){ sizeToZero = 0.0; }'+
 91 		'vec2 pos = vec2(aPos_Dir.x, aPos_Dir.y);'+
 92 		'float angle = aRa_RaD_A_DPS.y;'+
 93 		'float scaleMultiplier = 1.0;'+
 94 		'if(uEmitterType == EMITTER_TYPE_RADIAL){'+
 95 			'angle += aRa_RaD_A_DPS.w * timeLapse;'+
 96 			'pos += vec2(-cos(angle) * aRa_RaD_A_DPS.x, -sin(angle) * aRa_RaD_A_DPS.x);'+
 97 		'} else {'+
 98 			'vec2 tmp = vec2(0.0, 0.0);'+
 99 			'vec2 radial = vec2(0.0, 0.0);'+
100 			'vec2 tangential = vec2(0.0, 0.0);'+
101 			
102 			'vec2 diff = aStartPos - radial;'+
103 			'pos -= diff;'+
104 			
105 			'if(pos.x != 0.0 || pos.y != 0.0){'+
106 				'radial = normalize(pos);'+
107 			'}'+
108 			'tangential = vec2(radial.x, radial.y);'+
109 			'radial *= aRo_RoD_RaA_TaA.z;'+
110 			
111 			'float newy = tangential.x;'+
112 			'tangential.x = -tangential.y;'+
113 			'tangential.y = newy;'+
114 			'tangential *= aRo_RoD_RaA_TaA.w;'+
115 			
116 			'vec2 direction = (radial + tangential + uGravity) * timeLapse;'+
117 			'pos += ((vec2(aPos_Dir.z, aPos_Dir.w) * timeLapse) + direction);'+
118 			'pos += diff;'+
119 
120 		'}'+
121 		
122 		'vColor = aColor + (aDeltaColor * timeLapse);'+
123 		'float size = aPS_PSD_T_TS.x + (aPS_PSD_T_TS.y * timeLapse);'+
124 		
125 		//TODO: rotation
126 		'size *= scaleMultiplier;'+
127 		'vTextureCoord = aVertexPosition;'+
128 		//- vec2(size * scaleMultiplier * 0.5, size * scaleMultiplier * 0.5)
129 		'gl_Position = vec4((uPMatrix * vec3((aVertexPosition * size * sizeToZero) + pos - vec2(size * scaleMultiplier * 0.5, size * scaleMultiplier * 0.5), 1.0)).xy, 0.0, 1.0);'+
130 	'}';

132 ss2d.materials.GPUParticle.FRAGMENT_SOURCE = ''+
133 	'precision mediump float;'+
134 	'uniform sampler2D uSampler;'+
135 	'varying vec2 vTextureCoord;'+
136 	'varying vec4 vColor;'+
137 	'void main(void){'+
138 		'vec4 color = texture2D(uSampler, vTextureCoord.xy);'+
139 		'gl_FragData[0] = color * vColor;'+
140 	'}';
