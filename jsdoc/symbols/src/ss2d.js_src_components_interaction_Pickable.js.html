  3 /**
  4  * @fileoverview Behaviour component that gives pickable interaction to a DisplayObject.
  5  * @author David Gallardo Moreno (portalg@gmail.com)
  6  */
  7 
  8 goog.provide('ss2d.Pickable');
  9 
 10 /**
 11  * @constructor
 12  * @param {ss2d.DisplayObject} owner The object that owns this component
 13  */
 14 ss2d.Pickable = function(owner)
 15 {
 16 	if(!owner)
 17 	{
 18 		throw "Pickable component need an owner";
 19 	}
 20 	
 21 	this.mOwner = owner||null;
 22 	this.mPicked = false;
 23 	this.mOffset = new ss2d.Point();
 24 };
 25 
 26 ss2d.Pickable.PICKED_COMPONENT = null;
 27 
 28 /**
 29  * Called on every frame by the owner with the delta time
 30  * @param {number} deltaTime
 31  */
 32 ss2d.Pickable.prototype.tick = function(deltaTime)
 33 {
 34 	var input = ss2d.CURRENT_VIEW.mInput;
 35 	if(input.mMouseDown && !input.mPreviousMouseDown)
 36 	{
 37 		//picking
 38 		if(this.mOwner.hitTestPoint(input.mMousePoint))
 39 		{
 40 			//last elements are near in z
 41 			if(ss2d.Pickable.PICKED_COMPONENT)
 42 			{
 43 				ss2d.Pickable.PICKED_COMPONENT.mPicked = false;
 44 			}
 45 			
 46 			ss2d.Pickable.PICKED_COMPONENT = this;
 47 			this.mPicked = true;
 48 			var localMouse = this.mOwner.worldToLocal(input.mMousePoint);
 49 			this.mOffset.set(this.mOwner.mLocation.mX - localMouse.mX, 
 50 							 this.mOwner.mLocation.mY - localMouse.mY);
 51 		}
 52 	}
 53 	
 54 	this.mPicked = (ss2d.Pickable.PICKED_COMPONENT == this) && input.mMouseDown;
 55 	
 56 	if(this.mPicked)
 57 	{
 58 		var localMouse = this.mOwner.worldToLocal(input.mMousePoint);
 59 		//var prevLocalMouse = this.mOwner.worldToLocal(input.mPreviousMousePoint);
 60 		//var mouseMovement = ss2d.Point.subtractPoints(prevLocalMouse, localMouse);
 61 		//this.mOwner.mLocation.mX += mouseMovement.mX;
 62 		//this.mOwner.mLocation.mY += mouseMovement.mY;
 63 		this.mOwner.mLocation.mX = localMouse.mX + this.mOffset.mX;
 64 		this.mOwner.mLocation.mY = localMouse.mY + this.mOffset.mY;
 65 	}
 66 	else
 67 	{
 68 		if(ss2d.Pickable.PICKED_COMPONENT == this)
 69 		{
 70 			ss2d.Pickable.PICKED_COMPONENT = null;
 71 		}
 72 	}
 73 	
 74 	
 75 };
