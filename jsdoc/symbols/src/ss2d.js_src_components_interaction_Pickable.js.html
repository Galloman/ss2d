  3 /**
  4  * @fileoverview Behaviour component that gives pickable interaction to a DisplayObject.
  5  * @author David Gallardo Moreno (portalg@gmail.com)
  6  */
  7 
  8 goog.provide('ss2d.Pickable');
  9 
 10 /**
 11  * @constructor
 12  * @param {ss2d.DisplayObject} owner The object that owns this component
 13  */
 14 ss2d.Pickable = function(owner, rb)
 15 {
 16 	if(!owner)
 17 	{
 18 		throw "Pickable component need an owner";
 19 	}
 20 	
 21 	this.mOwner = owner||null;
 22 	this.mPicked = false;
 23 	this.mOffset = new ss2d.Point();
 24 	this.mMoveObject = true;
 25 	this.mRigidBody = rb||null;
 26 	this.mPreviousPicked = false;
 27 	this.mJustPicked = false;
 28 	this.mMouseJoint = null;
 29 	this.mInput = null;
 30 };
 31 
 32 //ss2d.Pickable.PICKED_COMPONENT = null;
 33 
 34 /**
 35  * Called on every frame by the owner with the delta time
 36  * @param {number} deltaTime
 37  * @param {ss2d.InputProxy} playerInput Input of one connected player (multiplayer feature)
 38  */
 39 ss2d.Pickable.prototype.tick = function(deltaTime, playerInput)
 40 {
 41 	var input = playerInput||ss2d.CURRENT_VIEW.mInput;
 42 	this.mJustPicked = false;
 43 	if(input.mMouseDown && !input.mPreviousMouseDown)
 44 	{
 45 		//picking
 46 		if(this.mOwner.hitTestPoint(input.mMousePoint))
 47 		{	
 48 			if(this.mInput)
 49 			{	
 50 				if(this.mRigidBody)
 51 				{
 52 					s2d.PhysicalWorld.getWorld().mWorld.DestroyJoint(this.mMouseJoint);
 53 				}
 54 				this.mMouseJoint  = null;
 55 				this.mInput.PICKED_COMPONENT = null;
 56 				this.mInput = null;
 57 			}
 58 			
 59 			//last elements are near in z
 60 			if(input.PICKED_COMPONENT)
 61 			{
 62 				if(this.mRigidBody && this.mMouseJoint)
 63 				{
 64 					ss2d.PhysicalWorld.getWorld().mWorld.DestroyJoint(this.mMouseJoint);
 65 					this.mMouseJoint  = null;
 66 					this.mInput = null;
 67 				}
 68 
 69 				input.PICKED_COMPONENT.mPicked = false;
 70 			}
 71 			
 72 			
 73 			
 74 			this.mJustPicked = true;
 75 			this.mPicked = true;
 76 			this.mInput = input;
 77 			input.PICKED_COMPONENT = this;
 78 			var localMouse = this.mOwner.worldToLocal(input.mMousePoint);
 79 			this.mOffset.set(this.mOwner.mLocation.mX - localMouse.mX, 
 80 							 this.mOwner.mLocation.mY - localMouse.mY);
 81 							 
 82 			if(this.mRigidBody)
 83 			{
 84 				this.mMouseJoint = ss2d.PhysicalWorld.getWorld().createMouseJoint(this.mRigidBody, this.mInput);
 85     			this.mRigidBody.mBody.WakeUp();
 86 			}
 87 		}
 88 	}
 89 	
 90 	this.mPicked = (input.PICKED_COMPONENT == this) && this.mInput && this.mInput.mMouseDown;
 91 	
 92 	if(this.mPicked)
 93 	{
 94 		input = this.mInput;
 95 		if(this.mMoveObject)
 96 		{
 97 			var localMouse = this.mOwner.worldToLocal(input.mMousePoint);
 98 			if(this.mRigidBody)
 99 			{ 
100 				if(this.mMouseJoint)
101 				{
102 					this.mMouseJoint.m_target.Set(localMouse.mX, localMouse.mY);
103 				}
104 			}
105 			else
106 			{
107 				this.mOwner.mLocation.mX = localMouse.mX + this.mOffset.mX;
108 				this.mOwner.mLocation.mY = localMouse.mY + this.mOffset.mY;
109 			}
110 		}
111 	}
112 	else if(input.PICKED_COMPONENT == this)
113 	{
114 		input.PICKED_COMPONENT = null;
115 		
116 		if(this.mRigidBody && this.mMouseJoint)
117 		{
118 			ss2d.PhysicalWorld.getWorld().mWorld.DestroyJoint(this.mMouseJoint);
119         	this.mMouseJoint = null;
120         	this.mInput = null;
121         }
122 	}
123 	
124 	this.mPreviousPicked = this.mPicked;
125 };
