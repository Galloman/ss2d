 11 /**
 12  * @constructor
 13  * @param {string} skeletonFileName
 14  */
 15 ss2d.Skeleton = function(skeletonFileName, callbackFunction, callbackTarget)
 16 {
 17 	this.mName = skeletonFileName;
 18 	this.mCallbackFunction = callbackFunction;
 19 	this.mCallbackTarget = callbackTarget||null;
 20 	this.mIncludeAnimations = false;
 21 	var sklFileRequest = new XMLHttpRequest();
 22 	sklFileRequest.mSkeleton = this;
 23 	sklFileRequest.open("GET", skeletonFileName, true);
 24 	sklFileRequest.responseType = "text";
 25 	sklFileRequest.onload = function() 
 26 	{
 27 		this.mSkeleton.skeletonDataLoaded(this.response);
 28 	};
 29 	
 30 	sklFileRequest.send();
 31 };
 32 
 33 ss2d.Skeleton.prototype.skeletonDataLoaded = function(data)
 34 {
 35 	this.mSkeletonData = JSON.parse(data);
 36 	ss2d.Skeleton.allDegToRad(this.mSkeletonData);
 37 	
 38 	if(this.mSkeletonData['animations'])
 39 		this.mIncludeAnimations = true;
 40 		
 41 	if(this.mCallbackFunction)
 42 		this.mCallbackFunction.call(this.mCallbackTarget, this);
 43 };
 44 
 45 ss2d.Skeleton.allDegToRad = function(jsonObject)
 46 {
 47 	var deg2Rad = 1 / (180/Math.PI);
 48 	for(var key in jsonObject)
 49 	{
 50 		if(key == 'rotation' || key == 'angle')
 51 		{
 52 			if(jsonObject[key] > 180)
 53 			{
 54 				jsonObject[key] -= 360;
 55 			}
 56 			else if(jsonObject[key] < -180)
 57 			{
 58 				jsonObject[key] += 360;
 59 			}
 60 			
 61 			jsonObject[key] = (jsonObject[key]*deg2Rad);
 62 		}
 63 		else if(key == 'y')
 64 		{
 65 			jsonObject[key] *= -1;
 66 		}
 67 		else if(typeof jsonObject[key] == 'object' && key != 'scale')
 68 		{
 69 			ss2d.Skeleton.allDegToRad(jsonObject[key]);
 70 		}
 71 	}
 72 };
