 17 /**
 18  * @constructor
 19  * @param {number=} x Position in x axis 
 20  * @param {number=} y Position in y axis
 21  * @param {number=} scale Scale in both x and y axis
 22  * @param {number} rotation
 23  * @param {(number|string|number[3]|ss2d.Color)} color
 24  * @param {number} alpha Transparency
 25  * @property {ss2d.Point} mLocation Object location point
 26  * @property {number} mRotation Rotation in rads
 27  * @property {number} mPivotX Pivot in x axis
 28  * @property {number} mPivotY Pivot in y axis
 29  * @property {number} mScaleX Scale in x axis
 30  * @property {number} mScaleY Scale in y axis
 31  * @property {ss2d.DisplayObjectContainer} mParent Parent in scene tree
 32  * @property {ss2d.Color} mColor Color representation
 33  * @property {number} mAlpha Transparency
 34  */
 35 ss2d.DisplayObject = function(x, y, scale, rotation, color, alpha)
 36 {
 37 	x = x || 0.0;
 38 	y = y || 0.0;
 39 	this.mLocation = new ss2d.Point(x, y);
 40 	//this.mX = x || 0.0;
 41 	//this.mY = y || 0.0;
 42 	this.mRotation = rotation || 0.0;
 43 	this.mPivotX = 0.0;
 44 	this.mPivotY = 0.0;
 45 	this.mScaleX = scale || 1.0;
 46 	this.mScaleY = scale || 1.0;
 47 	this.mParent = null;
 48 	
 49 	if(color instanceof ss2d.Color)
 50 	{
 51 		this.mColor = color;
 52 	}
 53 	else
 54 	{
 55 		color = color||0xffffff;
 56 		this.mColor = new ss2d.Color(color);
 57 	}
 58 	
 59 	this.mAlpha = alpha || 1.0;
 60 	
 61 	
 62 	this.mObjectId = ss2d.Object.OBJECT_COUNT++;
 63 	
 64 	if(COMPILING_SERVER)
 65 	{
 66 		this.mSoundList = [];
 67 	}
 68 };

 72 /**
 73  * @param {ss2d.Matrix3} targetMatrix Matrix where dump the result
 74  * @return {ss2d.Matrix3} the local transformation of the object. 
 75  */
 76 ss2d.DisplayObject.prototype.getTransformationMatrix = function(targetMatrix)
 77 {
 78 	targetMatrix = targetMatrix || new ss2d.Matrix3();
 79 	
 80 	if (this.mPivotX != 0.0 || this.mPivotY != 0.0)				{ targetMatrix.translate(-this.mPivotX,-this.mPivotY); }
 81 	if (this.mScaleX != 1.0 || this.mScaleY != 1.0)				{ targetMatrix.scale(this.mScaleX,this.mScaleY); }
 82 	if (this.mRotation != 0.0)			 						{ targetMatrix.rotate(this.mRotation); }
 83 	if (this.mLocation.mX != 0.0 || this.mLocation.mY != 0.0)	{ targetMatrix.translate(this.mLocation.mX,this.mLocation.mY); }
 84     
 85 	return targetMatrix;
 86 };

 88 /**
 89  * @param {ss2d.Matrix3} targetMatrix Matrix where dump the result
 90  * @return {ss2d.Matrix3} The world transformation of the object. 
 91  */
 92 //TODO: Modify to accept a top parent to iterate instead of the first one without parent.
 93 ss2d.DisplayObject.prototype.getWorldTransformationMatrix = function(targetMatrix)
 94 {
 95 	var parentsStack = [];
 96 	var currentParent = this;
 97 	
 98 	while(currentParent.mParent)
 99 	{
100 		parentsStack.unshift(currentParent.mParent);
101 		currentParent = currentParent.mParent;
102 	}
103 	
104 	var transMatrix = new ss2d.Matrix3();
105 	var curMatrix = new ss2d.Matrix3();
106 	
107 	for(matIndex in parentsStack)
108 	{
109 		curMatrix.identity();
110 		parentsStack[matIndex].getTransformationMatrix(curMatrix);
111 		transMatrix.concatMatrix(curMatrix);
112 	}
113 	
114 	return transMatrix;
115 };

117 /**
118  * Transform a point from world to local coordinate space
119  * @param {ss2d.Point} point
120  * @return {ss2d.Point} Transformed point
121  */
122 ss2d.DisplayObject.prototype.worldToLocal = function(point)
123 {
124 	var invWorldMatrix = this.getWorldTransformationMatrix().invert();
125 	var transformedPoint = invWorldMatrix.transformPoint(point);
126 	return transformedPoint;
127 };

200 if(COMPILING_CLIENT)
201 {
202 	/**
203 	 * Cast a JSON parsed object to a object of this type replacing the original object.
204 	 * This is currently not used by the mutiplayer subsystem.
205 	 * @param {Object} obj The JSON parsed object with 'cid' property.
206 	 * @return {ss2d.DisplayObject} A reference to the converted object.
207 	 */
208 	ss2d.DisplayObject.convert = function(obj)
209 	{
210 		var objBackup =  ss2d.Object.backupAndDeleteObjectProperties(obj);
211 		
212 		ss2d.DisplayObject.call(obj);
213 		
214 		obj.__proto__ = ss2d.DisplayObject.prototype;
215 		
216 		//now obj is a DisplayObject
217 		obj.restoreSerializedProperties(objBackup);
218 		
219 		return obj;
220 	}
221 	
222 	/**
223 	 * Copy all properties from a JSON parsed object to this object.
224 	 * This is currently not used by the mutiplayer subsystem.
225 	 * @param {Object} obj The JSON parsed object with 'cid' property.
226 	 */
227 	ss2d.DisplayObject.prototype.restoreSerializedProperties = function(objBackup)
228 	{
229 		this.CLASS_ID = objBackup['cid'];
230 		this.mLocation.mX = objBackup['x'] || this.mLocation.mX;
231 		this.mLocation.mY = objBackup['y'] || this.mLocation.mY;
232 		this.mRotation = objBackup['r'] || this.mRotation;
233 		this.mPivotX = objBackup['px'] || this.mPivotX;
234 		this.mPivotY = objBackup['py'] || this.mPivotY;
235 		this.mScaleX = objBackup['sx'] || this.mScaleX;
236 		this.mScaleY = objBackup['sy'] || this.mScaleY;
237 		this.mColor = new ss2d.Color(objBackup['c']) || this.mColor;
238 		this.mAlpha = objBackup['a'] || this.mAlpha;
239 	}
240 	
241 	/** 
242 	 * Create a new instance of this type from a JSON parsed object.
243 	 * Used by the multiplayer subsystem to create new objects coming from server.
244 	 * @expose
245 	 * @param {Object} obj The JSON parsed object with 'cid' property.
246 	 * @return {ss2d.DisplayObject}  
247 	 */
248 	ss2d.DisplayObject.createFromSerializedObject = function(obj)
249 	{
250 		var newDo = new ss2d.DisplayObject(obj['x'], obj['y'], obj['s'], obj['r'], obj['c']);
251 		newDo.mObjectId = obj['doid'];
252 		return newDo;
253 	}
254 	
255 	/**
256 	 * Interpolate the object state between a previous and future JSON parsed states.
257 	 * This method also calls the tick method for client side updates.
258 	 * @param {Object} prevState
259 	 * @param {Object} nextState
260 	 * @param {number} part Difference between both states to add to the current state.
261 	 * @param {number} deltaTime
262 	 */
263 	ss2d.DisplayObject.prototype.interpolateState = function(prevState, nextState, part, deltaTime)
264 	{	
265 		this.mLocation.mX = prevState['x'] + (nextState['x'] - prevState['x']) * part;
266 		this.mLocation.mY = prevState['y'] + (nextState['y'] - prevState['y']) * part;
267 		this.mRotation = prevState['r'] + (nextState['r'] - prevState['r']) * part;
268 		this.mPivotX = prevState['px'] + (nextState['px'] - prevState['px']) * part;
269 		this.mPivotY = prevState['py'] + (nextState['py'] - prevState['py']) * part;
270 		this.mScaleX = prevState['sx'] + (nextState['sx'] - prevState['sx']) * part;
271 		this.mScaleY = prevState['sy'] + (nextState['sy'] - prevState['sy']) * part;
272 		var ca = ss2d.Color.interpolate(prevState['c'], nextState['c'], part).getRGBArray();
273 		this.mColor.setRGB(ca[0],ca[1],ca[2]);
274 		this.mAlpha = prevState['a'] + (nextState['a'] - prevState['a']) * part;
275 		
276 		this.playSoundList(prevState);
277 		
278 		//client can update an object state even in an online game, this type of update is intended for non-gameplay 
279 		//related logic, for example when a player receives from server the state "Walking" and the client uses this 
280 		//information to play a SpriteReel animation that represents an avatar of the player walking.
281 		//tick can be defined only for the server, or for both with a block of code for the server and an other for the client.
282 		if(this.tick)
283 		{
284 			this.tick(deltaTime);
285 		}
286 	};
287 	
288 	/**
289 	 * Plays a list of sounds received from the server
290 	 * @param {Object} prevState
291 	 */
292 	ss2d.DisplayObject.prototype.playSoundList = function(prevState)
293 	{
294 		var soundList = prevState['snds'];
295 		if(soundList != null)
296 		{
297 			for(var soundIndex in soundList)
298 			{
299 				var sound = prevState['snds'][childIndex];
300 				try{
301 					ss2d.ResourceLoader.loadSound(sound).play();
302 				}catch(e){}
303 			}
304 			
305 			delete prevState['snds'];
306 		}
307 	};
308 }

310 if(COMPILING_SERVER)

312 	/**
313 	 * Converts this object into a JSON complaint string.
314 	 * @return {string} The serialized version of the object
315 	 */
316 	ss2d.DisplayObject.prototype.toJSON = function()
317 	{
318 		return '{'+this.getPropertiesJSON()+'}';
319 	};
320 	
321 	/**
322 	 * @return A string with all properties serialized in JSON format.
323 	 * <pre>
324 	 * SERIALIZACION SPEC
325 	 * 
326 	 * CLASS_ID : cid
327 	 * mObjectId : doid
328 	 * mLocation.mX : x
329 	 * mLocation.mY : y
330 	 * mRotation : r
331 	 * mPivotX : px
332 	 * mPivotY : py
333 	 * mScaleX : sx
334 	 * mScaleY : sy
335 	 * mColor : c
336 	 * mAlpha : a
337 	 * mSoundList : snds
338 	 * </pre>
339 	 */
340 	ss2d.DisplayObject.prototype.getPropertiesJSON = function()
341 	{
342 		var str = '';
343 		
344 		str += '"cid":'+this.CLASS_ID+',';
345 		str += '"doid":'+this.mObjectId+',';
346 		//str += '"mLocation":{"mX":'+this.mLocation.mX+', "mY":'+this.mLocation.mY+'},';
347 		str += '"x":'+Math.floor(this.mLocation.mX*1000.0)/1000.0+',';
348 		str += '"y":'+Math.floor(this.mLocation.mY*1000.0)/1000.0+',';
349 		str += '"r":'+(Math.floor(this.mRotation*100)/100)+',';
350 		str += '"px":'+this.mPivotX+',';
351 		str += '"py":'+this.mPivotY+',';
352 		str += '"sx":'+this.mScaleX+',';
353 		str += '"sy":'+this.mScaleY+',';
354 		str += '"c":'+JSON.stringify(this.mColor.getRGBArray())+',';
355 		str += '"a":'+this.mAlpha;
356 		str += ',"snds":'+JSON.stringify(this.mSoundList);
357 		
358 		return str;
359 	};
360 }
