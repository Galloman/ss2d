  8 goog.provide('ss2d.TextSprite');

 10 goog.require('ss2d.Quad');
 11 
 12 /**
 13  * @constructor
 14  * @extends {ss2d.Quad}
 15  * @param {number=} x Position in x axis 
 16  * @param {number=} y Position in y axis
 17  * @param {string} textString The text that will be displayed.
 18  * @param {(number|string|number[3]|ss2d.Color)} color 
 19  * @param {number} fontSize size in pixels of the line-height
 20  * @param {string} fontStyle [normal | italic | oblique] [bold | weight-value from 100 to 900]
 21  * @property {*} inherited @see ss2d.Quad
 22  * @property {string} mTextString 
 23  * @property {number} mFontSize
 24  * @property {string}Â mFontName 
 25  * @property {string} mFontStyle
 26  * @property {string} mDisplayChars Number of chars that will be displayed, by default -1 (all), used for text scrolling.
 27  */
 28 ss2d.TextSprite = function(x, y, textString, color, fontSize, fontName, fontStyle)
 29 {
 30 	ss2d.Quad.call(this, x, y, 0, 0, color);
 31 	
 32 	this.mTextString = textString||'';
 33 	this.mFontSize = fontSize||16;
 34 	this.mFontName = fontName||'Verdana';
 35 	this.mFontStyle = fontStyle||'normal';
 36 	this.mDisplayChars = -1;
 37 	//WEBGL
 38 	//if(RENDER_CONTEXT == 'webgl')
 39 	//{
 40 		
 41 	
 42 	//}
 43 };
 44 
 45 goog.inherits(ss2d.TextSprite, ss2d.Quad);
 46 
 47 ss2d.Object.assignClassId(ss2d.TextSprite, 1007);
 48 
 49 /**
 50  * Sets the number of chars that will be displayed, use with ss2d.Tween to create scroll animation.
 51  * @param {number} displayChars
 52  */
 53 ss2d.TextSprite.prototype.setDisplayChars = function(displayChars)
 54 {
 55 	this.mDisplayChars = Math.floor(displayChars);
 56 }
 57 
 58 /**
 59  * Gets the number of chars that will be displayed, use with ss2d.Tween to create scroll animation.
 60  * @return {number}
 61  */
 62 ss2d.TextSprite.prototype.getDisplayChars = function()
 63 {
 64 	return this.mDisplayChars;
 65 }

 67 if(COMPILING_CLIENT||COMPILING_OFFLINE)
 68 {
 69 	/** @override */
 70 	ss2d.TextSprite.prototype.render = function(renderSupport)
 71 	{
 72 
 73 		//if(RENDER_CONTEXT == 'webgl')
 74 		//{
 75 			
 76 		//}
 77 		//else
 78 		//{
 79 			renderSupport.pushTransform(this);
 80 			var ctx = renderSupport.mContext;
 81 			ctx.font = this.mFontStyle+' '+this.mFontSize+'px '+this.mFontName;
 82 			ctx.fillStyle = this.mColor.getHexString();
 83 			var textToDisplay = this.mTextString;
 84 			
 85 			if(this.mDisplayChars > -1)
 86 				textToDisplay = this.mTextString.substring(0, this.mDisplayChars);
 87 			
 88 			ctx.fillText(textToDisplay, 0, this.mFontSize);	 
 89 
 90 			renderSupport.popTransform();
 91 			
 92 		//}
 93 	};
 94 }

100 if(COMPILING_CLIENT)
101 {
102 	/** @override */
103 	ss2d.TextSprite.convert = function(obj)
104 	{
105 		var objBackup =  ss2d.Object.backupAndDeleteObjectProperties(obj);
106 		
107 		ss2d.TextSprite.call(obj);
108 		
109 		obj.__proto__ = ss2d.TextSprite.prototype;
110 		
111 		//now obj is a TextSprite
112 		obj.restoreSerializedProperties(objBackup);
113 		
114 		return obj;
115 	}
116 	
117 	ss2d.TextSprite.prototype.restoreSerializedProperties = function(objBackup)
118 	{
119 		ss2d.Quad.prototype.restoreSerializedProperties.call(this, objBackup);
120 		
121 		this.mTextString = objBackup['str'];
122 		this.mFontSize = objBackup['fsize'];
123 		this.mFontName = objBackup['fname'];
124 		this.mFontStyle = objBackup['fstyle'];
125 		this.mDisplayChars = objBackup['dc'];
126 	}
127 	
128 	ss2d.TextSprite.createFromSerializedObject = function(obj)
129 	{
130 		var newObj = new ss2d.TextSprite(obj['x'], 
131 										 obj['y'], 
132 										 obj['str'], 
133 										 obj['c'],  
134 										 obj['fsize'],  
135 										 obj['fname'],  
136 										 obj['fstyle']);
137 		newObj.mObjectId = obj['doid'];
138 		return newObj;
139 	}
140 	
141 	ss2d.TextSprite.prototype.interpolateState = function(prevState, nextState, part)
142 	{
143 		ss2d.Quad.prototype.interpolateState.call(this, prevState, nextState, part, deltaTime);
144 		
145 		this.mTextString = prevState['str'];
146 		this.mFontSize = prevState['fsize'];
147 		this.mFontName = prevState['fname'];
148 		this.mFontStyle = prevState['fstyle'];
149 		this.mDisplayChars = Math.floor(prevState['dc'] + (nextState['dc'] - prevState['dc']) * part);
150 	}
151 }
