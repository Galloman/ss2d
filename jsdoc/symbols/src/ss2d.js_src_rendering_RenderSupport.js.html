  8 
  9 goog.provide('ss2d.RenderSupport');
 10 
 11 goog.require('ss2d.Defines');
 12 goog.require('ss2d.Matrix3');
 13 
 14 /**
 15  * @constructor
 16  * @param {Object} context
 17  */
 18 ss2d.RenderSupport = function(context)
 19 {
 20 	this.mContext = context;
 21 	this.mAuxMatrix = new ss2d.Matrix3();
 22 	
 23 	if(RENDER_CONTEXT == 'webgl')
 24 	{
 25 		this.mMatrixStack = [];
 26 		this.mCurrentMatrix = new ss2d.Matrix3();
 27 	}
 28 };
 29 
 30 /**
 31  * Push the transformation matrix of the object being rendered.
 32  * @param {ss2d.DisplayObject} displayObject 
 33  */
 34 ss2d.RenderSupport.prototype.pushTransform = function(displayObject){};
 35 if(RENDER_CONTEXT == 'webgl')
 36 {
 37 	ss2d.RenderSupport.prototype.pushTransform = function(displayObject)
 38 	{
 39 		this.mAuxMatrix.identity();
 40 		var curMatrix = displayObject.getTransformationMatrix(this.mAuxMatrix);
 41 		this.mMatrixStack.push(curMatrix);
 42 		this.mCurrentMatrix = this.mCurrentMatrix.concatMatrix(curMatrix);
 43 	};
 44 }
 45 else
 46 {
 47 	ss2d.RenderSupport.prototype.pushTransform = function(displayObject)
 48 	{
 49 		this.mAuxMatrix.identity();
 50 		var curMatrix = displayObject.getTransformationMatrix(this.mAuxMatrix);
 51 		this.mContext.save();
 52 		this.mContext.transform(curMatrix.mA,
 53 								curMatrix.mB,
 54 								curMatrix.mC,
 55 								curMatrix.mD,
 56 								curMatrix.mTx,
 57 								curMatrix.mTy);
 58 	};
 59 }
 60 
 61 /**
 62  * Pop the last transformation matrix.
 63  */
 64 ss2d.RenderSupport.prototype.popTransform = function(){};
 65 if(RENDER_CONTEXT == 'webgl')
 66 {
 67 	ss2d.RenderSupport.prototype.popTransform = function()
 68 	{
 69 		if(this.mMatrixStack.length > 0)
 70 		{
 71 			this.mCurrentMatrix.concatMatrix(this.mMatrixStack.pop().invert());
 72 		}
 73 	};
 74 }
 75 else
 76 {
 77 	ss2d.RenderSupport.prototype.popTransform = function()
 78 	{
 79 		this.mContext.restore();
 80 	};
 81 }
