 58 ss2d.Object.backupAndDeleteObjectProperties = function(obj)
 59 {
 60 	var objBackup = {};
 61 	for(valueKey in obj)
 62 	{
 63 		objBackup[valueKey] = obj[valueKey];
 64 		delete obj[valueKey];
 65 	}
 66 	
 67 	return objBackup;
 68 };

 75 //** @class Compression for websocket packages. Not used yet. */
 76 ss2d.Object.LZWCompressor = {
 77     compress: function (uncompressed) {
 78         "use strict";
 79         // Build the dictionary.
 80         var i,
 81             dictionary = {},
 82             c,
 83             wc,
 84             w = "",
 85             result = [],
 86             dictSize = 256;
 87         for (i = 0; i < 256; i += 1) {
 88             dictionary[String.fromCharCode(i)] = i;
 89         }
 90  
 91         for (i = 0; i < uncompressed.length; i += 1) {
 92             c = uncompressed.charAt(i);
 93             wc = w + c;
 94             //Do not use dictionary[wc] because javascript arrays 
 95             //will return values for array['pop'], array['push'] etc
 96            // if (dictionary[wc]) {
 97             if (dictionary.hasOwnProperty(wc)) {
 98                 w = wc;
 99             } else {
100                 result.push(dictionary[w]);
101                 // Add wc to the dictionary.
102                 dictionary[wc] = dictSize++;
103                 w = String(c);
104             }
105         }
106  
107         // Output the code for w.
108         if (w !== "") {
109             result.push(dictionary[w]);
110         }
111         return result;
112     },
113  
114  
115     decompress: function (compressed) {
116         "use strict";
117         // Build the dictionary.
118         var i,
119             dictionary = [],
120             w,
121             result,
122             k,
123             entry = "",
124             dictSize = 256;
125         for (i = 0; i < 256; i += 1) {
126             dictionary[i] = String.fromCharCode(i);
127         }
128  
129         w = String.fromCharCode(compressed[0]);
130         result = w;
131         for (i = 1; i < compressed.length; i += 1) {
132             k = compressed[i];
133             if (dictionary[k]) {
134                 entry = dictionary[k];
135             } else {
136                 if (k === dictSize) {
137                     entry = w + w.charAt(0);
138                 } else {
139                     return null;
140                 }
141             }
142  
143             result += entry;
144  
145             // Add w+entry[0] to the dictionary.
146             dictionary[dictSize++] = w + entry.charAt(0);
147  
148             w = entry;
149         }
150         return result;
151     }
152 }
