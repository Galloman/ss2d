 12 /**
 13  * @constructor
 14  * @extends {ss2d.Quad}
 15  * @param {number=} x Position in x axis 
 16  * @param {number=} y Position in y axis
 17  * @param {string} textString The text that will be displayed.
 18  * @param {string} bitmapFont Bitmap font name.
 19  * @param {number} fontSize size in pixels of the line-height
 20  * @property {*} inherited @see ss2d.Quad
 21  * @property {string} mTextString 
 22  * @property {number} mFontSize
 23  * @property {string}Â mBitmapFont 
 24  * @property {string} mDisplayChars Number of chars that will be displayed, by default -1 (all), used for text scrolling.
 25  * @property {number[5]} mClip
 26  */
 27 ss2d.BitmapTextSprite = function(x, y, textString, bitmapFont, fontSize)
 28 {
 29 	ss2d.Quad.call(this, x, y, 0, 0);
 30 	
 31 	this.mTextString = textString||'';
 32 	this.mFontSize = fontSize||16;
 33 	this.mDisplayChars = -1;
 34 	this.mBitmapFont = ss2d.ResourceManager.loadBitmapFont(bitmapFont);
 35 	
 36 	
 37 	if(COMPILING_CLIENT || COMPILING_OFFLINE)
 38 	{
 39 		this.mGlyphSprite = new ss2d.Sprite;
 40 		this.mClip = [];
 41 	}
 42 	//WEBGL
 43 	//if(RENDER_CONTEXT == 'webgl')
 44 	//{
 45 		
 46 	
 47 	//}
 48 };
 49 
 50 goog.inherits(ss2d.BitmapTextSprite, ss2d.Quad);
 51 
 52 ss2d.Object.assignClassId(ss2d.BitmapTextSprite, 1007);
 53 
 54 /**
 55  * Sets the number of chars that will be displayed, use with ss2d.Tween to create scroll animation.
 56  * @param {number} displayChars
 57  */
 58 ss2d.BitmapTextSprite.prototype.setDisplayChars = function(displayChars)
 59 {
 60 	this.mDisplayChars = Math.floor(displayChars);
 61 }
 62 
 63 /**
 64  * Gets the number of chars that will be displayed, use with ss2d.Tween to create scroll animation.
 65  * @return {number}
 66  */
 67 ss2d.BitmapTextSprite.prototype.getDisplayChars = function()
 68 {
 69 	return this.mDisplayChars;
 70 }

 74 	/** @override */
 75 	ss2d.BitmapTextSprite.prototype.render = function(renderSupport)
 76 	{
 77 
 78 		//if(RENDER_CONTEXT == 'webgl')
 79 		//{
 80 			
 81 		//}
 82 		//else
 83 		//{
 84 			renderSupport.pushTransform(this);
 85 			var ctx = renderSupport.mContext;
 86 			try{
 87 				this.mWidth = 0;
 88 				var xTranslation = 0;
 89 				this.mGlyphSprite.mTexture = this.mBitmapFont.mTexture;
 90 				var s = this.mFontSize /this.mBitmapFont.mMidHeight;
 91 				this.mScaleX = this.mScaleY = s;
 92 				var charsToShow = (this.mDisplayChars > -1)?
 93 								  Math.min(this.mDisplayChars, this.mTextString.length):
 94 								  this.mTextString.length;
 95 				
 96 				for(var c = 0; c < charsToShow; ++c)
 97 				{
 98 					var charCode = this.mTextString.charCodeAt(c);
 99 					if(charCode == 32){ 
100 						ctx.translate(this.mBitmapFont.mMidWidth*0.5, 0);
101 						continue;
102 					}
103 					
104 					this.mBitmapFont.getGlyphClip(charCode, this.mClip);
105 				
106 					//this.mGlyphSprite.mScaleX = this.mGlyphSprite.mScaleY = s;
107 					//ctx.scale(s, s);
108 					this.mGlyphSprite.mLocation.mX = xTranslation;
109 					this.mGlyphSprite.mLocation.mY = this.mClip[5];
110 					//ctx.translate(0, this.mClip[5]);
111 					
112 					//this.mGlyphSprite.mRotation = this.mRotation
113 					this.mGlyphSprite.mClip = this.mClip;
114 					this.mGlyphSprite.mWidth =  this.mClip[2];
115 					this.mGlyphSprite.mHeight =  this.mClip[3];
116 					
117 					
118 					this.mGlyphSprite.render(renderSupport);
119 					//ctx.rotate(this.mRotation);
120 					
121 					/*ctx.drawImage(this.mBitmapFont.mTexture.mTextureElement,
122 								  this.mClip[0],

125 								  this.mClip[3],
126 								  0,
127 								  0,
128 								  this.mClip[2], 
129 								  this.mClip[3]);
130 					ctx.translate(0, -this.mClip[5]);
131 					ctx.translate(this.mClip[4], 0);
132 					ctx.scale(1/s, 1/s);*/
133 					xTranslation += this.mClip[4];
134 					
135 				}

147 if(COMPILING_CLIENT)
148 {
149 	/** @override */
150 	ss2d.BitmapTextSprite.convert = function(obj)
151 	{
152 		var objBackup =  ss2d.Object.backupAndDeleteObjectProperties(obj);
153 		
154 		ss2d.BitmapTextSprite.call(obj);
155 		
156 		obj.__proto__ = ss2d.BitmapTextSprite.prototype;
157 		
158 		//now obj is a BitmapTextSprite
159 		obj.restoreSerializedProperties(objBackup);
160 		
161 		return obj;
162 	}
163 	
164 	ss2d.BitmapTextSprite.prototype.restoreSerializedProperties = function(objBackup)
165 	{
166 		ss2d.Quad.prototype.restoreSerializedProperties.call(this, objBackup);
167 		
168 		this.mTextString = objBackup['str'];
169 		this.mFontSize = objBackup['fsize'];
170 		this.mBitmapFont = ss2d.ResourceManager.loadBitmapFont(objBackup['bmfname']);
171 		this.mFontStyle = objBackup['fstyle'];
172 		this.mDisplayChars = objBackup['dc'];
173 	}
174 	
175 	ss2d.BitmapTextSprite.createFromSerializedObject = function(obj)
176 	{
177 		var newObj = new ss2d.BitmapTextSprite(obj['x'], 
178 											   obj['y'], 
179 											   obj['str'], 
180 											   obj['bmfname'],
181 											   obj['fsize']);
182 		newObj.mObjectId = obj['doid'];
183 		return newObj;
184 	}
185 	
186 	ss2d.BitmapTextSprite.prototype.interpolateState = function(prevState, nextState, part)
187 	{
188 		ss2d.Quad.prototype.interpolateState.call(this, prevState, nextState, part, deltaTime);
189 		
190 		this.mTextString = prevState['str'];
191 		this.mFontSize = prevState['fsize'];
192 		
193 		if(!this.mBitmapFont ||this.mBitmapFont.mName != prevState['bmfname'])
194 			this.mBitmapFont = ss2d.ResourceManager.loadBitmapFont(prevState['bmfname']);
195 
196 		this.mDisplayChars = Math.floor(prevState['dc'] + (nextState['dc'] - prevState['dc']) * part);
197 	}
198 }
