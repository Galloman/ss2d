 17 /**
 18  * @constructor
 19  * @param {string} canvasId
 20  * @param {ss2d.DisplayObject=} mainScene
 21  * @param {number=} canvasWidth
 22  * @param {number=} canvasHeight
 23  * @param {number=} frameRate Frames per second, 60.0 by default
 24  * @property {HTMLCanvasElement} mCanvas Canvas HTML Canvas element
 25  * @property {CanvasRenderingContext2d} mContext Canvas context
 26  * @property {boolean} mRunning Is main loop running?
 27  * @property {number} mFrameRate Frames per second (60 by default)
 28  * @property {ss2d.Input} mInput Mapped user input
 29  * @property {ss2d.DisplayObjectContainer} mMainScene Scene that is being rendering
 30  * @property {ss2d.RenderSupport} mRenderSupport Rendering support
 31  * @property {string} mBackgroundFillStyle Background fill style mapped to Context fillstyle
 32  * @property {ss2d.PhysicalWorld} mPhysicalWorld A box2d physical world
 33  */
 34 ss2d.View = function(canvasId, mainScene, canvasWidth, canvasHeight, frameRate)
 35 {
 36 	this.mCanvas = document.getElementById(canvasId);
 37 	this.mContext = this.mCanvas.getContext('2d'); 
 38 	this.mRunning = false;
 39 	this.mFrameRate = frameRate || 60.0;
 40 	this.mInput = new ss2d.Input(this);
 41 	
 42 	this.mLastFrameTimestamp = new Date().getTime();
 43 	
 44 	ss2d.CURRENT_VIEW = this;
 45 	ss2d.View.CANVAS_CONTEXT = this.mContext;
 46 	
 47 	this.mCanvas.width = canvasWidth || this.mCanvas.width;
 48 	this.mCanvas.height = canvasHeight || this.mCanvas.height;
 49 	
 50 	ss2d.AUDIO_CONTEXT = ss2d.WebAudio.getAudioContext();
 51 
 52 	this.mMainScene = mainScene || new ss2d.DisplayObjectContainer();
 53 	this.mRenderSupport = new ss2d.RenderSupport(this.mContext);
 54 	
 55 	this.mBackgroundFillStyle = '#202020';
 56 	
 57 	this.mPhysicalWorld = ss2d.PhysicalWorld.getWorld();
 58 };
 59 
 60 /** @type {Object} */
 61 ss2d.View.CANVAS_CONTEXT = null;
 62 
 63 /** @type {function} */
 64 ss2d.View.NEXT_FRAME_CALLER = function(){ ss2d.CURRENT_VIEW.nextFrame() };
 65 
 66 /**
 67  * Called every frame to update and render every object
 68  */
 69 ss2d.View.prototype.nextFrame = function()
 70 {
 71 	//get current time in miliseconds
 72 	var now = new Date().getTime();
 73 	var timePassed = now - this.mLastFrameTimestamp;
 74 	
 75 	//called with canvas width and height every frame
 76 	this.resizeCanvas(this.mCanvas.width, this.mCanvas.height);
 77 	
 78 	//update scene
 79 	this.mMainScene.tick(timePassed/1000.0);
 80 	
 81 	//update input
 82 	this.mInput.tick(timePassed/1000.0);
 83 	
 84 	//update physics
 85 	var worldUpdates = Math.floor(Math.max(1, timePassed/(1000.0/ss2d.PhysicalWorld.UPDATE_RATE)));
 86 	for(var i = 0; i < worldUpdates; ++i)
 87 	{
 88 		this.mPhysicalWorld.tick((timePassed/worldUpdates)/1000.0);
 89 	}
 90 	
 91 	//clean background
 92 	this.mContext.fillStyle = this.mBackgroundFillStyle;  
 93 	this.mContext.fillRect(0, 0, this.mCanvas.width, this.mCanvas.height); 
 94 	
 95 	//render scene
 96 	this.mMainScene.render(this.mRenderSupport);
 97 	
 98 	//calculate the delay time for the nextFrame call based on the 
 99 	//frameRate and time spend in update and render operations.
100 	this.mLastFrameTimestamp = now;
101 	var wastedTime = (new Date().getTime() - now) / 1000.0;
102 	var nextFrameTime = Math.max(0.0, (1.0/this.mFrameRate) - wastedTime);
103 	
104 	
105 	//if game is running call the next frame with the specified time
106 	if(this.mRunning)
107 	{ 
108 		setTimeout(ss2d.View.NEXT_FRAME_CALLER, nextFrameTime * 1000.0);
109 	}
110 };
111 
112 //start the main loop if it's ready.
113 ss2d.View.prototype.startLoop = function()
114 {
115 	if(!this.mRunning && this.mMainScene != null)
116 	{
117 		this.mRunning = true;
118 		this.nextFrame();
119 	}	
120 };
121 
122 /**
123  * Start the main loop if it's ready.
124  */
125 ss2d.View.prototype.stopLoop = function()
126 {
127 	this.mRunning = false;	
128 };
