 11 goog.require('ss2d.Input');

 15 /**
 16  * @constructor
 17  * @param {string} canvasId
 18  * @param {ss2d.DisplayObject} mainScene
 19  * @param {number} canvasWidth
 20  * @param {number} canvasHeight
 21  * @param {number} frameRate Frames per second, 60.0 by default
 22  */
 23 ss2d.View = function(canvasId, mainScene, canvasWidth, canvasHeight, frameRate)
 24 {
 25 	this.mCanvas = document.getElementById(canvasId);
 26 	this.mContext = this.mCanvas.getContext('2d'); 
 27 	this.mRunning = false;
 28 	this.mFrameRate = frameRate || 60.0;
 29 	this.mInput = new ss2d.Input(this);
 30 	this.mLastFrameTimestamp = new Date().getTime();
 31 	
 32 	ss2d.CURRENT_VIEW = this;
 33 	ss2d.View.CANVAS_CONTEXT = this.mContext;
 34 	
 35 	this.mCanvas.width = canvasWidth || 800;
 36 	this.mCanvas.height = canvasHeight || 600;
 37 	
 38 	ss2d.AUDIO_CONTEXT = ss2d.WebAudio.getAudioContext();
 39 	
 40 	this.mMainScene = mainScene || null;
 41 	this.mRenderSupport = new ss2d.RenderSupport(this.mContext);
 42 	this.mBackgroundFillStyle = '#202020';
 43 };
 44 
 45 /** @type {Object} */
 46 ss2d.View.CANVAS_CONTEXT = null;
 47 
 48 /** @type {function} */
 49 ss2d.View.NEXT_FRAME_CALLER = function(){ ss2d.CURRENT_VIEW.nextFrame() };
 50 
 51 /**
 52  * Called every frame to update and render every object
 53  */
 54 ss2d.View.prototype.nextFrame = function()
 55 {
 56 	//get current time in miliseconds
 57 	var now = new Date().getTime();
 58 	var timePassed = now - this.mLastFrameTimestamp;
 59 	
 60 	//called with canvas width and height every frame
 61 	this.resizeCanvas(this.mCanvas.width, this.mCanvas.height);
 62 	
 63 	//update scene
 64 	this.mMainScene.tick(timePassed/1000.0);
 65 	
 66 	//update input
 67 	this.mInput.tick(timePassed/1000.0);
 68 	
 69 	//clean background
 70 	this.mContext.fillStyle = this.mBackgroundFillStyle;  
 71 	this.mContext.fillRect(0, 0, this.mCanvas.width, this.mCanvas.height); 
 72 	
 73 	//render scene
 74 	this.mMainScene.render(this.mRenderSupport);
 75 	
 76 	//calculate the delay time for the nextFrame call based on the 
 77 	//frameRate and time spend in update and render operations.
 78 	this.mLastFrameTimestamp = now;
 79 	var wastedTime = (new Date().getTime() - now) / 1000.0;
 80 	var nextFrameTime = Math.max(0.0, (1.0/this.mFrameRate) - wastedTime);
 81 	
 82 	
 83 	//if game is running call the next frame with the specified time
 84 	if(this.mRunning)
 85 	{ 
 86 		setTimeout(ss2d.View.NEXT_FRAME_CALLER, nextFrameTime * 1000.0);
 87 	}
 88 };
 89 
 90 //start the main loop if it's ready.
 91 ss2d.View.prototype.startLoop = function()
 92 {
 93 	if(!this.mRunning && this.mMainScene != null)
 94 	{
 95 		this.mRunning = true;
 96 		this.nextFrame();
 97 	}	
 98 };
 99 
100 /**
101  * Start the main loop if it's ready.
102  */
103 ss2d.View.prototype.stopLoop = function()
104 {
105 	this.mRunning = false;	
106 };
