 17 /**
 18  * @constructor
 19  * @param {string} canvasId
 20  * @param {ss2d.DisplayObject=} mainScene
 21  * @param {number=} canvasWidth
 22  * @param {number=} canvasHeight
 23  * @param {number=} frameRate Frames per second, 60.0 by default
 24  * @property {HTMLCanvasElement} mCanvas Canvas HTML Canvas element
 25  * @property {CanvasRenderingContext2d|WebGLRenderginContext} mContext Canvas context
 26  * @property {number[16]} mProjection Orthogonal projection matrix
 27  * @property {boolean} mRunning Is main loop running?
 28  * @property {number} mFrameRate Frames per second (60 by default)
 29  * @property {ss2d.Input} mInput Mapped user input
 30  * @property {ss2d.DisplayObjectContainer} mMainScene Scene that is being rendering
 31  * @property {ss2d.RenderSupport} mRenderSupport Rendering support
 32  * @property {string} mBackgroundFillStyle Background fill style mapped to Context fillstyle
 33  * @property {ss2d.PhysicalWorld} mPhysicalWorld A box2d physical world
 34  * @property {Array} mPreTickFunctions Functions called every frame before the main scene tick method.
 35  * @property {Array} mPostTickFunctions Functions called every frame after the main scene tick method.
 36  * @property {Array} mPreRenderFunctions Functions called every frame before the main scene render method.
 37  * @property {Array} mPostRenderFunctions Functions called every frame after the main scene render method.
 38  * @property {number} mTotalTime Time passed since the view loop starts.
 39  */
 40 ss2d.View = function(canvasId, mainScene, canvasWidth, canvasHeight, frameRate)
 41 {
 42 	this.mCanvas = document.getElementById(canvasId);
 43 	
 44 	if(RENDER_CONTEXT == 'webgl')
 45 	{
 46 		this.mContext = this.mCanvas.getContext('webgl', { alpha: false })||this.mCanvas.getContext('experimental-webgl', { alpha: false });
 47 		this.mProjection = ss2d.RenderSupport.make2DProjection(this.mCanvas.width, this.mCanvas.height);
 48 		this.mClearColor = [0.8, 0.8, 0.8, 1.0];
 49 	}
 50 	else
 51 	{
 52 		this.mContext = this.mCanvas.getContext('2d'); 
 53 		this.mBackgroundFillStyle = '#cccccc';
 54 	}
 55 	
 56 	if(!this.mContext)
 57 	{
 58 		throw 'Incompatible render context: '+RENDER_CONTEXT;
 59 	}
 60 
 61 	ss2d.CURRENT_VIEW = this;
 62 	
 63 	this.mRenderSupport = new ss2d.RenderSupport(this.mContext);
 64 	
 65 	this.mRunning = false;
 66 	this.mFrameRate = frameRate || 60.0;
 67 	this.mInput = new ss2d.Input(this);
 68 	
 69 	this.mLastFrameTimestamp = new Date().getTime();
 70 
 71 	this.mCanvas.width = canvasWidth || this.mCanvas.width;
 72 	this.mCanvas.height = canvasHeight || this.mCanvas.height;
 73 	
 74 	ss2d.AUDIO_CONTEXT = ss2d.WebAudio.getAudioContext();
 75 
 76 	this.mMainScene = mainScene || new ss2d.DisplayObjectContainer();
 77 	
 78 	this.mPhysicalWorld = ss2d.PhysicalWorld.getWorld();
 79 	
 80 	//plug-in development
 81 	this.mPreTickFunctions = [];
 82 	this.mPostTickFunctions = [];
 83 	this.mPreRenderFunctions = [];
 84 	this.mPostRenderFunctions = [];
 85 	
 86 	this.mTotalTime = 0;
 87 };
 88 
 89 /** @type {function} */
 90 ss2d.View.NEXT_FRAME_CALLER = function(){ ss2d.CURRENT_VIEW.nextFrame() };
 91 
 92 /**
 93  * Called every frame to update and render every object
 94  */
 95 ss2d.View.prototype.nextFrame = function()
 96 {
 97 	//get current time in miliseconds
 98 	var now = new Date().getTime();
 99 	var timePassed = now - this.mLastFrameTimestamp;
100 	
101 	//called with canvas width and height every frame
102 	this.resizeCanvas(this.mCanvas.width, this.mCanvas.height);
103 	
104 	//update physics
105 	var worldUpdates = Math.floor(Math.max(1, timePassed/(1000.0/ss2d.PhysicalWorld.UPDATE_RATE)));
106 	for(var i = 0; i < worldUpdates; ++i)
107 	{
108 		this.mPhysicalWorld.tick((timePassed/worldUpdates)/1000.0);
109 	}
110 	
111 	var timePassedInSeconds = timePassed / 1000.0;
112 	
113 	this.mRealFPS = Math.floor(100.0 / timePassedInSeconds)*0.01;
114 	this.mTotalTime += timePassedInSeconds;
115 	
116 	//update input
117 	this.mInput.tick(timePassedInSeconds);
118 	
119 	for(var methodIndex in this.mPreTickFunctions)
120 	{ 
121 		this.mPreTickFunctions[methodIndex].call(null, timePassedInSeconds); 
122 	}
123 	
124 	//update scene
125 	this.mMainScene.tick(timePassedInSeconds);
126 
127 	for(var methodIndex in this.mPostTickFunctions)
128 	{ 
129 		this.mPostTickFunctions[methodIndex].call(null, timePassedInSeconds); 
130 	}
131 	
132 	for(var methodIndex in this.mPreRenderFunctions)
133 	{ 
134 		this.mPreRenderFunctions[methodIndex].call(null, this.mRenderSupport); 
135 	}
136 	
137 	this.render();
138 	
139 	for(var methodIndex in this.mPostRenderFunctions)
140 	{ 
141 		this.mPostRenderFunctions[methodIndex].call(null, this.mRenderSupport); 
142 	}
143 
144 	//calculate the delay time for the nextFrame call based on the 
145 	//frameRate and time spend in update and render operations.
146 	this.mLastFrameTimestamp = now;
147 	var wastedTime = (new Date().getTime() - now) / 1000.0;
148 	var nextFrameTime = Math.max(0.0, (1.0/this.mFrameRate) - wastedTime);
149 	
150 	
151 	//if game is running call the next frame with the specified time
152 	if(this.mRunning)
153 	{ 
154 		setTimeout(ss2d.View.NEXT_FRAME_CALLER, nextFrameTime * 1000.0);
155 	}
156 };
157 
158 /**
159  * Called every frame by nextFrame method to render the scene in the proper context.
160  */
161 ss2d.View.prototype.render = function(){};
162 
163 if(RENDER_CONTEXT == 'webgl')
164 {
165 	ss2d.View.prototype.render = function()
166 	{	
167 		var gl = this.mContext;
168 		gl.viewport(0, 0, this.mCanvas.width, this.mCanvas.height);
169 		var cc = this.mClearColor;
170 		gl.clearColor(cc[0], cc[1], cc[2], cc[3]);
171 		gl.clear(gl.COLOR_BUFFER_BIT);
172 		
173 		//render scene
174 		this.mMainScene.render(this.mRenderSupport);
175 	}
176 }
177 else
178 {
179 	ss2d.View.prototype.render = function()
180 	{
181 		//clean background
182 		this.mContext.fillStyle = this.mBackgroundFillStyle;  
183 		this.mContext.fillRect(0, 0, this.mCanvas.width, this.mCanvas.height); 
184 
185 		//render scene
186 		this.mMainScene.render(this.mRenderSupport);
187 	};
188 }
189 
190 //start the main loop if it's ready.
191 ss2d.View.prototype.startLoop = function()
192 {
193 	if(!this.mRunning && this.mMainScene != null)
194 	{
195 		this.mRunning = true;
196 		this.nextFrame();
197 	}	
198 };
199 
200 /**
201  * Start the main loop if it's ready.
202  */
203 ss2d.View.prototype.stopLoop = function()
204 {
205 	this.mRunning = false;	
206 };
