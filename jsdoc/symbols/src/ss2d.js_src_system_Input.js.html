 11 /**
 12  * @param {ss2d.View} view
 13  * @constructor
 14  * @implements {EventHandler} 
 15  * @property {number} mMouseX Mouse location in x axis
 16  * @property {number} mMouseY Mouse location in y axis
 17  * @property {number} mPreviousMouseX Last frame mouse location in x axis
 18  * @property {number} mPreviousMouseY Last frame mouse location in y axis
 19  * @property {ss2d.Point} mMousePoint Mouse location point
 20  * @property {ss2d.Point} mPreviousMousePoint Last frame mouse location point
 21  * @property {boolean} mMouseDown Is mouse left button down?
 22  * @property {boolean} mPreviousMouseDown Was mouse left button down last frame?
 23  */
 24 ss2d.Input = function(view)
 25 {
 26 	this.mView = view;
 27 	var canvas = view.mCanvas;
 28 	canvas.style.outline = 'none';
 29 	this.mView.mHaveFocus = true;
 30 	this.mMobileDevice = false;
 31 	if((navigator.userAgent.match(/Android/i)) || (navigator.userAgent.match(/iPhone/i)) || (navigator.userAgent.match(/iPod/i)) ||  (navigator.userAgent.match(/iPad/i)))
 32 	{
 33 		this.mMobileDevice = true;
 34    	}
 35 	
 36 	
 37    	if(!this.mMobileDevice){
 38    		if (window.navigator.msPointerEnabled) { // WIN8
 39    			//Gorkinovich contribution: Windows 8 Touchscreen
 40    			window.addEventListener("MSPointerDown", ss2d.Input.windowMouseEventHandler, true);
 41 		} 
 42 		else 
 43 		{
 44 			window.addEventListener('mousedown', ss2d.Input.windowMouseEventHandler, true);
 45 		}
 46    	}
 47    	window.addEventListener('keydown', ss2d.Input.handleEventCaller, true);
 48 	window.addEventListener('keyup', ss2d.Input.handleEventCaller, true);
 49    	
 50    	if (window.navigator.msPointerEnabled) { // WIN8
 51    		a.addEventListener("MSPointerDown", ss2d.Input.handleEventCaller, true);
 52 		a.addEventListener("MSPointerUp", ss2d.Input.handleEventCaller, true);
 53 		a.addEventListener("MSPointerMove", ss2d.Input.handleEventCaller, true);
 54    	} 
 55    	else
 56    	{
 57    		canvas.addEventListener('mousedown', ss2d.Input.handleEventCaller, true);
 58 		canvas.addEventListener('mouseup', ss2d.Input.handleEventCaller, true);
 59 		canvas.addEventListener('mousemove', ss2d.Input.handleEventCaller, true);
 60    	}
 61 
 62 
 63 	canvas.addEventListener("touchstart", ss2d.Input.handleEventCaller, false);
 64 	canvas.addEventListener("touchend", ss2d.Input.handleEventCaller, false);
 65 	canvas.addEventListener("touchcancel", ss2d.Input.handleEventCaller, false);
 66 	canvas.addEventListener("touchleave", ss2d.Input.handleEventCaller, false);
 67 
 68 	this.mPressedKeys = {};
 69 	this.mMouseX = -50;
 70 	this.mMouseY = -50;
 71 	this.mPreviousMouseX = 0;
 72 	this.mPreviousMouseY = 0;
 73 	this.mMouseDown = false;
 74 	this.mMousePoint = new ss2d.Point(this.mMouseX, this.mMouseY);
 75 	this.mPreviousMousePoint = new ss2d.Point(this.mPreviousMouseX, this.mPreviousMouseY);
 76 	this.mPreviousMouseDown = false;
 77 	this.mClicked = false;
 78 }

135 ss2d.Input.prototype.tick = function()
136 {
137 	this.mPreviousMouseX = this.mMouseX;
138 	this.mPreviousMouseY = this.mMouseY;
139 	this.mPreviousMousePoint.mX = this.mPreviousMouseX;
140 	this.mPreviousMousePoint.mY = this.mPreviousMouseY;
141 	this.mPreviousMouseDown = this.mMouseDown;
142 	this.mMouseDown = this.mClicked;	
143 }

145 //catch all events
146 ss2d.Input.handleEventCaller = function(e)
147 { 
148 	ss2d.CURRENT_VIEW.mInput.handleEvent(e); 
149 }
150 
151 //catch mobile events
152 ss2d.Input.windowMouseEventHandler = function(e)
153 {
154 	var input = ss2d.CURRENT_VIEW.mInput;
155 	if(input.mView.mHaveFocus)
156 	{
157 		input.onFocusOut();
158 	}
159 }

161 /** @override */
162 ss2d.Input.prototype.handleEvent = function(event)
163 {
164 	var r = true;
165 	switch(event.type){
166 	case 'mousedown': case 'touchstart': case "MSPointerDown": r = this.onMouseDown(event); break;
167 	case 'mouseup': case 'touchend': case 'touchcancel': case "MSPointerUp": r = this.onMouseUp(event); break;
168 	case 'keydown': r = this.onKeyDown(event); break;
169 	case 'keyup': r = this.onKeyUp(event); break;
170 	case 'mousemove': case 'touchmove':  case "MSPointerMove": r = this.onMouseMove(event); break;
171 	}
172 
173 	return r;
174 }

176 ss2d.Input.prototype.onMouseDown = function(pe)
177 {	
178 	var event = this.mMobileDevice ? window.event.targetTouches[ 0 ] : pe;
179 	
180 	this.mMouseX = event.offsetX || event.pageX - this.mView.mCanvas.offsetLeft;
181 	this.mMouseY = event.offsetY || event.pageY - this.mView.mCanvas.offsetTop;
182 	this.mMousePoint.mX = this.mMouseX;
183 	this.mMousePoint.mY = this.mMouseY;
184 
185 	if((this.mMobileDevice && event) || !this.mMobileDevice){
186 		this.onFocusIn();
187 		this.mClicked = true;
188 		pe.preventDefault();
189 		pe.stopPropagation();
190 		this.mView.mCanvas.addEventListener("touchmove", ss2d.Input.handleEventCaller, false);
191 	}
192 	
193 	return false;
194 }
195 
196 ss2d.Input.prototype.onMouseUp = function(pe)
197 {
198 	var event = this.mMobileDevice ? window.event.targetTouches[ 0 ] : pe;
199 	if((this.mMobileDevice && !event) || !this.mMobileDevice){
200 		this.mMouseDown = false;
201 		this.mClicked = false;
202 		pe.preventDefault();
203 		pe.stopPropagation();
204 		this.mView.mCanvas.removeEventListener("touchmove", ss2d.Input.handleEventCaller, false);
205 	}
206 	return false;
207 }

209 ss2d.Input.prototype.onKeyDown = function(event)
210 {
211 	if(this.mView.mHaveFocus)
212 	{ 
213 		this.mPressedKeys['_'+parseInt(event.keyCode)] = true;
214 		if(event.keyCode == 37 || event.keyCode == 38 || event.keyCode == 39 || event.keyCode == 40 || event.keyCode == 32)
215 		{
216 			event.preventDefault();	
217 		}
218 		return false;
219 	}
220 	return true;
221 }
222 
223 ss2d.Input.prototype.onKeyUp = function(event)
224 {
225 	if(this.mView.mHaveFocus)
226 	{ 
227 		this.mPressedKeys['_'+parseInt(event.keyCode)] = false;
228 		delete this.mPressedKeys['_'+parseInt(event.keyCode)];
229 		return false;
230 	}
231 	return true;
232 }
233 
234 ss2d.Input.prototype.onFocusIn = function(event)
235 {
236 	this.mView.mHaveFocus = true;
237 }
238 
239 ss2d.Input.prototype.onFocusOut = function(event)

251 ss2d.Input.prototype.onMouseMove = function(pe)
252 {
253 	var event = this.mMobileDevice ? window.event.targetTouches[0] : pe;
254 	if(event){
255 		this.mMouseX = event.offsetX || event.pageX - this.mView.mCanvas.offsetLeft;
256 		this.mMouseY = event.offsetY || event.pageY - this.mView.mCanvas.offsetTop;
257 		this.mMousePoint.mX = this.mMouseX;
258 		this.mMousePoint.mY = this.mMouseY;
259 	}
260 }
261 
262 ss2d.Input.prototype.isKeyPressed = function(keycode)
263 {
264 	return (this.mPressedKeys['_'+keycode] != null)?true:false;
265 };

267 ss2d.Input.prototype.getTransformedMousePoint = function()
268 {
269 	if(ss2d.CURRENT_VIEW.mMainScene instanceof ss2d.DisplayObjectContainer)
270 	{
271 		return ss2d.CURRENT_VIEW.mMainScene.getTransformationMatrix().invert().transformPoint(this.mMousePoint);
272 	} else {
273 		return this.mMousePoint;
274 	}
275 }

277 ss2d.Input.prototype.toJSON = function()
278 {
279 	var transMouse = this.getTransformedMousePoint();
280 	
281 	if(isNaN(transMouse.mX) || isNaN(transMouse.mY))
282 	{
283 		transMouse = this.mMousePoint;
284 	}
285 	
286 	var str = '{';
287 	str += '"keys":'+JSON.stringify(this.mPressedKeys)+',';
288 	str += '"mx":'+transMouse.mX+',';
289 	str += '"my":'+transMouse.mY+',';
290 	str += '"md":'+this.mMouseDown;
291 	return str+'}';
292 }
