  3 /**
  4  * @fileoverview A simple 2D point.

  8 goog.provide('ss2d.Point');
  9 
 10 /**
 11  * @constructor
 12  * @param {number} x coordinate
 13  * @param {number} y coordinate
 14  */
 15 ss2d.Point = function(x, y)
 16 {
 17 	this.mX = x;
 18 	this.mY = y;
 19 };
 20 
 21 /**
 22  * Set point values
 23  * @param {number} xv
 24  * @param {number} yv
 25  */
 26 ss2d.Point.prototype.set = function(xv, yv)
 27 {
 28 	this.mX = xv;
 29 	this.mY = yv;
 30 };
 31 
 32 /**
 33  * @return {number} the length of the vector
 34  */
 35 ss2d.Point.prototype.length = function()
 36 {
 37 	return Math.sqrt((this.mX*this.mX) + (this.mY*this.mY));
 38 }
 39 
 40 /**
 41  * @return {ss2d.Point} reutn the normalized version of this point
 42  */
 43 ss2d.Point.prototype.normalize = function()
 44 {
 45 	return ss2d.Point.scalePoint(this, 1.0 / this.length());
 46 }
 47 
 48 /**
 49  * @return {number} the x coordinate
 50  */
 51 ss2d.Point.prototype.getX = function()
 52 { 
 53 	return this.mX; 
 54 };
 55 
 56 /**
 57  * @param {number} x
 58  */
 59 ss2d.Point.prototype.setX = function(x)
 60 { 
 61 	this.mX = x; 
 62 };
 63 
 64 /**
 65  * @return {number} the y coordinate
 66  */
 67 ss2d.Point.prototype.getY = function()
 68 { 
 69 	return this.mY; 
 70 };
 71 
 72 /**
 73  * @param {number} y
 74  */
 75 ss2d.Point.prototype.setY = function(y)
 76 { 
 77 	this.mY = y	
 78 };
 79 
 80 /**
 81  * @static
 82  * @param {ss2d.Point} p1
 83  * @param {ss2d.Point} p2
 84  * @return {number} distance between points
 85  */
 86 ss2d.Point.distanceBetweenPoints = function(p1, p2)
 87 {
 88 	return Math.abs(Math.sqrt(((p2.mX - p1.mX) * (p2.mX - p1.mX)) + ((p2.mY - p1.mY) * (p2.mY - p1.mY))));
 89 }
 90 
 91 /**
 92  * @static
 93  * @param {ss2d.Point} p1
 94  * @param {ss2d.Point} p2
 95  * @return {ss2d.Point} the two points added
 96  */
 97 ss2d.Point.addPoints = function(p1, p2)
 98 {
 99 	return new ss2d.Point(p1.mX + p2.mX, p1.mY + p2.mY);
100 }
101 
102 /**
103  * @static
104  * @param {ss2d.Point} p1
105  * @param {ss2d.Point} p2
106  * @return {ss2d.Point} the subtract result
107  */
108 ss2d.Point.subtractPoints = function(p1, p2)
109 {
110 	return new ss2d.Point(p2.mX - p1.mX, p2.mY - p1.mY);
111 }
112 
113 /**
114  * @static
115  * @param {ss2d.Point} p1
116  * @param {number} scalar
117  * @return {ss2d.Point} the scaled point
118  */
119 ss2d.Point.scalePoint = function(p, scalar)
120 {
121 	return new ss2d.Point(scalar * p.mX, scalar * p.mY);
122 }
