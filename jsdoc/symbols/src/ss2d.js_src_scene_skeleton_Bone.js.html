 10 goog.require('ss2d.DisplayObjectContainer');
 11 goog.require('ss2d.Transitions');
 12 /**
 13  * @constructor
 14  * @extends {ss2d.DisplayObjectContainer} 
 15  * @param {ss2d.SkeletalSprite} skeletalSprite
 16  * @param {string} name
 17  * @param {number} x
 18  * @param {number} y
 19  * @param {number} scale
 20  * @param {number} rotation
 21  * @param {number} length
 22  */
 23 ss2d.Bone = function(skeletalSprite, name, x, y, scaleX, scaleY, rotation, length)
 24 {
 25 	ss2d.DisplayObjectContainer.call(this, x, y, 1, rotation);
 26 	this.mScaleX = scaleX;
 27 	this.mScaleY = scaleY;
 28 	
 29 	//for a graphical representation
 30 	this.mLength = length; 
 31 	
 32 	this.mName = name;
 33 	this.mParentSkeletalSprite = skeletalSprite;
 34 	
 35 	this.mRepresentation = new ss2d.Quad(0, 0, 5, length, '#ff0000');
 36 	this.mRepresentation.mAlpha = 0.7;
 37 	this.mRepresentation.mRotation = Math.PI*0.5;
 38 	this.mRepresentation.mInheritColor = this.mRepresentation.mInheritAlpha = false;
 39 	this.addObject(this.mRepresentation);
 40 	
 41 	this.mSetupPose = {'x': x, 'y': y, 'r': rotation, 'sx': scaleX, 'sy':scaleY};
 42 };
 43 
 44 goog.inherits(ss2d.Bone, ss2d.DisplayObjectContainer);
 45 
 46 /**
 47  * Interpolate between two bone states
 48  * @param {Object} prevState
 49  * @param {Object} nextState
 50  * @param {number} part
 51  */
 52 ss2d.Bone.prototype.interpolateBoneStates = function(prevState, nextState, parts, curves)
 53 {
 54 	//TODO: smart rotation interpolation: 
 55 	//change the next state value if the difference between both states is greater than 360 degrees
 56 	this.mLocation.mX = this.mSetupPose['x'] + prevState['x'] + ((nextState['x'] - prevState['x']) * parts['xy']);
 57 	this.mLocation.mY = this.mSetupPose['y'] + prevState['y'] + ((nextState['y'] - prevState['y']) * parts['xy']);
 58 	this.mRotation = this.mSetupPose['r'] + prevState['r'] + ((nextState['r'] - prevState['r']) * parts['r']);
 59 	this.mScaleX = prevState['sx'] + ((nextState['sx'] - prevState['sx']) * parts['s']);
 60 	this.mScaleY = prevState['sy'] + ((nextState['sy'] - prevState['sy']) * parts['s']);
 61 	
 62 	if(this.mScaleX == 0 ||this.mScaleY == 0) throw 'scale 0';
 63 };

 65 ss2d.Bone.prototype.tick = function(deltaTime)
 66 {
 67 	this.updateBone(deltaTime);
 68 }
 69 /**
 70  * Update bone state based on active animation
 71  * @param {number} deltaTime
 72  */
 73 ss2d.Bone.prototype.updateBone = function(deltaTime)
 74 {
 75 	this.tickChildren(deltaTime);
 76 	
 77 	var parent = this.mParentSkeletalSprite;
 78 	
 79 	if(!parent.mCurrentAnimation)
 80 		return;
 81 		
 82 	var animation = parent.mCurrentAnimation.mAnimationData;
 83 	
 84 	if(!animation || !animation['bones'] || !animation['bones'][this.mName])
 85 		return;
 86 		
 87 	var playReverse = (parent.mTimeDilation < 0)? true : false;
 88 	var currentTime = this.mParentSkeletalSprite.mCurrentAnimationTime;
 89 	var boneRotations = animation['bones'][this.mName]['rotate'];
 90 	var boneTranslations = animation['bones'][this.mName]['translate'];
 91 	var boneScalations =  animation['bones'][this.mName]['scale'];
 92 	
 93 	var prevState = {'x':0, 'y':0, 'r': 0, 'sx':this.mSetupPose['sx'], 'sy':this.mSetupPose['sy']};
 94 	var nextState = {'x':0, 'y':0, 'r': 0, 'sx':this.mSetupPose['sx'], 'sy':this.mSetupPose['sy']};
 95 	var parts = {'xy':0, 'r': 0, 's':0};
 96 	var curves = {'xy':ss2d.Transitions.linear, 'r': ss2d.Transitions.linear, 's':ss2d.Transitions.linear};
 97 	//nearest rotation and translation
 98 
 99 	var nextRotation = null;
100 	var prevRotation = null;
101 	var nextTranslation = null;
102 	var prevTranslation = null;
103 	var nextScalation = null;
104 	var prevScalation = null;
105 	var translatePart = 1;
106 	var rotatePart = 1;
107 	var scalePart = 1;
108 	
109 	if(boneRotations)
110 	{
111 		for(var ri = 0; ri < boneRotations.length && nextRotation == null; ri++)
112 		{
113 			if(boneRotations[ri]['time'] > currentTime)
114 			{
115 				nextRotation = boneRotations[ri]; 
116 				
117 				if(ri == 0)
118 				{
119 					prevRotation = boneRotations[boneRotations.length - 1];
120 				}
121 				else
122 				{
123 					prevRotation = boneRotations[ri - 1];
124 				}
125 			}
126 		}
127 		
128 		if(prevRotation && nextRotation)
129 		{
130 			prevState['r'] = prevRotation['angle'];
131 			nextState['r'] = nextRotation['angle'];
132 			parts['r'] = (currentTime - prevRotation['time']) / (nextRotation['time'] - prevRotation['time']);
133 		}
134 	}
135 	
136 	if(boneScalations)
137 	{
138 		for(var ri = 0; ri < boneScalations.length && nextScalation == null; ri++)
139 		{
140 			if(boneScalations[ri]['time'] > currentTime)
141 			{
142 				nextScalation = boneScalations[ri]; 
143 				
144 				if(ri == 0)
145 				{
146 					prevScalation = boneScalations[boneScalations.length - 1];
147 				}
148 				else
149 				{
150 					prevScalation = boneScalations[ri - 1];
151 				}
152 			}
153 		}
154 		
155 		if(prevScalation && nextScalation)
156 		{
157 			prevState['sx'] = prevScalation['x'];
158 			nextState['sx'] = nextScalation['x'];
159 			prevState['sy'] = prevScalation['y'];
160 			nextState['sy'] = nextScalation['y'];
161 			parts['s'] = (currentTime - prevScalation['time']) / (nextScalation['time'] - prevScalation['time']);
162 		}
163 	}
164 	
165 	if(boneTranslations)
166 	{
167 		for(var ti = 0; ti < boneTranslations.length && nextTranslation == null; ti++)
168 		{
169 			if(boneTranslations[ti]['time'] > currentTime)
170 			{
171 				nextTranslation = boneTranslations[ti]; 
172 				
173 				if(ti == 0)
174 				{
175 					prevTranslation = boneTranslations[boneTranslations.length - 1];
176 				}
177 				else
178 				{
179 					prevTranslation = boneTranslations[ti - 1];
180 				}
181 			}
182 		}
183 		
184 		if(prevTranslation && nextTranslation)
185 		{
186 			prevState['x'] = prevTranslation['x'];
187 			nextState['x'] = nextTranslation['x'];
188 			prevState['y'] = prevTranslation['y'];
189 			nextState['y'] = nextTranslation['y'];
190 			parts['xy'] = (currentTime - prevTranslation['time']) / (nextTranslation['time'] - prevTranslation['time']);
191 		}
192 	}
193 	
194 	//TODO: Add curve
195 	
196 	if(!playReverse)
197 	{
198 		this.interpolateBoneStates(prevState, nextState, parts, curves);
199 	}
200 	else
201 	{
202 		parts['xy'] = 1 - parts['xy'];
203 		parts['r'] = 1 - parts['r'];
204 		parts['s'] = 1 - parts['s'];
205 		this.interpolateBoneStates(nextState, prevState, parts, curves);
206 	}
207 };
