 50 /**
 51  * Load resources of the specified types
 52  * 
 53  * <pre>
 54  * example of use:
 55  * sample.App = new sample.Game();
 56  * ss2d.ResourceManager.loadResources(
 57  * [
 58  *    ss2d.ResourceManager.Loaders.TEXTURE,  //next elements will be loaded as textures
 59  *   'textures/ball01.png',
 60  *   'textures/ball02.png',
 61  *    ss2d.ResourceManager.Loaders.SOUND,  //next elements will be loaded as sounds
 62  *   'sounds/clink.wav',
 63  *   'sounds/pom.wav',
 64  *   'sounds/door.wav',
 65  *    sample.CustomLoadr,  //next elements will be loaded as a custom type
 66  *   'customAssets/customAsset01.json
 67  * ],
 68  * //For each element loaded callback
 69  * function(resourceName, numLoaded, totalToLoad)  
 70  * {
 71  *   console.log(resourceName+' '+numLoaded+'/'+totalToLoad);
 72  * },
 73  * //Resource loading ends callback
 74  * function()  
 75  * {
 76  *   sample.App.initScene();
 77  *   sample.App.run();
 78  * });
 79  * </pre>
 80  * @param {(Array)} resources List of resource loaders plus resource names to load.
 81  * @param {function(string, number, number)} oneElementLoadedCallback Called for each loaded resource with the element name,
 82  * 																		  the number of loaded elements, and the total elements to load.
 83  * @param {function} elementsLoadedCallback  Called once all resources are loaded.
 84  */
 85 ss2d.ResourceManager.loadResources = function(resources, oneElementLoadedCallback, elementsLoadedCallback)
 86 {
 87 	if(resources instanceof Array)
 88 	{
 89 		ss2d.ResourceManager.ELEMENTS_LOADED_CALLBACK = elementsLoadedCallback || function(){};
 90 		ss2d.ResourceManager.ONE_ELEMENT_LOADED_CALLBACK = oneElementLoadedCallback || function(){};
 91 		ss2d.ResourceManager.CURRENT_ELEMENT_TO_LOAD = 0;
 92 		
 93 		var loaderKeys = 0;
 94 		
 95 		for(resKey in resources)
 96 		{
 97 			if(typeof resources[resKey] == 'object')
 98 			{
 99 				++loaderKeys;
100 			}
101 		}
102 		
103 		ss2d.ResourceManager.ELEMENTS_TO_LOAD = resources.length - loaderKeys;
104 		
105 		for(resKey in resources)
106 		{
107 			var resName = resources[resKey];
108 			if(typeof resName == 'object')
109 			{
110 				ss2d.ResourceManager.ACTIVE_LOADER = resName;
111 			}
112 			else
113 			{
114 				ss2d.ResourceManager.loadResourceWithLoader(ss2d.ResourceManager.ACTIVE_LOADER, 
115 															resName, 
116 															ss2d.ResourceManager.loadEndsCallback);
117 			}
118 		}
119 		ss2d.ResourceManager.ACTIVE_LOADER = function(res){ throw 'No active loader for: '+res; };
120 	}
121 };

123 /**
124  * Loads a resource with the specified loader.
125  * @param {ss2d.ILoader} loader The static class that loads this resource
126  * @param {string} name Resource file name
127  * @param {function=} loadedCallback
128  */
129 ss2d.ResourceManager.loadResourceWithLoader = function(loaderClass, resourceName, loadedCallback)
130 {
131 	var lrArray = ss2d.ResourceManager.LOADED_RESOURCES;
132 	var resHash = resourceName+'.'+loaderClass.RESOURCE_EXTENSION;
133 	if(!lrArray[resHash])
134 	{	
135 		lrArray[resHash] = loaderClass.loadResource(resourceName, loadedCallback);
136 	}
137 	
138 	return lrArray[resHash];
139 };
140 
141 /**
142  * Called by loadResources every time a resource is loaded
143  * @param {Object} resource
144  */
145 ss2d.ResourceManager.loadEndsCallback = function(resource)
146 {
147 	++ss2d.ResourceManager.CURRENT_ELEMENT_TO_LOAD;
148 	
149 	ss2d.ResourceManager.ONE_ELEMENT_LOADED_CALLBACK(resource.mName, ss2d.ResourceManager.CURRENT_ELEMENT_TO_LOAD, ss2d.ResourceManager.ELEMENTS_TO_LOAD);
150 	
151 	if(ss2d.ResourceManager.ELEMENTS_TO_LOAD == ss2d.ResourceManager.CURRENT_ELEMENT_TO_LOAD)
152 	{
153 		console.log('Load resources ends');
154 		ss2d.ResourceManager.ELEMENTS_LOADED_CALLBACK.call();
155 		
156 		ss2d.ResourceManager.ELEMENTS_LOADED_CALLBACK = function(){};
157 		ss2d.ResourceManager.ONE_ELEMENT_LOADED_CALLBACK = function(){};
158 	}
159 };
