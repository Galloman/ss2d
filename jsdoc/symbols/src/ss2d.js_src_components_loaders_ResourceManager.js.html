 58 /**
 59  * Load resources of the specified types
 60  * 
 61  * <pre>
 62  * example of use:
 63  * sample.App = new sample.Game();
 64  * ss2d.ResourceManager.loadResources(
 65  * [
 66  *    ss2d.ResourceManager.Loaders.TEXTURE,  //next elements will be loaded as textures
 67  *   'textures/ball01.png',
 68  *   'textures/ball02.png',
 69  *    ss2d.ResourceManager.Loaders.SOUND,  //next elements will be loaded as sounds
 70  *   'sounds/clink.wav',
 71  *   'sounds/pom.wav',
 72  *   'sounds/door.wav',
 73  *    sample.CustomLoadr,  //next elements will be loaded as a custom type
 74  *   'customAssets/customAsset01.json
 75  * ],
 76  * //For each element loaded callback
 77  * function(resourceName, numLoaded, totalToLoad)  
 78  * {
 79  *   console.log(resourceName+' '+numLoaded+'/'+totalToLoad);
 80  * },
 81  * //Resource loading ends callback
 82  * function()  
 83  * {
 84  *   sample.App.initScene();
 85  *   sample.App.run();
 86  * });
 87  * </pre>
 88  * @param {(Array)} resources List of resource loaders plus resource names to load.
 89  * @param {function(string, number, number)} oneElementLoadedCallback Called for each loaded resource with the element name,
 90  * 																		  the number of loaded elements, and the total elements to load.
 91  * @param {function} elementsLoadedCallback  Called once all resources are loaded.
 92  */
 93 ss2d.ResourceManager.loadResources = function(resources, oneElementLoadedCallback, elementsLoadedCallback)
 94 {
 95 	if(resources instanceof Array)
 96 	{
 97 		ss2d.ResourceManager.ELEMENTS_LOADED_CALLBACK = elementsLoadedCallback || function(){};
 98 		ss2d.ResourceManager.ONE_ELEMENT_LOADED_CALLBACK = oneElementLoadedCallback || function(){};
 99 		ss2d.ResourceManager.CURRENT_ELEMENT_TO_LOAD = 0;
100 		
101 		if(resources.length == 0)
102 		{
103 			ss2d.ResourceManager.ELEMENTS_LOADED_CALLBACK.call();
104 			ss2d.ResourceManager.ELEMENTS_LOADED_CALLBACK = function(){};
105 			ss2d.ResourceManager.ONE_ELEMENT_LOADED_CALLBACK = function(){};
106 		}
107 		
108 		var loaderKeys = 0;
109 		
110 		for(resKey in resources)
111 		{
112 			if(typeof resources[resKey] == 'object')
113 			{
114 				++loaderKeys;
115 			}
116 		}
117 		
118 		ss2d.ResourceManager.ELEMENTS_TO_LOAD = resources.length - loaderKeys;
119 		
120 		for(resKey in resources)
121 		{
122 			var resName = resources[resKey];
123 			if(typeof resName == 'object')
124 			{
125 				ss2d.ResourceManager.ACTIVE_LOADER = resName;
126 			}
127 			else
128 			{
129 				ss2d.ResourceManager.loadResourceWithLoader(ss2d.ResourceManager.ACTIVE_LOADER, 
130 															resName, 
131 															ss2d.ResourceManager.loadEndsCallback);
132 			}
133 		}
134 		ss2d.ResourceManager.ACTIVE_LOADER = function(res){ throw 'No active loader for: '+res; };
135 	}
136 };

138 /**
139  * Loads a resource with the specified loader.
140  * @param {ss2d.ILoader} loader The static class that loads this resource
141  * @param {string} name Resource file name
142  * @param {function=} loadedCallback
143  */
144 ss2d.ResourceManager.loadResourceWithLoader = function(loaderClass, resourceName, loadedCallback, callbackTarget)
145 {
146 	var lrArray = ss2d.ResourceManager.LOADED_RESOURCES;
147 	var resHash = resourceName+'.'+loaderClass.RESOURCE_EXTENSION;
148 	if(!lrArray[resHash])
149 	{	
150 		lrArray[resHash] = loaderClass.loadResource(resourceName, loadedCallback, callbackTarget||null);
151 	}
152 	else if(loadedCallback)
153 	{
154 		loadedCallback.call(callbackTarget, lrArray[resHash]);
155 	}
156 	
157 	return lrArray[resHash];
158 };
159 
160 /**
161  * Called by loadResources every time a resource is loaded
162  * @param {Object} resource
163  */
164 ss2d.ResourceManager.loadEndsCallback = function(resource)
165 {
166 	++ss2d.ResourceManager.CURRENT_ELEMENT_TO_LOAD;
167 	
168 	ss2d.ResourceManager.ONE_ELEMENT_LOADED_CALLBACK(resource.mName, ss2d.ResourceManager.CURRENT_ELEMENT_TO_LOAD, ss2d.ResourceManager.ELEMENTS_TO_LOAD);
169 	
170 	if(ss2d.ResourceManager.ELEMENTS_TO_LOAD == ss2d.ResourceManager.CURRENT_ELEMENT_TO_LOAD)
171 	{
172 		console.log('Load resources ends');
173 		ss2d.ResourceManager.ELEMENTS_LOADED_CALLBACK.call();
174 		
175 		ss2d.ResourceManager.ELEMENTS_LOADED_CALLBACK = function(){};
176 		ss2d.ResourceManager.ONE_ELEMENT_LOADED_CALLBACK = function(){};
177 	}
178 };
