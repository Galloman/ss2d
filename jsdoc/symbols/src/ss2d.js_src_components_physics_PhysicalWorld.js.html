 18 /** @constructor */
 19 ss2d.PhysicalWorld = function()
 20 {
 21 	var worldAABB = new box2d.AABB();  
 22 	worldAABB.minVertex.Set(-2048, -2048);  
 23 	worldAABB.maxVertex.Set(2048, 2048);  
 24 	var gravity = new box2d.Vec2(0, 300);  
 25 	var doSleep = true;  
 26 	this.mWorld = new box2d.World(worldAABB, gravity, doSleep);
 27 };
 28 
 29 /** @type {ss2d.PhysicalWorld} Current world instance*/
 30 ss2d.PhysicalWorld.WORLD_INSTANCE = null;
 31 
 32 /** @type {number} world updates per second */
 33 ss2d.PhysicalWorld.UPDATE_RATE = 30;
 34 
 35 ss2d.PhysicalWorld.getWorld = function()
 36 {
 37 	if(!ss2d.PhysicalWorld.WORLD_INSTANCE)
 38 	{
 39 		ss2d.PhysicalWorld.WORLD_INSTANCE = new ss2d.PhysicalWorld();
 40 	}
 41 	
 42 	return ss2d.PhysicalWorld.WORLD_INSTANCE;
 43 }
 44 
 45 /**
 46  * Sets the world gravity
 47  * @param {number} x Gravity force in X axis
 48  * @param {number} y Gravity force in Y axis
 49  */
 50 ss2d.PhysicalWorld.prototype.setGravity = function(x, y)
 51 {
 52 	this.mWorld.m_gravity.x = x;
 53 	this.mWorld.m_gravity.y = y;
 54 };
 55 
 56 /**
 57  * @param {number} deltaTime Time passed since the last update
 58  */
 59 ss2d.PhysicalWorld.prototype.tick = function(deltaTime)  
 60 {  
 61     this.mWorld.Step(deltaTime, 1);  
 62 };
 63 
 64 /**
 65  * Creates and returns a circle body
 66  * @param {number} x
 67  * @param {number} y
 68  * @param {number} radius
 69  * @return {Object} A ball
 70  */
 71 ss2d.PhysicalWorld.prototype.createBall = function(x, y, radius) {  
 72     radius = radius || 20;  
 73     var ballSd = new box2d.CircleDef();  
 74     ballSd.density = 1.0;  
 75     ballSd.radius = radius;  
 76     ballSd.restitution = 0.8;  
 77     ballSd.friction = 0.9;  
 78     var ballBd = new box2d.BodyDef();  
 79     ballBd.AddShape(ballSd);  
 80     ballBd.position.Set(x, y);  
 81     return this.mWorld.CreateBody(ballBd);  
 82 };  
 83 
 84 /**
 85  * Creates and returns a box body
 86  * @param {number} x Origin of the box in X axis
 87  * @param {number} y Origin of the box in Y axis
 88  * @param {number} width
 89  * @param {number} height
 90  * @param {boolean} fixed
 91  * @param {boolean} filled
 92  * @return {Object} A box
 93  */
 94 ss2d.PhysicalWorld.prototype.createBox = function(x, y, width, height, fixed, filled) 
 95 {  
 96 	if (typeof(fixed) == 'undefined') {  
 97 		fixed = true;  
 98 	}
 99 	 
100 	if (typeof(filled) == 'undefined') {  
101 		filled = false;  
102 	} 
103     
104 	var boxSd = new box2d.BoxDef();  
105 	if (!fixed) {  
106 	    boxSd.density = 1.0;  
107 	}  
108 	if (filled) {  
109 	    boxSd.userData = 'filled';  
110 	}  
111 	boxSd.extents.Set(width, height);  
112 	var boxBd = new box2d.BodyDef();  
113 	boxBd.AddShape(boxSd);  
114 	boxBd.position.Set(x, y);  
115 	return this.mWorld.CreateBody(boxBd);  
116 };

118 /**
119  * Create a mouse joint for target body
120  * @param {ss2d.RigidBody} body Target body
121  * @param {ss2d.InputProxy} playerInput multiplayer input
122  */
123 ss2d.PhysicalWorld.prototype.createMouseJoint = function(rigidBody, playerInput) 
124 {  
125 	var input = playerInput||ss2d.CURRENT_VIEW.mInput;
126 	md = new box2d.MouseJointDef();
127 	md.body1 = this.mWorld.m_groundBody;
128 	md.body2 = rigidBody.mBody;
129 	var localMouse = rigidBody.mOwner.worldToLocal(input.mMousePoint);
130 	md.target.Set(localMouse.mX, localMouse.mY);
131 	md.maxForce = 1000.0 * rigidBody.mBody.m_mass;
132 	md.m_collideConnected = true;
133 	md.timeStep = 1.0/ss2d.PhysicalWorld.UPDATE_RATE;
134 	return this.mWorld.CreateJoint(md);
135 };
