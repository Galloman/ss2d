 11 goog.require('ss2d.Skeleton');

 13 /**
 14  * @constructor
 15  * @param {string} SkeletalAnimationFileName
 16  */
 17 ss2d.SkeletalAnimation = function(animationFileName, callbackFunction, callbackTarget)
 18 {
 19 	if(!animationFileName)
 20 		return;
 21 		
 22 	this.mName = animationFileName;
 23 	this.mCallbackFunction = callbackFunction;
 24 	this.mCallbackTarget = callbackTarget||null;
 25 	
 26 	var animFileRequest = new XMLHttpRequest();
 27 	animFileRequest.mSkeletalAnimation = this;
 28 	animFileRequest.open("GET", animationFileName, true);
 29 	animFileRequest.responseType = "text";
 30 	animFileRequest.onload = function() 
 31 	{
 32 		this.mSkeletalAnimation.animationDataLoaded(this.response);
 33 	};
 34 	
 35 	animFileRequest.send();
 36 };
 37 
 38 ss2d.SkeletalAnimation.prototype.animationDataLoaded = function(data)
 39 {
 40 	this.mAnimationData = JSON.parse(data);
 41 	ss2d.Skeleton.allDegToRad(this.mAnimationData);
 42 	this.mDuration = ss2d.SkeletalAnimation.findMaxTime(this.mAnimationData, 0);
 43 	
 44 	if(this.mCallbackFunction)
 45 		this.mCallbackFunction.call(this.mCallbackTarget, this);
 46 };
 47 
 48 ss2d.SkeletalAnimation.prototype.setFromSkeleton = function(animationName)
 49 {
 50 	this.mAnimationData = animationName;
 51 	this.mDuration = ss2d.SkeletalAnimation.findMaxTime(this.mAnimationData, 0);
 52 };
 53 
 54 ss2d.SkeletalAnimation.findMaxTime = function(jsonObject, maxTime)
 55 {
 56 	maxTime = maxTime||0;
 57 	for(var key in jsonObject)
 58 	{
 59 		if(key == 'time')
 60 		{
 61 			maxTime = Math.max(jsonObject[key], maxTime);
 62 		}
 63 		else if(typeof jsonObject[key] == 'object')
 64 		{
 65 			maxTime = ss2d.SkeletalAnimation.findMaxTime(jsonObject[key], maxTime);
 66 		}
 67 	}
 68 	
 69 	return maxTime;
 70 };
