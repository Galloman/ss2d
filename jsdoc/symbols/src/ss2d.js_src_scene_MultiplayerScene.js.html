  3 /**
  4  * @fileoverview A subclass of ss2d.DisplayObjectContainer that implements some multiplayer features.
  5  * @author David Gallardo Moreno (portalg@gmail.com)
  6  */
  7 
  8 goog.provide('ss2d.MultiplayerScene');
  9 
 10 goog.require('ss2d.Defines');
 11 goog.require('ss2d.DisplayObjectContainer');
 12 
 13 /** 
 14  * @constructor 
 15  * @extends {ss2d.DisplayObjectContainer}
 16  * */
 17 ss2d.MultiplayerScene = function()
 18 {
 19 	ss2d.DisplayObjectContainer.call(this);
 20 	
 21 	//seted by the view
 22 	this.mParentView = null;
 23 }
 24 
 25 goog.inherits(ss2d.MultiplayerScene, ss2d.DisplayObjectContainer);
 26 
 27 ss2d.Object.assignClassId(ss2d.MultiplayerScene, 1006);
 28 
 29 if(COMPILING_CLIENT)
 30 {
 31 	/**
 32 	 * @override
 33 	 */
 34 	ss2d.MultiplayerScene.prototype.interpolateState = function(prevState, nextState, part, deltaTime, childrenOnly)
 35 	{
 36 		ss2d.DisplayObjectContainer.prototype.interpolateState.call(this, prevState, nextState, part, deltaTime, true)
 37 	}
 38 }
 39 
 40 if(COMPILING_SERVER)
 41 {
 42 	
 43 	/** @override */
 44 	ss2d.MultiplayerScene.prototype.tick = function(deltaTime)
 45 	{
 46 		this.tickChildren(deltaTime);
 47 		this.mParentView.mComunication.broadcastSceneToClients(this);
 48 	};
 49 	
 50 	/**
 51 	 * Server side method called every time a user connects.
 52  	 * @param {Object} connection
 53 	 */
 54 	ss2d.MultiplayerScene.prototype.onUserConnect = function(connection){};
 55 	
 56 	/**
 57 	 * Server side method called every time a user disconnects.
 58  	 * @param {Object} connection
 59 	 */
 60 	ss2d.MultiplayerScene.prototype.onUserDisconnect = function(connection){};
 61 	
 62 	/**
 63 	 * Server side method called every time a user change his or her name.
 64  	 * @param {Object} connection
 65 	 */
 66 	ss2d.MultiplayerScene.prototype.onUserChangeName = function(connection){};
 67 	
 68 	/**
 69 	 * Server side method called every time a user sent a message not intepreted by the communication interface.
 70 	 * @param {string} commandName
 71 	 * @param {Object} commandParam
 72 	 * @param {Object} connection
 73 	 */
 74 	ss2d.MultiplayerScene.prototype.onUserMessage = function(commandName, commandParam, connection){};
 75 	
 76 	/**
 77 	 * Return a JSON compliant string representation of the entire scene.
 78 	 * @override
 79 	 */
 80 	ss2d.MultiplayerScene.prototype.getPropertiesJSON = function()
 81 	{
 82 		return ss2d.DisplayObjectContainer.prototype.getPropertiesJSON.call(this, true, true);
 83 	}
 84 }
