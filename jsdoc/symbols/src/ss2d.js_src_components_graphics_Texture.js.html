 12 /**
 13  * @constructor
 14  * @param {string} textureFile The resource name
 15  * @param {function} callbackFunction Called when the texture is loaded
 16  */
 17 ss2d.Texture = function(textureFile, callbackFunction, callbackTarget)
 18 {	
 19 	//resource loading info
 20 	this.mName = textureFile;
 21 	
 22 	/** @type {Image} */
 23 	this.mTextureElement = new Image();
 24 	this.mTextureElement.mTexture = this;
 25 	this.mCallbackFunction = callbackFunction || function(){};
 26 	this.mCallbackTarget = callbackTarget||null;
 27 	this.mTextureElement.onload = function()
 28 	{ 
 29 		this.mTexture.handleLoadedTexture(this); 
 30 	};
 31 
 32 	this.mTextureElement.src = textureFile;
 33 };
 34 
 35 /**
 36  * Method called when the texture is loaded
 37  */
 38 ss2d.Texture.prototype.handleLoadedTexture = function(textureElement)
 39 {
 40 	if(this.mCallbackFunction)
 41 		this.mCallbackFunction.call(this.mCallbackTarget, this); 
 42 };

 94 /**
 95  * @return {number} the width of the texture in pixels
 96  */
 97 ss2d.Texture.prototype.getWidth = function()
 98 { 
 99 	return this.mTextureElement.width; 
100 };
101 
102 /**
103  * @return {number} the height of the texture in pixels
104  */
105 ss2d.Texture.prototype.getHeight = function()
106 { 
107 	return this.mTextureElement.height; 
108 };

110 ss2d.Texture.isPowerOfTwo = function(x) 
111 {
112     return (x & (x - 1)) == 0;
113 };
114  
115 ss2d.Texture.nextHighestPowerOfTwo = function(x) 
116 {
117     --x;
118     for (var i = 1; i < 32; i <<= 1) {
119         x = x | x >> i;
120     }
121     return x + 1;
122 };
