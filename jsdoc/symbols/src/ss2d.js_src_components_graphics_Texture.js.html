 12 /**
 13  * @constructor
 14  * @param {string} textureFile The resource name
 15  * @param {function} callbackFunction Called when the texture is loaded
 16  */
 17 ss2d.Texture = function(textureFile, callbackFunction, callbackTarget)
 18 {	
 19 	//resource loading info
 20 	this.mName = textureFile;
 21 	
 22 	/** @type {Image} */
 23 	this.mTextureElement = new Image();
 24 	this.mTextureElement.mTexture = this;
 25 	this.mCallbackFunction = callbackFunction || function(){};
 26 	this.mCallbackTarget = callbackTarget||null;
 27 	this.mTextureElement.onload = function()
 28 	{ 
 29 		this.mTexture.handleLoadedTexture(); 
 30 	};
 31 
 32 	this.mTextureElement.src = textureFile;
 33 }
 34 
 35 /**
 36  * Method called when the texture is loaded
 37  */
 38 ss2d.Texture.prototype.handleLoadedTexture = function()
 39 {
 40 	this.mCallbackFunction.call(this.mCallbackTarget, this); 
 41 	//if(RENDER_CONTEXT == 'webgl')
 42 	//{
 43 		//create texture in webgl
 44 	//}
 45 };

 47 /**
 48  * @return {number} the width of the texture in pixels
 49  */
 50 ss2d.Texture.prototype.getWidth = function()
 51 { 
 52 	return this.mTextureElement.width; 
 53 };
 54 
 55 /**
 56  * @return {number} the height of the texture in pixels
 57  */
 58 ss2d.Texture.prototype.getHeight = function()
 59 { 
 60 	return this.mTextureElement.height; 
 61 };
