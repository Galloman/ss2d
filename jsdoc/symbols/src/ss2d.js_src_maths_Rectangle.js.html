 11 /**
 12  * @constructor 
 13  * @param {number} x
 14  * @param {number} y
 15  * @param {number} w
 16  * @param {number} h
 17  */
 18 ss2d.Rectangle = function(x,y,w,h)
 19 {
 20 	this.mX = x;
 21 	this.mY = y;
 22 	this.mWidth = w;
 23 	this.mHeight = h;	
 24 }
 25 
 26 /**
 27  * @param {number} x
 28  * @param {number} y
 29  * @param {number} w
 30  * @param {number} h
 31  * @return {ss2d.Rectangle} this object
 32  */
 33 ss2d.Rectangle.prototype.setValues = function(x,y,w,h)
 34 {
 35 	this.mX = x;
 36 	this.mY = y;
 37 	this.mWidth = w;
 38 	this.mHeight = h;
 39 	
 40 	return this;
 41 }
 42 
 43 /**
 44  * @return {ss2d.Rectangle} a copy of this object
 45  */
 46 ss2d.Rectangle.prototype.clone = function()
 47 {
 48 	return new ss2d.Rectangle(this.mX, this.mY, this.mWidth, this.mHeight);
 49 }
 50 
 51 /**
 52  * @return {string}
 53  */
 54 ss2d.Rectangle.prototype.toString = function()
 55 {
 56 	return "("+this.mX+","+this.mY+","+this.mWidth+","+this.mHeight+")";
 57 }

 59 /**
 60  * Determines if a coordinate is within the rectangle.
 61  * @param {number} x
 62  * @param {number} y
 63  * @return {boolean}
 64  */
 65 ss2d.Rectangle.prototype.contains = function(x, y)
 66 {
 67 	return  x >= this.mX && y >= this.mY && x <= this.mX + this.mWidth && y <= this.mY + this.mHeight;
 68 }
 69 
 70 /**
 71  * Determines if a point is within the rectangle.
 72  * @param {ss2d.Point} point
 73  * @return {boolean}
 74  */
 75 ss2d.Rectangle.prototype.containsPoint = function(point)
 76 {
 77 	return this.contains(point.mX, point.mY);	
 78 }

 80 /**
 81  * Determines if another rectangle is within the rectangle.
 82  * @param {ss2d.Rectangle} otherRectangle
 83  * @return {boolean}
 84  */
 85 ss2d.Rectangle.prototype.containsRectangle = function(otherRectangle)
 86 {
 87 	return (otherRectangle.mX >= this.mX && otherRectangle.mX + otherRectangle.mWidth <= this.mX + this.mWidth &&
 88 		otherRectangle.mY >= this.mY && otherRectangle.mY + otherRectangle.mHeight <= this.mY + this.mHeight);
 89 }

 91 /**
 92  * Determines if another rectangle contains or intersects the rectangle.
 93  * @param {ss2d.Rectangle} otherRectangle
 94  * @return {boolean}
 95  */
 96 ss2d.Rectangle.prototype.intersectsRectangle = function(otherRectangle)
 97 {
 98 	var rX = otherRectangle.mX;
 99     var rY = otherRectangle.mY;
100     var rWidth = otherRectangle.mWidth;
101     var rHeight = otherRectangle.mHeight;
102     
103     var outside = (rX <= this.mX && rX + rWidth <= this.mX)  || 
104     			  (rX >= this.mX + this.mWidth && rX + rWidth >= this.mX + this.mWidth) ||
105 				  (rY <= this.mY && rY + rHeight <= this.mY) || 
106 				  (rY >= this.mY + this.mHeight && rY + rHeight >= this.mY + this.mHeight);
107 
108     return !outside;
109 };

111 /**
112  * If the specified rectangle intersects with the rectangle, returns the area of intersection.
113  * @param {ss2d.Rectangle} otherRectangle
114  * @param {ss2d.Rectangle} targetRectangle Empty rectangle to dump the result
115  * @return {ss2d.Rectangle} Area of intersection
116  */
117 ss2d.Rectangle.prototype.intersectionWithRectangle = function(otherRectangle, targetRectangle)
118 {
119 	if(!otherRectangle)
120 	{
121 		return null;
122 	} 
123 	
124 	targetRectangle = targetRectangle || new ss2d.Rectangle(0.0, 0.0, 0.0, 0.0);
125 
126 	var left   = Math.max(this.mX, otherRectangle.mX);
127 	var right  = Math.min(this.mX + this.mWidth, otherRectangle.mX + otherRectangle.mWidth);
128 	var top    = Math.max(this.mY, otherRectangle.mY);
129 	var bottom = Math.min(this.mY + this.mHeight, otherRectangle.mY + otherRectangle.mHeight);
130     
131 	if (!(left > right || top > bottom))
132 	{
133 		targetRectangle.setValues(left, top, right-left, bottom-top);
134 	}
135 	
136 	return targetRectangle;
137 }

139 /**
140  * Adds two rectangles together to create a new Rectangle object (by filling in the space between the two rectangles).
141  * @param {ss2d.Rectangle} otherRectangle
142  */
143 ss2d.Rectangle.prototype.uniteWithRectangle = function(otherRectangle)
144 {
145 	if(!targetRectangle)
146 	{
147 		targetRectangle = this.clone();
148 	} 
149 	
150 	if(!otherRectangle)
151 	{
152 		if(!targetRectangle){
153 			return targetRectangle;
154 		} else {
155 			return targetRectangle.setValues(this.mX, this.mY, this.mWidth, this.mHeight);
156 		}
157 	} 
158 
159 	var left   = Math.min(this.mX, otherRectangle.mX);
160 	var right  = Math.max(this.mX + this.mWidth, otherRectangle.mX + otherRectangle.mWidth);
161 	var top    = Math.min(this.mY, otherRectangle.mY);
162 	var bottom = Math.max(this.mY + this.mHeight, otherRectangle.mY + otherRectangle.mHeight);
163 	
164 	return targetRectangle.setValues(left, top, right - left, bottom - top);
165 }
