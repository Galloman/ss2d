  9 goog.provide('ss2d.ParticleSystem');
 10 
 11 goog.require('ss2d.Texture');
 12 
 13 /**
 14  * @constructor
 15  * @param {string} systemXMLFile The resource name
 16  * @param {function} callbackFunction Called when the ParticleSystem is loaded
 17  * @param {Object} callbackTarget
 18  */
 19 ss2d.ParticleSystem = function(systemXMLFile, callbackFunction, callbackTarget)
 20 {
 21 	this.mName = systemXMLFile;
 22 	this.mCallbackFunction = callbackFunction;
 23 	this.mCallbackTarget = callbackTarget;
 24 	this.mSystemDescriptor = null;
 25 	this.mTexture = null; 
 26 	
 27 	var rqst = new XMLHttpRequest();
 28 	rqst.mSystem = this;
 29 	rqst.open("GET", systemXMLFile, true);
 30 	rqst.responseType = "text";
 31 	rqst.onload = function() 
 32 	{
 33 		this.mSystem.loaded(this.response);
 34 	};
 35 	
 36 	rqst.send();
 37 };
 38 
 39 /**
 40  * @param {string} fileData
 41  */
 42 ss2d.ParticleSystem.prototype.loaded = function(fileData)
 43 {
 44 	this.mSystemDescriptor = (new XML.ObjTree()).parseXML(fileData)['particleEmitterConfig'];	
 45 	var pathEnd = this.mName.lastIndexOf('/') + 1;
 46 	var img = this.mSystemDescriptor['texture']['-name'];
 47 	var texturePath = (pathEnd > 0)?this.mName.substring(0, pathEnd)+img:img;
 48 	
 49 	this.mTexture = ss2d.CURRENT_VIEW.mRenderSupport.mAux8x8Texture;
 50 	
 51 	if(img.length > 0)
 52 		this.mTexture = ss2d.ResourceManager.loadTexture(texturePath, this.textureLoaded, this);
 53 };
 54 
 55 ss2d.ParticleSystem.prototype.textureLoaded = function(texture)
 56 {
 57 	this.mTexture = texture;
 58 	if(this.mCallbackFunction)
 59 		this.mCallbackFunction.call(this.mCallbackTarget, this);
 60 	
 61 };
