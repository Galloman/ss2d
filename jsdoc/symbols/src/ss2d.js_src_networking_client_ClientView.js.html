 17 /**
 18  * @constructor
 19  * @param {Object} canvasId Id of the canvas element where the scene will be rendered
 20  * @param {number=} canvasWidth Canvas width in pixels.
 21  * @param {number=} canvasHeight Canvas height in pixels.
 22  * @param {number=} frameRate Frames to render per second, 60 by default.
 23  * @param {number=} inputRate Input updates to send to the server per second, 20 by default.
 24  * @param {number=} delay Latency in milliseconds
 25  */
 26 ss2d.ClientView = function(canvasId, canvasWidth, canvasHeight, frameRate, inputRate, delay)
 27 {
 28 	this.mCanvas = document.getElementById(canvasId);
 29 	
 30 	if(RENDER_CONTEXT == 'webgl')
 31 	{
 32 		this.mContext = this.mCanvas.getContext('webgl')||this.mCanvas.getContext('experimental-webgl');
 33 		this.mProjection = ss2d.RenderSupport.orthogonalProjectionMatrix(0, this.mWidth, this.mHeight, 0, 1, -1);
 34 	}
 35 	else
 36 	{
 37 		this.mContext = this.mCanvas.getContext('2d'); 
 38 	}
 39 	
 40 	this.mRunning = false;
 41 	this.mFrameRate = frameRate || 60.0;
 42 	this.mInput = new ss2d.Input(this);
 43 	this.mLastFrameTimestamp = new Date().getTime();
 44 	
 45 	this.mCanvas.width = canvasWidth || this.mCanvas.height;
 46 	this.mCanvas.height = canvasHeight || this.mCanvas.height;
 47 	
 48 	ss2d.CURRENT_VIEW = this;	
 49 	ss2d.AUDIO_CONTEXT = ss2d.WebAudio.getAudioContext();
 50 	
 51 	/** @type {ss2d.DisplayObject} */
 52 	this.mMainScene = null;
 53 	
 54 	this.mRenderSupport = new ss2d.RenderSupport(this.mContext);
 55 	this.mBackgroundFillStyle = '#202020';
 56 	
 57 	this.mSceneQueue = []; //commands received by server
 58 	
 59 	this.mComunication = null;
 60 	
 61 	this.mDelay = delay || 100;
 62 	this.mInputRate = inputRate || 20.0;
 63 	
 64 	//plug-in development
 65 	this.mPreTickFunctions = [];
 66 	this.mPostTickFunctions = [];
 67 	this.mPreRenderFunctions = [];
 68 	this.mPostRenderFunctions = [];
 69 	
 70 	this.mTotalTime = 0;
 71 };

 74 ss2d.ClientView.NEXT_FRAME_CALLER = function(){ ss2d.CURRENT_VIEW.nextFrame() };
 75 
 76 /** @type {function} */
 77 ss2d.ClientView.NEXT_INPUT_UPDATE_CALLER = function(){ ss2d.CURRENT_VIEW.nextInputUpdate() };

 79 /**
 80  * Called every frame to update and render every object
 81  */
 82 ss2d.ClientView.prototype.nextFrame = function()
 83 {
 84 	//get current time in miliseconds
 85 	var now = new Date().getTime();
 86 	var timePassed = now - this.mLastFrameTimestamp;
 87 	
 88 	//called with canvas width and height every frame
 89 	this.resizeCanvas(this.mCanvas.width, this.mCanvas.height);
 90 	
 91 	var timePassedInSeconds = timePassed/1000.0;
 92 	
 93 	this.mTotalTime += timePassedInSeconds;
 94 	
 95 	//update input
 96 	this.mInput.tick(timePassedInSeconds);
 97 	
 98 	//interpolate scene
 99 	//if(this.mSceneQueue.length > 1)
100 	//{
101 		for(var methodIndex in this.mPreTickFunctions)
102 		{ 
103 			this.mPreTickFunctions[methodIndex].call(null, timePassedInSeconds); 
104 		}
105 	
106 		this.updateSceneState(now, timePassedInSeconds);
107 		
108 		for(var methodIndex in this.mPostTickFunctions)
109 		{ 
110 			this.mPostTickFunctions[methodIndex].call(null, timePassedInSeconds); 
111 		}
112 				
113 		for(var methodIndex in this.mPreRenderFunctions)
114 		{ 
115 			this.mPreRenderFunctions[methodIndex].call(null, this.mRenderSupport); 
116 		}
117 	
118 		//render scene
119 		if(this.mMainScene != null)
120 		{
121 			this.render();
122 		}
123 		
124 		for(var methodIndex in this.mPostRenderFunctions)
125 		{ 
126 			this.mPostRenderFunctions[methodIndex].call(null, this.mRenderSupport); 
127 		}
128 	//}
129 
130 	//calculate the delay time for the nextFrame call based on the 
131 	//frameRate and time spend in update and render operations.
132 	this.mLastFrameTimestamp = now;
133 	var wastedTime = (new Date().getTime() - now) / 1000.0;
134 	var nextFrameTime = Math.max(0.0, (1.0/this.mFrameRate) - wastedTime);
135 	
136 	//if game is running call the next frame with the specified time
137 	if(this.mRunning)
138 	{ 
139 		setTimeout(ss2d.ClientView.NEXT_FRAME_CALLER, nextFrameTime * 1000.0);
140 	}
141 };

143 /**
144  * Called every frame by nextFrame method to render the scene in the proper context.
145  */
146 ss2d.ClientView.prototype.render = function(){};
147 
148 if(RENDER_CONTEXT == 'webgl')
149 {
150 	ss2d.ClientView.prototype.render = function()
151 	{	
152 		var gl = this.mContext;
153 		gl.viewport(0, 0, this.mCanvas.width, this.mCanvas.height);
154 		var cc = this.mClearColor;
155 		gl.clearColor(cc[0], cc[1], cc[2], cc[3]);
156 		gl.clear(gl.COLOR_BUFFER_BIT);
157 		
158 		//render scene
159 		this.mMainScene.render(this.mRenderSupport);
160 	}
161 }
162 else
163 {
164 	ss2d.ClientView.prototype.render = function()
165 	{
166 		//clean background
167 		this.mContext.fillStyle = this.mBackgroundFillStyle;  
168 		this.mContext.fillRect(0, 0, this.mCanvas.width, this.mCanvas.height); 
169 
170 		//render scene
171 		this.mMainScene.render(this.mRenderSupport);
172 	};
173 }

175 /**
176  * Called in an interval specified by the inputRate property seconds and sends the user input to the server
177  */
178 ss2d.ClientView.prototype.nextInputUpdate = function()
179 {
180 	this.mComunication.packAndSendInput(this.mInput);
181 	if(this.mRunning)
182 	{ 
183 		setTimeout(ss2d.ClientView.NEXT_INPUT_UPDATE_CALLER, (1.0/this.mInputRate)*1000.0);
184 	}
185 }

187 /**
188  * Called by the nextFrame method to interpolate scene elements state.
189  * @param {number} currentTime
190  */
191 ss2d.ClientView.prototype.updateSceneState = function(currentTime, deltaTime)
192 {
193     var renderTime = currentTime - this.mDelay;
194     
195     var prevStateIndex = -1;
196     var nextStateIndex = -1;
197     
198     for(var i = 0; i < this.mSceneQueue.length && nextStateIndex == -1; i++)
199     {
200     	var stateTime = this.mSceneQueue[i][0];
201     	var command = this.mSceneQueue[i][1];
202 		var state = command['STATE'];
203 
204 		if(stateTime < renderTime)
205    	 	{
206    		 	prevStateIndex = i;
207    		 
208    		 	if(command['PLAYSOUNDS'] != null)
209    		 	{
210    			 	//play and remove unplayed sounds
211    			 	//playSounds(command['PLAYSOUNDS']);
212    			 	command['PLAYSOUNDS'] = null;
213    			 	delete command['PLAYSOUNDS'];
214    		 	}
215    	 	}
216    	 
217    	 	if(stateTime >= renderTime)
218    	 	{
219    		 	nextStateIndex = i;
220    	 	}
221     }
222     
223     //if(nextStateIndex == -1)
224     //{
225     	//nextStateIndex = prevStateIndex;
226     //}
227     
228     if(prevStateIndex != -1 && nextStateIndex != -1)
229     {
230     	//interpolate all previous states, then the current 
231     	for(var istates = 0; istates < prevStateIndex; ++istates)
232     	{
233     		this.interpolateSeceneStates(istates, prevStateIndex, renderTime, deltaTime);
234     	}
235     	
236     	this.interpolateSeceneStates(prevStateIndex, nextStateIndex, renderTime, deltaTime);
237     }
238     
239     this.mSceneQueue.splice(0, prevStateIndex);
240 }
241 
242 ss2d.ClientView.prototype.interpolateSeceneStates = function(prevStateIndex, nextStateIndex, renderTime, deltaTime)
243 {
244 	var prevStateTime = this.mSceneQueue[prevStateIndex][0];
245 	var nextStateTime = this.mSceneQueue[nextStateIndex][0];
246 	var fraction = (renderTime - prevStateTime) / (nextStateTime - prevStateTime); 
247 	
248 	if(this.mMainScene == null)
249 	{
250 		var sceneClass = ss2d.Object.getObjectPrototype(this.mSceneQueue[prevStateIndex][1]);
251 		this.mMainScene = sceneClass.createFromSerializedObject(this.mSceneQueue[prevStateIndex][1]);
252 		console.debug('main scene created');
253 	}
254 	
255 	this.mMainScene.interpolateState(this.mSceneQueue[prevStateIndex][1], this.mSceneQueue[nextStateIndex][1], fraction, deltaTime);
256 }

258 /**
259  * Start the main loop if it's ready.
260  */
261 ss2d.ClientView.prototype.startLoop = function()
262 {
263 	if(!this.mRunning && this.mComunication != null)
264 	{
265 		this.mRunning = true;
266 		this.nextFrame();
267 		this.nextInputUpdate();
268 	}	
269 };
270 
271 /**
272  * Pause the main loop
273  */
274 ss2d.ClientView.prototype.stopLoop = function()
275 {
276 	this.mRunning = false;	
277 };
