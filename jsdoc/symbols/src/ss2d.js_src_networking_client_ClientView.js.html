 17 /**
 18  * @constructor
 19  * @param {Object} canvasId Id of the canvas element where the scene will be rendered
 20  * @param {number=} canvasWidth Canvas width in pixels.
 21  * @param {number=} canvasHeight Canvas height in pixels.
 22  * @param {number=} frameRate Frames to render per second, 60 by default.
 23  * @param {number=} inputRate Input updates to send to the server per second, 20 by default.
 24  * @param {number=} delay Latency in milliseconds
 25  */
 26 ss2d.ClientView = function(canvasId, canvasWidth, canvasHeight, frameRate, inputRate, delay)
 27 {
 28 	this.mCanvas = document.getElementById(canvasId);
 29 	this.mContext = this.mCanvas.getContext('2d'); 
 30 	this.mRunning = false;
 31 	this.mFrameRate = frameRate || 60.0;
 32 	this.mInput = new ss2d.Input(this);
 33 	this.mLastFrameTimestamp = new Date().getTime();
 34 	
 35 	this.mCanvas.width = canvasWidth || this.mCanvas.height;
 36 	this.mCanvas.height = canvasHeight || this.mCanvas.height;
 37 	
 38 	ss2d.CURRENT_VIEW = this;
 39 	ss2d.ClientView.CANVAS_CONTEXT = this.mContext;
 40 	
 41 	ss2d.AUDIO_CONTEXT = ss2d.WebAudio.getAudioContext();
 42 	
 43 	/** @type {ss2d.DisplayObject} */
 44 	this.mMainScene = null;
 45 	
 46 	this.mRenderSupport = new ss2d.RenderSupport(this.mContext);
 47 	this.mBackgroundFillStyle = '#202020';
 48 	
 49 	this.mSceneQueue = []; //commands received by server
 50 	
 51 	this.mComunication = null;
 52 	
 53 	this.mDelay = delay || 100;
 54 	this.mInputRate = inputRate || 20.0;
 55 	window['wastedTimePerFrame'] = 0;
 56 };

 61 /** @type {function} */
 62 ss2d.ClientView.NEXT_FRAME_CALLER = function(){ ss2d.CURRENT_VIEW.nextFrame() };

 64 /** @type {function} */

 67 /**
 68  * Called every frame to update and render every object
 69  */
 70 ss2d.ClientView.prototype.nextFrame = function()
 71 {
 72 	//get current time in miliseconds
 73 	var now = new Date().getTime();
 74 	var timePassed = now - this.mLastFrameTimestamp;
 75 
 76 	//interpolate scene
 77 	if(this.mSceneQueue.length > 1)
 78 	{
 79 		this.updateSceneState(now);
 80 		
 81 		//clean background
 82 		this.mContext.fillStyle = this.mBackgroundFillStyle;  
 83 		this.mContext.fillRect(0, 0, this.mCanvas.width, this.mCanvas.height); 
 84 		
 85 		//render scene
 86 		if(this.mMainScene != null)
 87 		{
 88 			this.mMainScene.render(this.mRenderSupport);
 89 		}
 90 	}
 91 	
 92 	//update input
 93 	this.mInput.tick(timePassed/1000.0);
 94 
 95 	//calculate the delay time for the nextFrame call based on the 
 96 	//frameRate and time spend in update and render operations.
 97 	this.mLastFrameTimestamp = now;
 98 	var wastedTime = (new Date().getTime() - now) / 1000.0;
 99 	var nextFrameTime = Math.max(0.0, (1.0/this.mFrameRate) - wastedTime);
100 	
101 	//if game is running call the next frame with the specified time
102 	if(this.mRunning)
103 	{ 
104 		setTimeout(ss2d.ClientView.NEXT_FRAME_CALLER, nextFrameTime * 1000.0);
105 	}
106 };
107 
108 /**
109  * Called in an interval specified by the inputRate property seconds and sends the user input to the server
110  */
111 ss2d.ClientView.prototype.nextInputUpdate = function()
112 {
113 	this.mComunication.packAndSendInput(this.mInput);
114 	if(this.mRunning)
115 	{ 
116 		setTimeout(ss2d.ClientView.NEXT_INPUT_UPDATE_CALLER, (1.0/this.mInputRate)*1000.0);
117 	}
118 }

120 /**
121  * Called by the nextFrame method to interpolate scene elements state.
122  * @param {number} currentTime
123  */
124 ss2d.ClientView.prototype.updateSceneState = function(currentTime)
125 {
126     var renderTime = currentTime - this.mDelay;
127     
128     var prevStateIndex = -1;
129     var nextStateIndex = -1;
130     
131     for(var i = 0; i < this.mSceneQueue.length && nextStateIndex == -1; i++)
132     {
133     	var stateTime = this.mSceneQueue[i][0];
134     	var command = this.mSceneQueue[i][1];
135 		var state = command['STATE'];
136 
137 		if(stateTime < renderTime)
138    	 	{
139    		 	prevStateIndex = i;
140    		 
141    		 	if(command['PLAYSOUNDS'] != null)
142    		 	{
143    			 	//play and remove unplayed sounds
144    			 	//playSounds(command['PLAYSOUNDS']);
145    			 	command['PLAYSOUNDS'] = null;
146    			 	delete command['PLAYSOUNDS'];
147    		 	}
148    	 	}
149    	 
150    	 	if(stateTime >= renderTime)
151    	 	{
152    		 	nextStateIndex = i;
153    	 	}
154     }
155     
156     if(prevStateIndex != -1 && nextStateIndex != -1)
157     {
158     	//interpolate all previous states, then the current 
159     	for(var istates = 0; istates < prevStateIndex; ++istates)
160     	{
161     		this.interpolateSeceneStates(istates, prevStateIndex, renderTime);
162     	}
163     	
164     	this.interpolateSeceneStates(prevStateIndex, nextStateIndex, renderTime);
165     }
166     
167     this.mSceneQueue.splice(0, prevStateIndex);
168 }
169 
170 ss2d.ClientView.prototype.interpolateSeceneStates = function(prevStateIndex, nextStateIndex, renderTime)
171 {
172 	var prevStateTime = this.mSceneQueue[prevStateIndex][0];
173 	var nextStateTime = this.mSceneQueue[nextStateIndex][0];
174 	var fraction = (renderTime - prevStateTime) / (nextStateTime - prevStateTime); 
175 	
176 	if(this.mMainScene == null)
177 	{
178 		var sceneClass = ss2d.Object.getObjectPrototype(this.mSceneQueue[prevStateIndex][1]);
179 		this.mMainScene = sceneClass.createFromSerializedObject(this.mSceneQueue[prevStateIndex][1]);
180 		console.debug('main scene created');
181 	}
182 	
183 	this.mMainScene.interpolateState(this.mSceneQueue[prevStateIndex][1], this.mSceneQueue[nextStateIndex][1], fraction);
184 }

186 /**
187  * Start the main loop if it's ready.
188  */
189 ss2d.ClientView.prototype.startLoop = function()
190 {
191 	if(!this.mRunning && this.mComunication != null)
192 	{
193 		this.mRunning = true;
194 		this.nextFrame();
195 		this.nextInputUpdate();
196 	}	
197 };
198 
199 /**
200  * Pause the main loop
201  */
202 ss2d.ClientView.prototype.stopLoop = function()
203 {
204 	this.mRunning = false;	
205 };
