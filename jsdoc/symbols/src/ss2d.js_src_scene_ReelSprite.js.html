  3 /**
  4  * @fileoverview An ss2d.Sprite with a frame based animation.

  7 
  8 goog.provide('ss2d.ReelSprite');
  9 
 10 goog.require('ss2d.Sprite');
 11 goog.require('ss2d.ReelSet');
 12 goog.require('ss2d.ResourceManager');
 13 
 14 /**
 15  * @constructor
 16  * @param {number} x
 17  * @param {number} y
 18  * @param {number} w
 19  * @param {number} h
 20  * @param {string} reelSet
 21  * @param {string} playingReel
 22  */
 23 ss2d.ReelSprite = function(x, y, w, h, reelSet, playingReel)
 24 {
 25 
 26 	ss2d.Sprite.call(this, x, y, w, h);
 27 	
 28 	if(!reelSet)
 29 	{
 30 		throw 'Trying to create a sprite reel without specifying a ss2d.ReelSet';
 31 	}
 32 
 33 	//Animation info
 34 	this.mReelSetName = reelSet;
 35 	this.mReelSet = null;
 36 	this.mPlayingReel = null;
 37 	this.mDefaultReel = playingReel||false;	
 38 	this.mLoop = true;
 39 	this.mTimeDilation = 1.0;
 40 	
 41 	//Animation state
 42 	this.mPlaying = false;
 43 	this.mComplete = false;
 44 	this.mFrameCount = 0; 
 45 	this.mElapsedTimeCurrentFrame = 0;
 46 	
 47 	//Sprite
 48 	this.mTexture = null;
 49 	this.mClip = [0, 0, 0, 0];
 50 	this.mOffsetX = 0;
 51 	this.mOffsetY = 0;
 52 	
 53 	//setup
 54 	ss2d.ResourceManager.loadReelSet(reelSet, this.setup, this);
 55 
 56 };
 57 
 58 goog.inherits(ss2d.ReelSprite, ss2d.Sprite);
 59 
 60 //static
 61 ss2d.Object.assignClassId(ss2d.ReelSprite, 1005);
 62 
 63 /**
 64  * Prepare teh ReelSprite to play animations once the reelset is loaded. 
 65  */
 66 ss2d.ReelSprite.prototype.setup = function()
 67 {
 68 	this.mReelSet = ss2d.ResourceManager.loadReelSet(this.mReelSetName);
 69 	if(!this.mDefaultReel)
 70 	{
 71 		for(var reelName in this.mReelSet.mReels)
 72 		{
 73 			//first reel
 74 			this.playReel(reelName);
 75 			break;
 76 		}
 77 	} else {
 78 		this.playReel(this.mDefaultReel);
 79 	}
 80 	
 81 	this.mTexture = this.mReelSet.mTexture;
 82 	var firstFrame = this.mPlayingReel.mFrames[0];
 83 	firstFrame.dumpClip(this.mClip);
 84 }
 85 
 86 /**
 87  * @param {string} reelName
 88  */
 89 ss2d.ReelSprite.prototype.playReel = function(reelName)
 90 {
 91 	if(this.mReelSet)
 92 	{
 93 		this.mPlayingReel = (reelName && this.mReelSet.mReels[reelName]) ? this.mReelSet.mReels[reelName] : this.mPlayingReel;
 94 		var firstFrame = this.mPlayingReel.mFrames[0];
 95 		firstFrame.dumpClip(this.mClip);
 96 	}
 97 	
 98 	this.mPlaying = true;
 99 	this.mComplete = false;
100 	this.mFrameCount = 0; 
101 	this.mElapsedTimeCurrentFrame = 0;
102 }
103 
104 ss2d.ReelSprite.prototype.pauseReel = function()
105 {
106 	this.mPlaying = false;
107 }
108 
109 ss2d.ReelSprite.prototype.resumeReel = function()
110 {
111 	this.mPlaying = true;
112 }

114 if(COMPILING_CLIENT || COMPILING_OFFLINE)
115 {
116 	ss2d.ReelSprite.prototype.tick = function(deltaTime)
117 	{
118 		this.updateReelAnimation(deltaTime);
119 	};
120 	
121 	ss2d.ReelSprite.prototype.updateReelAnimation = function(deltaTime)
122 	{	
123 		if(this.mPlaying && this.mReelSet)
124 		{
125 			if(!this.mTexture)
126 			{
127 				this.mTexture = this.mReelSet.mTexture;
128 			}
129 			
130 			var timePerFrame = this.mPlayingReel.getTimePerFrame() * this.mTimeDilation;	
131 			
132 			this.mElapsedTimeCurrentFrame += deltaTime;
133 			
134 			//a frame change
135 			if(this.mElapsedTimeCurrentFrame >= timePerFrame)
136 			{
137 				var framesToAdd = Math.floor(this.mElapsedTimeCurrentFrame / timePerFrame);
138 				this.mFrameCount += framesToAdd;
139 				this.mElapsedTimeCurrentFrame -= timePerFrame;
140 				
141 				if(this.mFrameCount >= this.mPlayingReel.mFrames.length)
142 				{
143 					if(this.mLoop)
144 					{	
145 						this.mFrameCount = 0;
146 						this.mElapsedTimeCurrentFrame = 0;
147 					}
148 					else
149 					{
150 						this.mFrameCount--;
151 						this.mComplete = true;
152 						this.mPlaying = false;
153 					}
154 				}
155 				
156 				var currentFrame = this.mPlayingReel.mFrames[this.mFrameCount];
157 				if(currentFrame)
158 				{
159 					currentFrame.dumpClip(this.mClip);
160 					this.mOffsetX = currentFrame.mOffsetX;
161 					this.mOffsetY = currentFrame.mOffsetY;
162 				}
163 			}
164 		}
165 	};
166 	
167 	ss2d.ReelSprite.prototype.render = function(renderSupport)
168 	{
169 		if(this.mTexture)
170 		{
171 			this.mLocation.mX += this.mOffsetX;
172 			this.mLocation.mY += this.mOffsetY;
173 			ss2d.Sprite.prototype.render.call(this, renderSupport);
174 			this.mLocation.mX -= this.mOffsetX;
175 			this.mLocation.mY -= this.mOffsetY;
176 		}
177 	};
178 }
