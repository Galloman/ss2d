  8 goog.provide('ss2d.ReelSprite');
  9 
 10 goog.require('ss2d.Sprite');
 11 goog.require('ss2d.ReelFrame');
 12 goog.require('ss2d.ReelSet');
 13 
 14 /** @constructor */
 15 ss2d.ReelSprite = function(x, y, w, h, descriptor)
 16 {
 17 	//not Sprite constructor, we only need methods
 18 	ss2d.Quad.call(this, x, y, w, h);
 19 	
 20 	if(!descriptor)
 21 	{
 22 		throw 'Trying to create a sprite reel without a SpriteReelDescriptor';
 23 	}
 24 	
 25 	//Animation info
 26 	this.mDescriptor = descriptor;
 27 	this.mFirstFrame = this.mDescriptor.mFrames[0];
 28 	this.mLoop = true;
 29 	this.mTimeDilation = 1.0;
 30 	
 31 	//Sprite
 32 	this.mTexture = this.mFirstFrame.mTexture;
 33 	this.mClip = this.mFirstFrame.copyClip(new Array(4));
 34 	
 35 	//Animation state
 36 	this.mPlaying = true;
 37 	this.mComplete = false;
 38 	this.mFrameCount = 0; 
 39 	this.mElapsedTimeCurrentFrame = 0;
 40 };
 41 
 42 goog.inherits(ss2d.ReelSprite, ss2d.Sprite);
 43 
 44 //static
 45 ss2d.Object.assignClassId(ss2d.ReelSprite, 1005);
 46 
 47 //prototype
 48 ss2d.ReelSprite.prototype.playAnimation = function(descriptor)
 49 {
 50 	this.mDescriptor = descriptor || this.mDescriptor;
 51 	this.mPlaying = true;
 52 	this.mComplete = false;
 53 	this.mFrameCount = 0; 
 54 	this.mElapsedTimeCurrentFrame = 0;
 55 }
 56 
 57 ss2d.ReelSprite.prototype.pauseAnimation = function()
 58 {
 59 	this.mPlaying = false;
 60 }
 61 
 62 if(COMPILING_CLIENT || COMPILING_OFFLINE)
 63 {
 64 	ss2d.ReelSprite.prototype.tick = function(deltaTime)
 65 	{
 66 		if(this.mPlaying)
 67 		{
 68 			var timePerFrame = this.mDescriptor.getTimePerFrame() * this.mTimeDilation;	
 69 			
 70 			this.mElapsedTimeCurrentFrame += deltaTime;
 71 			
 72 			//a frame change
 73 			if(this.mElapsedTimeCurrentFrame >= timePerFrame)
 74 			{
 75 				this.mFrameCount++;
 76 				this.mElapsedTimeCurrentFrame -= timePerFrame;
 77 				
 78 				if(this.mFrameCount >= this.mDescriptor.mFrames.length)
 79 				{
 80 					if(this.mLoop)
 81 					{	
 82 						this.mFrameCount = 0;
 83 					}
 84 					else
 85 					{
 86 						this.mFrameCount--;
 87 						this.mComplete = true;
 88 					}
 89 				}
 90 				
 91 				var currentFrame = this.mDescriptor.mFrames[this.mFrameCount];
 92 				
 93 				currentFrame.dumpClip(this.mClip);
 94 				this.mTexture = currentFrame.mTexture;
 95 				this.mOffsetX = currentFrame.mOffsetX;
 96 				this.mOffsetY = currentFrame.mOffsetY;
 97 			}
 98 		}
 99 	};
100 }
