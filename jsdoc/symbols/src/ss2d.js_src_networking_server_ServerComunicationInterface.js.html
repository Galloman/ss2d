 14 /**
 15  * @constructor
 16  * @param {ss2d.ServerView} serverView
 17  * @param {number} port Port that will be opened
 18  * @param {number=} userLimit
 19  */
 20 ss2d.ServerCommunicationInterface = function(serverView, serverPort, userLimit)
 21 {
 22 	//TODO: create websocket externs file and replace the ['symbol'] trick
 23 	
 24 	this.mUserLimit = userLimit || 30;
 25 	this.mWebSocketModule = require('websocket')['server'];
 26 	this.mHTTPModule = require('http');
 27 	
 28 	//this.mConnection.mInterface = this;
 29 	
 30 	this.mServerView = serverView;
 31 	
 32 	this.mUserConnections = [];
 33 
 34 	this.mHTTPServer = this.mHTTPModule.createServer(function(request, response) 
 35 	{
 36 		
 37 	});
 38 	
 39 	var wsParams = {};
 40 	wsParams['httpServer'] = this.mHTTPServer;
 41 	this.mWebSocketServer = new this.mWebSocketModule(wsParams);
 42 	this.mWebSocketServer.mInterface = this;
 43 	
 44 	this.mWebSocketServer['on']('request', function(request) {
 45 		console.log('a user log in');
 46     	var connection = request['accept'](null, request['origin']);
 47     	
 48     	if(this.mInterface.mUserConnections.length >= this.mInterface.mUserLimit)
 49 		{
 50 			console.log('user rejected');
 51 			connection.send('["REJECTED","Too many players connected."]');
 52 			connection.close();
 53 			return;
 54 		}
 55     	
 56     	this.mInterface.mUserConnections.push(connection);
 57     	
 58     	
 59     	connection.mUserName = '';
 60     	connection.mServer = this;
 61     	connection.mInput = new ss2d.InputProxy();
 62     	connection.mConnectionId = ss2d.ServerCommunicationInterface.CONNECTIONS_COUNT++;
 63     	
 64     	this.mInterface.mServerView.mMainScene.onUserConnect(connection);
 65     	
 66     	connection['on']('message', function(message) 
 67     	{
 68     		//var commandPackage = JSON.parse(ss2d.Object.convertArrayBufferToString(data));
 69     		var commandPackage = JSON.parse(message['utf8Data']);
 70     		this.mServer.mInterface.processClientCommand(commandPackage, this);
 71     	});
 72     	
 73 		connection['on']('close', function(con) 
 74 		{
 75         	this.mServer.mInterface.mUserConnections.splice(this.mServer.mInterface.mUserConnections.indexOf(this), 1);
 76         	console.log('a user disconnect');
 77         	this.mServer.mInterface.mServerView.mMainScene.onUserDisconnect(this);
 78 		});
 79 	});
 80 	
 81 	this.mHTTPServer.listen(serverPort, function(){});
 82 	console.log('Server comunication initialized');
 83 };
 84 
 85 /** @type {number} */
 86 ss2d.ServerCommunicationInterface.CONNECTIONS_COUNT = 1000;

120 /**
121  * Called when the server receives a message from a client.
122  * @param {Object} commandPackage 
123  * @param {Object} connection
124  */
125 ss2d.ServerCommunicationInterface.prototype.processClientCommand = function(commandPackage, connection)
126 {
127 	var commandName = commandPackage[0];
128 	var commandParam = commandPackage[1];
129 	
130 	switch(commandName)
131 	{
132 		case 'UINPUT':
133 			connection.mInput.updateFromClient(commandParam);
134 		break;
135 		case 'UNAME':
136 			connection.mUserName = commandParam;
137 			connection.mServer.mInterface.mServerView.mMainScene.onUserChangeName(connection);
138 		break;
139 		default:
140 			connection.mServer.mInterface.mServerView.mMainScene.onUserMessage(commandName, commandParam, connection);
141 		break;
142 	}
143 };
