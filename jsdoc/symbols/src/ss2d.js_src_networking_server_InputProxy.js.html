  3 /**
  4  * @fileoverview Server side input representation for each user connected to the server.

  7 
  8 goog.provide('ss2d.InputProxy');
  9 
 10 /** 
 11  * @constructor 
 12  * @property {number} mMouseX Mouse location in x axis
 13  * @property {number} mMouseY Mouse location in y axis
 14  * @property {number} mPreviousMouseX Last frame mouse location in x axis
 15  * @property {number} mPreviousMouseY Last frame mouse location in y axis
 16  * @property {ss2d.Point} mMousePoint Mouse location point
 17  * @property {ss2d.Point} mPreviousMousePoint Last frame mouse location point
 18  * @property {boolean} mMouseDown Is mouse left button down?
 19  * @property {boolean} mPreviousMouseDown Was mouse left button down last frame?
 20  */
 21 ss2d.InputProxy = function()
 22 {
 23 	this.mPressedKeys = {};
 24 	this.mMouseX = -50;
 25 	this.mMouseY = -50;
 26 	this.mPreviousMouseX = 0;
 27 	this.mPreviousMouseY = 0;
 28 	this.mMousePoint = new ss2d.Point(this.mMouseX, this.mMouseY);
 29 	this.mPreviousMousePoint = new ss2d.Point(this.mPreviousMouseX, this.mPreviousMouseY);
 30 	this.mMouseDown = false;
 31 	this.mPreviousMouseDown = false;
 32 };
 33 
 34 /**
 35  * @param {Object} clientProxy Input received from the client
 36  */
 37 ss2d.InputProxy.prototype.updateFromClient = function(clientProxy)
 38 {
 39 	this.mPreviousMouseX = this.mMouseX;
 40 	this.mPreviousMouseY = this.mMouseY;
 41 	this.mPreviousMousePoint.mX = this.mPreviousMouseX;
 42 	this.mPreviousMousePoint.mY = this.mPreviousMouseY;
 43 	this.mPreviousMouseDown = this.mMouseDown;
 44 	this.mMouseDown = this.mClicked;
 45 	
 46 	this.mPressedKeys = clientProxy['keys'];
 47 	this.mMouseX = clientProxy['mx'];
 48 	this.mMouseY = clientProxy['my'];
 49 	this.mMouseDown = clientProxy['md'];
 50 	this.mMousePoint.mX = this.mMouseX;
 51 	this.mMousePoint.mY = this.mMouseY;
 52 };
 53 
 54 /**
 55  * @param {Object} keyCode ASCII key code
 56  * @return {boolean} Return true if the specified key is pressed
 57  */
 58 ss2d.InputProxy.prototype.isKeyPressed = function(keyCode)
 59 {
 60 	return (this.mPressedKeys['_'+keyCode] != null)?true:false;
 61 };
