  3 /**
  4  * @fileoverview Server side input representation for each user connected to the server.
  5  * @author David Gallardo Moreno (portalg@gmail.com)

  7 
  8 goog.provide('ss2d.InputProxy');
  9 
 10 /** @constructor */
 11 ss2d.InputProxy = function()
 12 {
 13 	this.mPressedKeys = {};
 14 	this.mMouseX = -50;
 15 	this.mMouseY = -50;
 16 	this.mPrevMouseX = 0;
 17 	this.mPrevMouseY = 0;
 18 	this.mMousePoint = new ss2d.Point(this.mMouseX, this.mMouseY);
 19 	this.mPrevMousePoint = new ss2d.Point(this.mPrevMouseX, this.mPrevMouseY);
 20 	this.mMouseDown = false;
 21 	this.mPreviousMouseDown = false;
 22 };
 23 
 24 /**
 25  * @param {Object} clientProxy Input received from the client
 26  */
 27 ss2d.InputProxy.prototype.updateFromClient = function(clientProxy)
 28 {
 29 	this.mPrevMouseX = this.mMouseX;
 30 	this.mPrevMouseY = this.mMouseY;
 31 	this.mPrevMousePoint.mX = this.mPrevMouseX;
 32 	this.mPrevMousePoint.mY = this.mPrevMouseY;
 33 	this.mPreviousMouseDown = this.mMouseDown;
 34 	this.mMouseDown = this.mClicked;
 35 	
 36 	this.mPressedKeys = clientProxy['keys'];
 37 	this.mMouseX = clientProxy['mx'];
 38 	this.mMouseY = clientProxy['my'];
 39 	this.mMouseDown = clientProxy['md'];
 40 	this.mMousePoint.mX = this.mMouseX;
 41 	this.mMousePoint.mY = this.mMouseY;
 42 };
 43 
 44 /**
 45  * @param {Object} keyCode ASCII key code
 46  * @return {boolean} Return true if the specified key is pressed
 47  */
 48 ss2d.InputProxy.prototype.isKeyPressed = function(keyCode)
 49 {
 50 	return (this.mPressedKeys['_'+keyCode] != null)?true:false;
 51 };
