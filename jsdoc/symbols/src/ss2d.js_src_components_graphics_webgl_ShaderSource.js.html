 10 /**
 11  * @constructor
 12  * @param {string} shaderFile The resource name
 13  * @param {function} shaderLoadedCallback Called when the shader file is loaded
 14  */
 15 ss2d.ShaderSource = function(shaderFile, shaderLoadedCallback, callbackTarget)
 16 {	
 17 	this.mShaderId = -1;
 18 	
 19 	//can be created from source
 20 	if(!shaderFile){ return; } 
 21 	
 22 	//resource loading info
 23 	this.mName = shaderFile;
 24 	this.mCallbackFunction = shaderLoadedCallback||null;
 25 	this.mCallbackTarget = callbackTarget||null;
 26 	
 27 	var shaderFileRequest = new XMLHttpRequest();
 28 	shaderFileRequest.mShaderSource = this;
 29 	shaderFileRequest.open("GET", atlasJSONFileName, true);
 30 	shaderFileRequest.responseType = "text";
 31 	shaderFileRequest.onload = function() 
 32 	{
 33 		var shaderType = ss2d.ShaderSource.getShaderType(this.mShaderSource.mName);
 34 		if(!shaderType)
 35 		{
 36 			throw 'Unable to find the shader type for: '+this.mName;	
 37 		}
 38 		this.mShaderSource.compileSource(this.response, shaderType);
 39 		
 40 		if(this.mCallbackFunction)
 41 		{
 42 			this.mCallbackFunction.call(this.mCallbackTarget, this.mShaderSource);
 43 		}
 44 	};
 45 	
 46 	shaderFileRequest.send();
 47 }
 48 
 49 ss2d.ShaderSource.VS_EXT = ['vert','vs','vertex'];
 50 ss2d.ShaderSource.FS_EXT = ['frag','fs','fragment'];
 51 
 52 /**
 53  * Method called when the shader source is loaded to compile it
 54  */
 55 ss2d.ShaderSource.prototype.compileSource = function(data, shaderType)
 56 {
 57 	var gl = ss2d.CURRENT_VIEW.mContext;
 58 
 59 	this.mShaderId = gl.createShader(shaderType);
 60 	gl.shaderSource(this.mShaderId, data);
 61 	gl.compileShader(this.mShaderId);
 62 	
 63 	if (!gl.getShaderParameter(this.mShaderId, gl.COMPILE_STATUS)) 
 64 	{
 65 		throw gl.getShaderInfoLog(this.mShaderId);
 66 	}
 67 
 68 	if(this.mCallbackFunction)
 69 	{
 70 		this.mCallbackFunction.call(null, this);
 71 	}
 72 	
 73 	return this;
 74 };
 75 
 76 /**
 77  * Retrieves the shader source type comparing the file extension.
 78  */
 79 ss2d.ShaderSource.getShaderType = function(shaderFile)
 80 {
 81 	var gl = ss2d.CURRENT_VIEW.mContext;
 82 	var ext = shaderFile.split('.').pop();
 83 	var shaderType = null;
 84 	
 85 	if(ss2d.ShaderSource.VS_EXT.indexOf(ext) != -1)
 86 	{
 87 		shaderType = gl.VERTEX_SHADER;
 88 	} 
 89 	else if(ss2d.ShaderSource.FS_EXT.indexOf(ext) != -1)
 90 	{
 91 		shaderType = gl.FRAGMENT_SHADER;
 92 	}
 93 	
 94 	return shaderType;
 95 };
