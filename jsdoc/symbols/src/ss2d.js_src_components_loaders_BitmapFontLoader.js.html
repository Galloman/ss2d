 27 /**
 28  * Load the texture
 29  * @param {string} resourceFileName
 30  * @param {function} resourceLoadedCallback
 31  * @return {ss2d.Texture} The loaded texture
 32  */

 34 {
 35 	return new ss2d.BitmapFont(resourceFileName, resourceLoadedCallback);
