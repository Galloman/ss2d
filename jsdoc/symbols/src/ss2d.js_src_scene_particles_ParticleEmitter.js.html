 16 ss2d.ParticleEmitter = function(x, y, w, h, particleSystem)
 17 {
 18 	
 19 	ss2d.Quad.call(this, x, y, w, h);
 20 	this.mPivotX = this.mWidth*0.5;
 21 	this.mPivotY = this.mHeight*0.5;
 22 	
 23 	if(RENDER_CONTEXT != 'webgl') 
 24 		throw 'ss2d.ParticleEmitter requires WebGL Rendering Context';
 25 		
 26 	if(!particleSystem)
 27 		throw 'No particle system provided';
 28 	
 29 	this.mParticleSystem = particleSystem;
 30 	this.mActive = false;
 31 	this.mShowHitBox = true;
 32 	
 33 	if(typeof particleSystem == 'string')
 34 		this.mParticleSystem = ss2d.ResourceManager.loadParticleSystem(particleSystem, this.setup, this);
 35 };
 36 
 37 goog.inherits(ss2d.ParticleEmitter, ss2d.Quad);

 41 ss2d.ParticleEmitter.prototype.setup = function(particleSystem)
 42 {
 43 	this.mParticleSystem = particleSystem;
 44 	this.mTexture = this.mParticleSystem.mTexture;
 45 	
 46 	var desc = this.mParticleSystem.mSystemDescriptor;
 47 	
 48 	this.mSourcePosition = new ss2d.Point(parseFloat(desc['sourcePosition']['-x']),
 49 										  parseFloat(desc['sourcePosition']['-y']));
 50 	this.mSourcePositionVariance = new ss2d.Point(parseFloat(desc['sourcePositionVariance']['-x']),
 51 										  		  parseFloat(desc['sourcePositionVariance']['-y']));		
 52 	this.mGravity = new ss2d.Point(parseFloat(desc['gravity']['-x']),
 53 								   parseFloat(desc['gravity']['-y']));
 54 								   
 55 	this.mAngle = parseFloat(desc['angle']['-value']); 
 56 	this.mAngleVariance = parseFloat(desc['angleVariance']['-value']);								
 57 	this.mSpeed = parseFloat(desc['speed']['-value']); 
 58 	this.mSpeedVariance = parseFloat(desc['speedVariance']['-value']);	
 59     this.mRadialAcceleration = parseFloat(desc['radialAcceleration']['-value']); 
 60     this.mTangentialAcceleration = parseFloat(desc['tangentialAcceleration']['-value']);
 61     this.mRadialAccelVariance = parseFloat(desc['radialAccelVariance']['-value']); 
 62     this.mTangentialAccelVariance = parseFloat(desc['tangentialAccelVariance']['-value']);
 63 	this.mParticleLifeSpan = Math.max(0.08, parseFloat(desc['particleLifeSpan']['-value'])); 
 64 	this.mParticleLifespanVariance = parseFloat(desc['particleLifespanVariance']['-value']);			
 65 	this.mStartParticleSize = parseFloat(desc['startParticleSize']['-value']); 
 66 	this.mStartParticleSizeVariance = parseFloat(desc['startParticleSizeVariance']['-value']);
 67 	this.mFinishParticleSize = parseFloat(desc['finishParticleSize']['-value']); 
 68 	var fpsvk = desc['finishParticleSizeVariance']||desc['FinishParticleSizeVariance'];
 69 	this.mFinishParticleSizeVariance = parseFloat(fpsvk['-value']);
 70 	this.mMaxParticles = parseFloat(desc['maxParticles']['-value']);
 71 	this.mDuration = parseFloat(desc['duration']['-value']);
 72     this.mRotationStart = parseFloat(desc['rotationStart']['-value']); 
 73     this.mRotationStartVariance = parseFloat(desc['rotationStartVariance']['-value']);
 74     this.mRotationEnd = parseFloat(desc['rotationEnd']['-value']); 
 75     this.mRotationEndVariance = parseFloat(desc['rotationEndVariance']['-value']);
 76     this.mBlendFuncSource = parseFloat(desc['blendFuncSource']['-value']);
 77     this.mBlendFuncDestination = parseFloat(desc['blendFuncDestination']['-value']);
 78     
 79     this.mStartColor = [parseFloat(desc['startColor']['-red']), 
 80     					parseFloat(desc['startColor']['-green']),
 81     					parseFloat(desc['startColor']['-blue']),
 82     					parseFloat(desc['startColor']['-alpha'])]; 
 83 	this.mStartColorVariance = [parseFloat(desc['startColorVariance']['-red']), 
 84 		    					parseFloat(desc['startColorVariance']['-green']),
 85 		    					parseFloat(desc['startColorVariance']['-blue']),
 86 		    					parseFloat(desc['startColorVariance']['-alpha'])];					
 87 	this.mFinishColor = [parseFloat(desc['finishColor']['-red']), 
 88     					 parseFloat(desc['finishColor']['-green']),
 89     					 parseFloat(desc['finishColor']['-blue']),
 90     					 parseFloat(desc['finishColor']['-alpha'])];
 91 	this.mFinishColorVariance = [parseFloat(desc['finishColorVariance']['-red']), 
 92 		    					 parseFloat(desc['finishColorVariance']['-green']),
 93 		    					 parseFloat(desc['finishColorVariance']['-blue']),
 94 		    					 parseFloat(desc['finishColorVariance']['-alpha'])];
 95     
 96     //These values are used for the special purpose of creating the spinning portal emitter.
 97     this.mMaxRadius = parseFloat(desc['maxRadius']['-value']);					// Max radius at which particles are drawn when rotating
 98 	this.mMaxRadiusVariance = parseFloat(desc['maxRadiusVariance']['-value']);			// Variance of the maxRadius
 99 	this.mMinRadius = parseFloat(desc['minRadius']['-value']);					// Radius from source below which a particle dies
100 	this.mRotatePerSecond = parseFloat(desc['rotatePerSecond']['-value']);				// Numeber of degress to rotate a particle around the source pos per second
101 	this.mRotatePerSecondVariance = parseFloat(desc['rotatePerSecondVariance']['-value']);		// Variance in degrees for rotatePerSecond
102 	this.mRadiusSpeed = 0; 		// The speed at which a particle moves from maxRadius to minRadius
103 	
104 	//control properties
105 	this.mRotation = - this.mAngle / (180.0 / Math.PI);
106 	this.mEmitterType = parseFloat(desc['emitterType']['-value']);
107 	this.mParticleCount = 0;
108 	this.mEmissionRate = this.mMaxParticles / this.mParticleLifeSpan;
109 	this.mEmitCounter = 0;	
110 	this.mElapsedTime = 0;
111 	
112 	
113 	this.mUseTexture = true;
114 	this.mParticleIndex = 0;
115 	this.mParticles = [];
116 	this.mParticleDataArray = new Float32Array(this.mMaxParticles);
117 	
118 	var worldLocation = this.localToWorld(this.mLocation);
119 	for(var i = 0; i < this.mMaxParticles; ++i)
120 	{
121 		this.mParticles[i] = this.generateParticle(null, worldLocation, i*0.008);
122 		this.mParticles[i].getPCBufferArray(this.mParticleDataArray, i * 24);
123 	}
124 	this.mParticleCount = this.mMaxParticles;
125 	var gl = ss2d.CURRENT_VIEW.mContext;
126 	var renderSupport = ss2d.CURRENT_VIEW.mRenderSupport;
127 	
128 	this.mParticleDataBuffer = renderSupport.createBuffer(gl.ARRAY_BUFFER, 
129 														  new Float32Array(this.mParticleDataArray), 
130 														  6,  this.mParticleDataArray.length / 6,
131 														  gl.DYNAMIC_DRAW);
132 	this.mFreeParticles = [];
133 	this.mReady = true;
134 	this.startEmitter();
135 };

152 ss2d.ParticleEmitter.prototype.generateParticle = function(particle, worldLocation, addedLifeTime)
153 {
154 	if(this.mParticleCount >= this.mMaxParticles)
155 		return;
156 		
157 	particle = particle||new ss2d.Particle();
158 	addedLifeTime = addedLifeTime||0;
159 	particle.mTimeStamp = ss2d.CURRENT_VIEW.mTotalTime;
160 	
161 	//var m1to1 = ss2d.ParticleEmitter.MINUS_ONE_TO_ONE;
162 	
163 	particle.mPosition.mX = worldLocation.mX + (((Math.random() * 2.0) - 1.0) * this.mSourcePositionVariance.mX);
164 	particle.mPosition.mY = worldLocation.mY + (((Math.random() * 2.0) - 1.0) * this.mSourcePositionVariance.mY);
165 	particle.mStartPos.mX = worldLocation.mX;
166 	particle.mStartPos.mY = worldLocation.mY;
167 	
168 	var newAngle = this.mRotation + (((Math.random() * 2.0) - 1.0) * this.mAngleVariance / (180 / Math.PI));
169 	var vector = new ss2d.Point(Math.cos(newAngle), Math.sin(newAngle));
170 	var vectorSpeed = this.mSpeed + (((Math.random() * 2.0) - 1.0) * this.mSpeedVariance);
171 	particle.mDirection = ss2d.Point.scalePoint(vector, vectorSpeed);
172 	particle.mRadius = this.mMaxRadius + (((Math.random() * 2.0) - 1.0) * this.mMaxRadiusVariance);
173 	
174 	particle.mRadiusDelta = (this.mMaxRadius / this.mParticleLifeSpan);// * (1.0 / ss2d.ParticleEmitter.MAX_UPDATE_RATE);
175 	particle.mAngle  = this.mRotation + (((Math.random() * 2.0) - 1.0) * this.mAngleVariance / (180 / Math.PI));
176 	particle.mDegreesPerSecond = (this.mRotatePerSecond + (((Math.random() * 2.0) - 1.0) *  this.mRotatePerSecondVariance)) / (180 / Math.PI);
177 	
178 	particle.mRadialAcceleration = this.mRadialAcceleration;
179 	particle.mTangentialAcceleration = this.mTangentialAcceleration;
180 	
181 	particle.mTimeToLive = Math.max(0, this.mParticleLifeSpan + (this.mParticleLifespanVariance * ((Math.random() * 2.0) - 1.0))) + addedLifeTime;
182 	
183 	var particleStartSize = this.mStartParticleSize + (this.mStartParticleSizeVariance * ((Math.random() * 2.0) - 1.0));
184 	var particleFinishSize = this.mFinishParticleSize + (this.mFinishParticleSizeVariance * ((Math.random() * 2.0) - 1.0));
185 	particle.mParticleSizeDelta = ((particleFinishSize - particleStartSize)  / this.mParticleLifeSpan) / this.mTexture.mTextureElement.width;
186 	particle.mParticleSize = Math.max(0, particleStartSize);
187 	
188 	var startColor = particle.mColor;
189 	startColor[0]  = this.mStartColor[0] + (this.mStartColorVariance[0] * ((Math.random() * 2.0) - 1.0));
190 	startColor[1]  = this.mStartColor[1] + (this.mStartColorVariance[1] * ((Math.random() * 2.0) - 1.0));
191 	startColor[2]  = this.mStartColor[2] + (this.mStartColorVariance[2] * ((Math.random() * 2.0) - 1.0));
192 	startColor[3]  = this.mStartColor[3] + (this.mStartColorVariance[3] * ((Math.random() * 2.0) - 1.0));
193 	
194 	var endColor = [];
195 	endColor[0]  = this.mFinishColor[0] + (this.mFinishColorVariance[0] * ((Math.random() * 2.0) - 1.0));
196 	endColor[1]  = this.mFinishColor[1] + (this.mFinishColorVariance[1] * ((Math.random() * 2.0) - 1.0));
197 	endColor[2]  = this.mFinishColor[2] + (this.mFinishColorVariance[2] * ((Math.random() * 2.0) - 1.0));
198 	endColor[3]  = this.mFinishColor[3] + (this.mFinishColorVariance[3] * ((Math.random() * 2.0) - 1.0));
199 	
200 	var deltaColor = particle.mDeltaColor;
201 	deltaColor[0] = ((endColor[0] - startColor[0]) / particle.mTimeToLive) / this.mParticleLifeSpan;
202 	deltaColor[1] = ((endColor[1] - startColor[1]) / particle.mTimeToLive) / this.mParticleLifeSpan;
203 	deltaColor[2] = ((endColor[2] - startColor[2]) / particle.mTimeToLive) / this.mParticleLifeSpan;
204 	deltaColor[3] = ((endColor[3] - startColor[3]) / particle.mTimeToLive) / this.mParticleLifeSpan;
205 	
206 	var startA = this.mRotationStart + (this.mRotationStartVariance * ((Math.random() * 2.0) - 1.0));
207 	var endA = this.mRotationEnd + (this.mRotationEndVariance * ((Math.random() * 2.0) - 1.0));
208 	
209 	particle.mRotation = startA;
210 	particle.mRotationDelta = (endA - startA) / this.mParticleLifeSpan;
211 	
212 	this.mParticleCount++;
213 	
214 	return particle;
215 };

219 	ss2d.ParticleEmitter.prototype.render = function(support)
220 	{
221 		if(!this.mReady)
222 			return;
223 			
224 		this.mPivotX = this.mWidth*0.5;
225 		this.mPivotY = this.mHeight*0.5;
226 
227 		var gl = support.mContext;
228 		
229 		gl.blendFunc(this.mBlendFuncSource, this.mBlendFuncDestination);
230 
231 		var material = support.mMaterials.mParticle;
232 		material.mActiveTexture = this.mTexture.mTextureId;
233 		material.mParticleDataBuffer = this.mParticleDataBuffer;
234 		material.mParticleDataArray = this.mParticleDataArray;
235 		
236 		material.apply(support);
237 		
238 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, support.mBuffers.mParticlesQVI);
239 		gl.drawElements(gl.TRIANGLES, this.mMaxParticles * 6, gl.UNSIGNED_SHORT, 0);
240 		
241 		gl.blendFunc(support.mDefaultBlendSource, support.mDefaultBlendDestination);
242 		
243 		if(this.mShowHitBox)
244 		{
245 			var prevAlpha = this.mAlpha;
246 			this.mAlpha = 0.5;
247 			ss2d.Quad.prototype.render.call(this, support);
248 			this.mAlpha = prevAlpha;
249 		}
250 	};

252 	ss2d.ParticleEmitter.prototype.tick = function(deltaTime)
253 	{
254 		this.updateParticles(deltaTime);
255 	};
256 	
257 	ss2d.ParticleEmitter.prototype.updateParticles = function(deltaTime)
258 	{
259 		if(!this.mReady)
260 			return;
261 		
262 		var gl = ss2d.CURRENT_VIEW.mContext;
263 		var renderSupport = ss2d.CURRENT_VIEW.mRenderSupport;
264 		if(this.mActive && this.mEmissionRate)
265 		{
266 			var rate = 1.0/this.mEmissionRate;
267 			this.mEmitCounter += deltaTime;
268 			var auxParticleArray = [];
269 			var worldLocation = this.localToWorld(this.mLocation);
270 			
271 			while(this.mParticleCount < this.mMaxParticles && this.mEmitCounter > rate)
272 			{
273 				this.mEmitCounter -= rate;
274 				
275 				if(this.mFreeParticles.length == 0)
276 					continue;
277 				
278 				
279 				var particleIndex = this.mFreeParticles.pop();
280 				var particle = this.generateParticle(this.mParticles[particleIndex], worldLocation);
281 				auxParticleArray = particle.getArray(auxParticleArray);
282 				
283 				//4 times, one per vertex
284 				auxParticleArray = auxParticleArray.concat(auxParticleArray);
285 				auxParticleArray = auxParticleArray.concat(auxParticleArray);
286 				
287 				gl.bindBuffer(gl.ARRAY_BUFFER, this.mParticleDataBuffer);
288 				gl.bufferSubData(gl.ARRAY_BUFFER, 
289 								 particleIndex * 4 * ss2d.Particle.SRUCT_SIZE * 4,
290 								 new Float32Array(auxParticleArray));
291 			}
292 			
293 			this.mElapsedTime += deltaTime;
294 			if(this.mDuration != -1 && this.mDuration < this.mElapsedTime)
295 				this.stopEmitter();
296 		}
297 		
298 		for(var p_index = 0; p_index < this.mMaxParticles; ++p_index)
299 		{
300 			var particle = this.mParticles[p_index];
301 			particle.mTimeToLive -= deltaTime;
302 			var timeLapse = ss2d.CURRENT_VIEW.mTotalTime - particle.mTimeStamp;
303 			particle.mRadius -= (particle.mRadiusDelta * deltaTime);
304 			
305 			//REPLACING SHADER FUNCTIONALITY
306 			
307 			
308 			
309 			if(particle.mTimeToLive <= 0.0 || (this.mEmitterType == 1.0 && particle.mRadius < this.mMinRadius))
310 			{
311 				this.mParticleCount--;
312 				this.mFreeParticles.push(p_index);
313 			}
314 		}
315 		
316 		//console.log(ss2d.CURRENT_VIEW.mTotalTime - this.mParticles[0].mTimeStamp);
317 	};
318 }
