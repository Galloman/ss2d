  3 /**
  4  * @fileoverview A frame for of a ss2d.Reel
  5  * @author David Gallardo Moreno (portalg@gmail.com)
  6  */
  7 
  8 goog.provide('ss2d.ReelFrame');
  9 
 10 goog.require('ss2d.ResourceLoader');
 11 
 12 /**
 13  * @constructor
 14  * @param {number} x The X texture coordinate
 15  * @param {number} y The Y texture coordinate
 16  * @param {number} width Width in texture pixels
 17  * @param {number} height Height in texture pixels
 18  * @param {(string|ss2d.Texture)} texture
 19  * @param {number=} ox Horizontal offset (optional)
 20  * @param {number=} oy Vertical offset (optional)
 21  * The offsets are used by the ss2d.SpriteReelPlayer to "center" the position of the figure.
 22  */
 23 ss2d.ReelFrame = function(x, y, width, height, texture, offsetX, offsetY)
 24 {
 25 	this.mX = x;
 26 	this.mY = y;
 27 	this.mWidth = width;
 28 	this.mHeight = height;
 29 	
 30 	this.mOffsetX = offsetX||0;
 31 	this.mOffsetY = offsetY||0;
 32 	
 33 	//reference to the texture for this frame
 34 	this.mTexture = (texture instanceof ss2d.Texture) ? texture : ss2d.ResourceLoader.loadTexture(texture);
 35 };
 36 
 37 /**
 38  * @param {Array=} clipArray array where the clip information will be dump
 39  * @return {number[4]} array for clipping the proper frame from the texture
 40  */
 41 ss2d.ReelFrame.prototype.dumpClip = function(clipArray)
 42 {
 43 	clipArray = clipArray || new Array(4);
 44 	
 45 	clipArray[0] = this.mX;
 46 	clipArray[1] = this.mY;
 47 	clipArray[2] = this.mWidth;
 48 	clipArray[3] = this.mHeight;
 49 	
 50 	return clipArray;
 51 };
